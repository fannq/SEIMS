00035 int PER_PI::Execute()
00036 {
00037     CheckInputData();
00038 
00039     if(m_perc == NULL) 
00040     {
00041         m_perc = new float*[m_nCells];
00042 
00043         #pragma omp parallel for
00044         for (int i = 0; i < m_nCells; i++)
00045         {
00046             m_perc[i] = new float[m_nSoilLayers];
00047             for (int j = 0; j < m_nSoilLayers; j++)
00048                 m_perc[i][j] = 0.f;
00049         }
00050     }
00051     
00052     #pragma omp parallel for
00053     for (int i = 0; i < m_nCells; i++)
00054     {
00055         float k, maxSoilWater, fcSoilWater, swater;
00056         float depth[2];
00057         depth[0] = m_upSoilDepth;
00058         depth[1] = m_rootDepth[i] - m_upSoilDepth;
00059         if(depth[1] < 0)
00060         {
00061             ostringstream oss;
00062             oss << "The root depth at cell(" << i << ") is " << m_rootDepth[i] << ", and is less than the upper soil depth (" << m_upSoilDepth << endl;
00063             throw ModelException("SSR_DA", "Execute",  oss.str());
00064         }
00065 
00066         m_sm[i][0] += m_infil[i] / depth[0];
00067         for (int j = 0; j < m_nSoilLayers; j++)
00068         {
00069             //No movement if soil moisture is below field capacity
00070             m_perc[i][j] = 0.f;
00071 
00072             // for the upper two layers, soil may be frozen
00073             if(j == 0 && m_soilT[i] <= m_frozenT) 
00074                 continue;
00075 
00076             if (m_sm[i][j] > m_fc[i][j])
00077             {
00078                 maxSoilWater = depth[j] * m_porosity[i][j];
00079                 swater = depth[j] * m_sm[i][j];
00080 
00081                 fcSoilWater = depth[j] * m_fc[i][j];
00082                 
00083                 //the moisture content can exceed the porosity in the way the algorithm is implemented
00084                 if (m_sm[i][j] > m_porosity[i][j])
00085                     k = m_ks[i][j];
00086                 else
00087                 {
00088                     float dcIndex = 2.f/m_poreIndex[i][j] + 3.f; // pore disconnectedness index
00089                     k = m_ks[i][j] * pow(m_sm[i][j]/m_porosity[i][j], dcIndex);
00090                 }
00091 
00092                 m_perc[i][j] = k * m_dt/3600.f;
00093 
00094                 if(swater - m_perc[i][j] < fcSoilWater)
00095                     m_perc[i][j] = swater - fcSoilWater;
00096                 else if(swater - m_perc[i][j] > maxSoilWater)
00097                     m_perc[i][j] = swater - maxSoilWater;
00098 
00099                 //Adjust the moisture content in the current layer, and the layer immediately below it
00100                 m_sm[i][j] -= m_perc[i][j]/depth[j];
00101 
00102                 if(j < m_nSoilLayers-1)
00103                     m_sm[i][j+1] += m_perc[i][j]/depth[j+1];
00104 
00105 
00106                 if(m_sm[i][j] != m_sm[i][j] || m_sm[i][j] < 0.f)
00107                 {
00108                     cout << "PER_PI CELL:" << i << "\tPerco:" << swater << "\t" << 
00109                         fcSoilWater << "\t" << m_perc[i][j] << "\t" << depth[j] << "\tValue:" << m_sm[i][j] << endl;
00110                     throw ModelException("PER_PI", "Execute", "moisture is less than zero.");
00111                 }
00112             }
00113         }
00114     }
00115     return 0;
00116
