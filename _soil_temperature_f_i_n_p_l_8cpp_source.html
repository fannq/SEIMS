00034 int SoilTemperatureFINPL::Execute()
00035 {
00036     //check the data
00037     CheckInputData();   
00038     if(this->m_soilTemp == NULL)
00039     {
00040         this->m_soilTemp = new float[this->m_size];
00041         this->m_t1 = new float[this->m_size];
00042         this->m_t2 = new float[this->m_size];
00043 
00044         #pragma omp parallel for
00045         for (int i=0; i<m_size; ++i)
00046         {
00047             m_t1[i] = (m_tMax[i] + m_tMin[i])/2;
00048             m_t2[i] = m_t1[i];
00049         }       
00050     }
00051 
00052     #pragma omp parallel for
00053     for (int i=0; i<m_size; ++i)
00054     {
00055         float t = (m_tMax[i] + m_tMin[i])/2;
00056         float t1 = m_t1[i];
00057         float t2 = m_t2[i];
00058         float t10 = m_a0 + m_a1*t2 + m_a2*t1 + m_a3*t 
00059             + m_b1*sin(w*m_julianDay) + m_d1*cos(w*m_julianDay) 
00060             + m_b2*sin(2*w*m_julianDay) + m_d2*cos(2*w*m_julianDay);
00061         m_soilTemp[i] = t10*m_kSoil10 + m_relativeFactor[i];
00062         if (m_soilTemp[i] > 60.0f || m_soilTemp[i] < -90.0f)
00063         {
00064             ostringstream oss;
00065             oss << "The calculated soil temperature at cell (" << i
00066             << ") is out of reasonable range, whose value is " << m_soilTemp[i] << ".\nJulianDay: " 
00067             << m_julianDay << endl;
00068             throw ModelException("STP_FP","Execute", oss.str());
00069         }
00070 
00071         //save the temperature
00072         m_t2[i] = m_t1[i];
00073         m_t1[i] = t;
00074     }
00075     
00076     return 0;
00077 }
