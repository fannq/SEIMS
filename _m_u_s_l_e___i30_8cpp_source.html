00133 {
00134     float p = this->m_p[cell];
00135     float snomlt = this->m_snowMelt[cell];

00137     else            p = 0.0f;
00138

00140     localtime_s(&datestruture,&m_Date);         //get month
00141     float max = 1.0f - exp(-125.0f / (p + 5));  //eq.1:3.2.3 p66
00142     float a15 = triangularDistribution(0.02083f,m_alpha_month[datestruture.tm_mon],max,&m_rndseed); //eq.1:3.2.4-5 p67
