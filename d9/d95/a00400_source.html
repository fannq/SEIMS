152 void SettingsInput::ReadSiteList()
153 {
154  bson query[1];
155  bson_init(query);
156 
157  // subbasin id
158  bson_append_int(query, Tag_SubbasinId, m_subbasinID);
159  // mode
160  bson_append_string(query, Tag_Mode, m_mode.c_str());
161 
162  bson_finish(query);
163  //bson_print(query);
164 
165  string siteListTable = SITELIST_TABLE;
166  bool stormMode = false;
167  if (StringMatch(m_mode, Tag_Mode_Storm))
168  stormMode = true;
169  mongo_cursor cursor[1];
170  ostringstream oss;
171  oss << m_dbName << "." << siteListTable;
172  mongo_cursor_init(cursor, m_conn, oss.str().c_str());
173  mongo_cursor_set_query(cursor, query);
174 
175  // loop stations of one class of climate data, such as "P"
176  while( mongo_cursor_next(cursor) == MONGO_OK )
177  {
178  const bson* stationInfo = mongo_cursor_bson(cursor);
179  bson_iterator it[1];
180 
181  if (bson_find(it, stationInfo, "DB"))
182  m_dbHydro = bson_iterator_string(it);
183  else
184  throw ModelException("SettingsInput", "ReadSiteList", "The DB field does not exist in SiteList table.");
185 
186  if (bson_find(it, stationInfo, SITELIST_TABLE_M))
187  {
188  string siteList = bson_iterator_string(it);
189  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_MeanTemperature, m_startDate, m_endDate);
190  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_MaximumTemperature, m_startDate, m_endDate);
191  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_MinimumTemperature, m_startDate, m_endDate);
192  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_WindSpeed, m_startDate, m_endDate);
193  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_SolarRadiation, m_startDate, m_endDate);
194  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_RelativeAirMoisture, m_startDate, m_endDate);
195  }
196 
197  if (bson_find(it, stationInfo, SITELIST_TABLE_P))
198  m_inputStation->ReadSitesData(m_dbHydro, bson_iterator_string(it), DataType_Precipitation, m_startDate, m_endDate, stormMode);
199 
200  if (bson_find(it, stationInfo, SITELIST_TABLE_PET))
201  m_inputStation->ReadSitesData(m_dbHydro, bson_iterator_string(it), DataType_PotentialEvapotranspiration, m_startDate, m_endDate, stormMode);
202 
203  }
204 
205  bson_destroy(query);
206  mongo_cursor_destroy(cursor);
