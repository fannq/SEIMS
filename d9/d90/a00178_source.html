151  sum2 += (v_vwgt[i]+u2_vwgt[i])*invtvwgt[i];
152  }
153  sum1 = sum1/ncon;
154  sum2 = sum2/ncon;
155 
156  for (i=0; i<ncon; i++) {
157  diff1 += rabs(sum1 - (v_vwgt[i]+u1_vwgt[i])*invtvwgt[i]);
158  diff2 += rabs(sum2 - (v_vwgt[i]+u2_vwgt[i])*invtvwgt[i]);
159  }
160 
161  return (diff1 - diff2 >= 0);
162 }
