65 int SSM_PE::Execute()
66 {
67  this->CheckInputData();
68 
69  this->initalOutputs();
70  if (m_SR == NULL)
71  {
72  m_SR = new float[m_nCells];
73  for(int i=0;i<this->m_nCells;i++)
74  m_SR[i] = 0.f;
75  }
76  if(m_isInitial)
77  {
78  int count = 0;
79  for(int i=0;i<this->m_nCells;i++)
80  {
81  if(this->m_tMean[i] < this->m_tsnow) {this->m_SA[i] = this->m_swe0; count++;} //winter
82  else this->m_SA[i] = 0.0f; // other seasons
83  }
84 
85  m_swe = this->m_swe0 * count / this->m_nCells;
86  m_isInitial = false;
87  }
88 
89  if(this->m_lastSWE == NODATA) this->m_lastSWE = this->m_swe;
90  if(m_swe < 0.01) //all cells have not snow, so snow sublimation is 0.
91  {
92  if(this->m_lastSWE >= 0.01)
93  {
94  for (int rw = 0; rw < this->m_nCells; rw++)
95  m_SE[rw] = 0.0f;
96  }
97  this->m_lastSWE = this->m_swe;
98  return true;
99  }
100 
101  for (int rw = 0; rw < this->m_nCells; rw++)
102  {
103  float snow = this->m_SA[rw] + this->m_SR[rw];
104  float tmean = this->m_tMean[rw];
105  if(tmean < this->m_tsnow) snow+=(1+this->m_kblow)*this->m_Pnet[rw];
106  if(snow < 0.01) this->m_SE[rw] = 0.0f;
107  else
108  {
109  if(tmean >= this->m_t0) this->m_SE[rw] = 0.0f; //if temperature is higher than t0, the sublimation is 0.
110  else
111  {
112  float se = this->m_ksubli * this->m_PET[rw];
113  this->m_SE[rw] = min(snow,se);
114  }
115  }
116  }
117  this->m_lastSWE = this->m_swe;
118  return true;
119  return 0;
120 }
