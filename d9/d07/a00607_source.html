233 int StormGreenAmpt::Execute(void)
234 {
235  initalOutputs();
236 
237 
238  // allocate intermediate variables
239  if (m_capillarySuction == NULL)
240  {
241  m_capillarySuction = new float[m_nCells];
242  m_accumuDepth = new float[m_nCells];
243 
244  #pragma omp parallel for
245  for (int i = 0; i < m_nCells; ++i)
246  {
247  m_accumuDepth[i] = 0.0f;
248  m_capillarySuction[i] = CalculateCapillarySuction(m_porosity[i], m_clay[i]*100, m_sand[i]*100);
249  }
250  }
251 
252 #pragma omp parallel for
253  for (int i = 0; i < m_nCells; i++)
254  {
255  float t = 10.f;
256  if (m_tMax != NULL && m_tMin != NULL)
257  t = (m_tMax[i] + m_tMin[i]) / 2;
258  float snowMelt = 0.f;
259  float snowAcc = 0.f;
260  if (m_snowMelt != NULL)
261  snowMelt = m_snowMelt[i];
262  if (m_snowAccu != NULL)
263  snowAcc = m_snowAccu[i];
264 
265  float hWater = m_pNet[i];
266  //account for the effects of snow melt and soil temperature
267  // snow, without snow melt
268  if (t <= m_tSnow)
269  {
270  hWater = 0.0f;
271  }
272  // rain on snow, no snow melt
273  else if (t > m_tSnow && t <= m_t0 && snowAcc > hWater)
274  {
275  hWater = 0.0f;
276  }
277  else
278  {
279  hWater = m_pNet[i] + m_sd[i] + snowMelt;
280  }
281 
282  hWater += m_sr[i];
283 
284  // effective matric potential (m)
285  float matricPotential = (m_porosity[i] - m_initSoilMoisture[i]) * m_capillarySuction[i] / 1000.f;
286  // algorithm of Li, 1996, uesd in C2SC2D
287  float ks = m_ks[i] / 1000.f / 3600.f; // mm/h -> m/s
288  float dt = m_dt;
289  float infilDepth = m_accumuDepth[i] / 1000.f; // mm ->m
290 
291  float p1 = (float)(ks*dt - 2.0*infilDepth);
292  float p2 = ks*(infilDepth + matricPotential);
293  // infiltration rate (m/s)
294  float infilRate = (float)((p1 + sqrt(pow(p1,2.0f) + 8.0f*p2*dt))/(2.0f*dt));
295 
296  float infilCap = (m_porosity[i] - m_soilMoisture[i])*m_rootDepth[i];
297 
298  if (hWater > 0)
299  {
300  // for frozen soil
301  //if (m_soilTemp[i] <= m_tSoilFrozen && m_soilMoisture[i] >= m_sFrozen*m_porosity[i])
302  //{
303  // m_pe[i] = pNet;
304  // m_infil[i] = 0.0f;
305  //}
306  //for saturation overland flow
307  if (m_soilMoisture[i] > m_porosity[i])
308  {
309  m_infil[i] = 0.0f;
310  m_infilCapacitySurplus[i] = 0.f;
311  }
312  else
313  {
314  m_infil[i] = min(infilRate*dt*1000.f, infilCap); // mm
315 
316  //cout << m_infil[i] << endl;
317  //check if the infiltration potential exceeds the available water
318  if(m_infil[i] > hWater)
319  {
320  m_infilCapacitySurplus[i] = m_infil[i] - hWater;
321  //limit infiltration rate to available water supply
322  m_infil[i] = hWater;
323  }
324  else
325  {
326  m_infilCapacitySurplus[i] = 0.f;
327  }
328 
329  //Compute the cumulative depth of infitration
330  m_accumuDepth[i] += m_infil[i];
331  m_sr[i] = hWater - m_infil[i]; // sr is temporarily used to stored the water depth including the depression storage
332 
333  if (m_rootDepth != NULL)
334  m_soilMoisture[i] += m_infil[i]/m_rootDepth[i];
335  }
336  }
337  else
338  {
339  m_sr[i] = 0.0f;
340  m_infil[i] = 0.0f;
341  m_infilCapacitySurplus[i] = min(infilRate*dt*1000.f, infilCap);
342  }
343  }
344 
345  return 0;
346 }

349 float StormGreenAmpt::CalculateCapillarySuction(float por, float clay, float sand)
350 {
351  float cs = 10.0f * exp(6.5309f - 7.32561f*por + 0.001583f*pow(clay, 2) + 3.809479f*pow(por, 2)
352  + 0.000344f*sand*clay - 0.049837f*por*sand + 0.001608f*pow(por, 2)*pow(sand, 2)
353  + 0.001602f*pow(por, 2)*pow(clay, 2) - 0.0000136f*pow(sand,2)*clay - 0.003479f*pow(clay, 2)*por
354  - 0.000799f*pow(sand, 2)*por);
355 
356  return cs;
357 }
