122 int AtmosphericDeposition::Execute() {
123  //check the data
124  CheckInputData();
125  for(int i = 0; i < m_nCells; i++) {
126  for(int k = 0; k < m_nSolLyrs; k++) {
127  // Calculate the amount of nitrite and ammonia added to the soil in rainfall,
128  m_addrno3 = 0.01 * m_rcn * m_preci[i];
129  m_addrnh3 = 0.01 * m_rca * m_preci[i];
130  m_sol_no3[i][0] = m_sol_no3[i][0] + m_addrno3 + m_drydep_no3 * 365;
131  m_sol_nh3[i][0] = m_sol_nh3[i][0] + m_addrnh3 + m_drydep_nh4 * 365;
132 
133  m_wshd_rno3 = m_wshd_rno3 + m_addrno3 * (1 / m_cellWidth);
134  }
135  }
136  return 0;
137 }
