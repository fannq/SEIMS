409 float DissolvedNutrient_CH::simpleSedCalc(float Qn, float Qin, float Sin, float dt, float vol, float sed)
410 {

421 void DissolvedNutrient_CH::GetNutrientInFlow(int iReach, int iCell, int id)
422 {
423  //sum the upstream overland flow
424  float Qin = 0.f;
425  float PUp = 0.0f;
426  float NH4Up = 0.0f;
427  float NO3Up = 0.0f;
428  if (iCell == 0)// inflow of this cell is the last cell of the upstream reach
429  {
430  for (int i = 0; i < m_reachUpStream[iReach].size(); ++i)
431  {
432  int upReachId = m_reachUpStream[iReach][i];
433  if (upReachId >= 0)
434  {
435  int upCellsNum = m_reachs[upReachId].size();
436  int upCellId = m_reachs[upReachId][upCellsNum-1];
437  Qin += m_ChQkin[upReachId][upCellsNum-1];
438  PUp += m_DissovP[upReachId][upCellsNum-1];
439  NH4Up += m_Ammonium[upReachId][upCellsNum-1];
440  NO3Up += m_Nitrate[upReachId][upCellsNum-1];
441  }
442  }
443  }
444  else
445  {
446  Qin = m_ChQkin[iReach][iCell-1];
447  PUp = m_DissovP[iReach][iCell-1]; //the iCell-1 is the prior of the iCell
448  NH4Up = m_Ammonium[iReach][iCell-1];
449  NO3Up = m_Nitrate[iReach][iCell-1];
450  }
451 
452  this->WaterVolumeCalc(iReach, iCell, id);
453  float WtVol = m_ChVol[iReach][iCell];
454 
455  m_DissovP_KG[iReach][iCell] = m_DissovPToCh[id];
456  m_Ammonium_KG[iReach][iCell] = m_AmmoniumToCh[id];
457  m_Nitrate_KG[iReach][iCell] = m_NitrateToCh[id];
458 
459  m_DissovP[iReach][iCell] = simpleSedCalc(m_ChQkin[iReach][iCell], Qin, PUp, m_dt, WtVol, m_DissovP_KG[iReach][iCell]);
460  m_Ammonium[iReach][iCell] = simpleSedCalc(m_ChQkin[iReach][iCell], Qin, NH4Up, m_dt, WtVol, m_Ammonium_KG[iReach][iCell]);
461  m_Nitrate[iReach][iCell] = simpleSedCalc(m_ChQkin[iReach][iCell], Qin, NO3Up, m_dt, WtVol, m_Nitrate_KG[iReach][iCell]);
462 
463  float tem = PUp + m_DissovP_KG[iReach][iCell]/m_dt; //kg/s
464  // no more dissovP outflow than total dissovP in cell
465  m_DissovP[iReach][iCell] = min(m_DissovP[iReach][iCell], tem);
466  //m_routQs[id] = m_Qsn[iReach][iCell];
467  tem = PUp * m_dt + m_DissovP_KG[iReach][iCell] - m_DissovP[iReach][iCell] * m_dt;
468  // new dissovP volume based on all fluxes or dissovP present
469  m_DissovP_KG[iReach][iCell] = max(0.0f, tem);
470 
471  tem = NH4Up + m_Ammonium_KG[iReach][iCell]/m_dt;
472  m_Ammonium[iReach][iCell] = min(m_Ammonium[iReach][iCell], tem);
473  tem = NH4Up * m_dt + m_Ammonium_KG[iReach][iCell] - m_Ammonium[iReach][iCell] * m_dt;
474  m_Ammonium_KG[iReach][iCell] = max(0.0f, tem);
475 
476  tem = NO3Up + m_Nitrate_KG[iReach][iCell]/m_dt;
477  m_Nitrate[iReach][iCell] = min(m_Nitrate[iReach][iCell], tem);
478  tem = NO3Up * m_dt + m_Nitrate_KG[iReach][iCell] - m_Nitrate[iReach][iCell] * m_dt;
479  m_Nitrate_KG[iReach][iCell] = max(0.0f, tem);
480 }

482 int DissolvedNutrient_CH::Execute()
483 {
484  this->CheckInputData();
485 
486  initalOutputs();
487  map<int, vector<int> >::iterator it;
488  for (it = m_reachLayers.begin(); it != m_reachLayers.end(); it++)
489  {
490  // There are not any flow relationship within each routing layer.
491  // So parallelization can be done here.
492  int nReaches = it->second.size();
493  // the size of m_reachLayers (map) is equal to the maximum stream order
494 #pragma omp parallel for

498  vector<int>& vecCells = m_reachs[reachIndex];
499  int n = vecCells.size();
500  for (int iCell = 0; iCell < n; ++iCell)
501  {
502  GetNutrientInFlow(reachIndex, iCell, vecCells[iCell]);
503  }
504  }
505  }
506 
507  return 0;
