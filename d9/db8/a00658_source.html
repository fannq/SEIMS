100 int SUR_MR::Execute()
101 {
102  CheckInputData();
103  initalOutputs();
104  m_pMax = m_pMax * m_dt / 86400.f;
105  if(m_upSoilDepth == NULL)
106  m_upSoilDepth = new float[m_nSoilLayers];
107 #pragma omp parallel for
108  for (int i = 0; i < m_nCells; i++)
109  {
110  //account for the effects of snow melt and soil temperature
111  float t = 0.f, snowMelt = 0.f, snowAcc = 0.f, sd = 0.f;
112  if (m_tMean != NULL) t = m_tMean[i];
113  if (m_snowMelt != NULL)snowMelt = m_snowMelt[i];
114  if (m_snowAccu != NULL)snowAcc = m_snowAccu[i];
115  if (m_sd != NULL)sd = m_sd[i];
116 
117  float hWater = 0.f;
118  // snow, without snow melt
119  if (t <= m_tSnow)
120  hWater = 0.0f;
121  // rain on snow, no snow melt
122  else if (t > m_tSnow && t <= m_t0 && snowAcc > m_pNet[i])
123  hWater = 0.0f;
124  else
125  hWater = m_pNet[i] + snowMelt + sd;
126 
127  if(hWater > 0.f)
128  {
130  int curSoilLayers = -1, j;
131  m_upSoilDepth[0] = m_soilDepth[i][0];
132  for(j = 1; j < m_nSoilLayers; j++)
133  {
134  if(!FloatEqual(m_soilDepth[i][j],NODATA_VALUE))
135  m_upSoilDepth[j] = m_soilDepth[i][j] - m_soilDepth[i][j-1];
136  else
137  break;
138  }
139  curSoilLayers = j;
140  float sm = 0.f, por = 0.f;
141  for (int j = 0; j < curSoilLayers; j++)
142  {
143  sm += m_soilMoisture[i][j] * m_upSoilDepth[j];
144  por += m_porosity[i][j] * m_upSoilDepth[j];
145  }
146 
147  float smFraction = sm / por;
148  smFraction = min(smFraction, 1.0f);
149 
150  // for frozen soil
151  if (m_soilTemp != NULL && m_soilTemp[i] <= m_tFrozen && smFraction >= m_sFrozen)
152  {
153  m_pe[i] = m_pNet[i] + snowMelt;
154  m_infil[i] = 0.0f;
155  }
156  //else if (m_soilMoisture[i] >= m_porosity[i])//for saturation overland flow
157  //{
158  // m_pe[i] = m_pNet[i] + snowMelt;
159  // m_infil[i] = 0.0f;
160  //}
161  else
162  {
163  float alpha = m_kRunoff - (m_kRunoff - 1) * hWater / m_pMax;
164  if (hWater >= m_pMax)
165  alpha = 1.0f;
166 
167  //runoff percentage
168  float runoffPercentage;
169  if (m_runoffCo[i] > 0.99f)
170  runoffPercentage = 1.0f;
171  else
172  runoffPercentage = m_runoffCo[i] * pow(smFraction, alpha);
173 
174  float surfq = hWater * runoffPercentage;
175 
176  m_infil[i] = hWater - surfq;
177  m_pe[i] = surfq;
178 
179  // check the output data
180  if (m_infil[i] != m_infil[i] || m_infil[i] < 0.0f)
181  {
182  //string datestr = getDate(&m_date);
183  ostringstream oss;
184  oss << "Cell id:" << i << "\tPrecipitation(mm) = " << m_pNet[i] << "\thwater = " << hWater
185  << "\tpercentage:" << runoffPercentage << "\tmoisture = " << sm
186  << "\tInfiltration(mm) = " << m_infil[i] << "\n";
187  throw ModelException(MID_SUR_MR,"Execute","Output data error: infiltration is less than zero. Where:\n"
188  + oss.str() + "Please contact the module developer. ");
189  }
190  }
191  }
192  else
193  {
194  m_pe[i] = 0.0f;
195  m_infil[i] = 0.0f;
196  }
197  }
198  return 0;
199 }
