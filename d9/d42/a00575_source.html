152 int PETPenmanMonteith::Execute()
153 {
154  //*****************************************************
155  //Now, all the grow codes are constant and equle to 1
156  if(this->m_growCode == NULL && this->m_size>0)
157  {
158  this->m_growCode = new float[this->m_size];
159  this->m_cht = new float[this->m_size];
160  this->m_lai = new float[this->m_size];
161  this->m_pet = new float[this->m_size];
162 
163  #pragma omp parallel for
164  for(int i = 0; i<m_size ; i++)
165  {
166  this->m_growCode[i] = 1;
167  this->m_cht[i] = 1.5f;
168  this->m_lai[i] = 0.95f;
169  m_pet[i] = 0.f;
170  }
171  }
172 
173  //check the data
174  if(!this->CheckInputData()) return false;
175 
176  int d = JulianDay(this->m_date);
177  //do the execute
178  #pragma omp parallel for
179  for (int j = 0; j < m_size; ++j)
180  {
182  //compute net radiation
183  //this->StatusMsg("compute net radiation ...");
184 
185  //net short-wave radiation for PET(from swat)
186  float raShortWave = m_sr[j] * (1.0f - 0.23f);
187  float tMean = (m_tMax[j] + m_tMin[j]) / 2;
188 
189  //*********************************************************
190  //if(m_snow[j] > 0.5f)
191  if(tMean < this->m_tSnow) //if the mean t < snow t, consider the snow depth is larger than 0.5mm.
192  raShortWave = m_sr[j] * (1.0f - 0.8f);
193  //*********************************************************
194 
195  //calculate the max solar radiation
196  float srMax = this->MaxSolarRadiation(d,this->m_latitude[j]);
197 
198  //calculate net long-wave radiation
199  //net emissivity equation 2.2.20 in SWAT manual
200  float satVaporPressure = SaturationVaporPressure(tMean);//kPa
201  float actualVaporPressure = m_rhd[j] * satVaporPressure; //why *0.01?
202  float rbo = -(0.34f - 0.139f * sqrt(actualVaporPressure)); //P37 1:1.2.22
203  //cloud cover factor equation 2.2.19
204  float rto = 0.0f;
205  if (srMax >= 1.0e-4)
206  rto = 0.9f * (m_sr[j] / srMax) + 0.1f;
207  //net long-wave radiation equation 2.2.21
208  float tk = tMean + 273.15f;
209  float raLongWave = rbo * rto * 4.9e-9f * pow(tk, 4);
210 
211  float raNet = raShortWave + raLongWave; //Hnet
212 
214  //calculate the slope of the saturation vapor pressure curve
215  //this->StatusMsg("calculate the slope of the saturation vapor pressure curve ...");
216 
217  float dlt = 4098.f* satVaporPressure / pow((tMean + 237.3f), 2);
218 
219  //calculate latent heat of vaporization(MJ/kg, from swat)
220  float latentHeat = 2.501f - 0.002361f * tMean;
221 
222  //calculate mean barometric pressure(kPa)
223  float pb = 101.3f - m_elev[j] * (0.01152f - 0.544e-6f * m_elev[j]); //p53 1:2.3.8 P
224  //psychrometric constant(kPa/deg C)
225  float gma = 1.013e-3f * pb / (0.622f * latentHeat);//p53 1:2.3.7
226 
227  float rho = 1710.f - 6.85f * tMean;//p127 2:2.2.19
228  //aerodynamic resistance to sensible heat and vapor transfer(s/m)
229  float rv = 114.f / (m_ws[j] * pow(170.f/1000.f, 0.2f)); // P127 2:2.2.20 ??the wind speed need modified?
230  //canopy resistance(s/m)
231  float rc = 49.f / (1.4f - 0.4f * m_co2 / 330.f); //P128 2:2.2.22
232  //vapor pressure deficit(kPa)
233  float vpd = satVaporPressure - actualVaporPressure;
234  float petValue = (dlt * raNet + gma * rho * vpd / rv) /
235  (latentHeat * (dlt + gma * (1.f + rc / rv))); //P122 2:2.2.2
236  petValue = max(0.0f, petValue);
237 
238  //*********************************************************
239  //The albedo would be obtained from plant growth module. But now it is assumed to be a constant.
240  //After the plant growth module is completed, the following codes shouble be removed.
241  float albedo = 0.8f;
242  if(tMean > this->m_tSnow) albedo = 0.23f;
243  //*********************************************************
244 
245  float raShortWavePlant = m_sr[j] * (1.0f -albedo);
246  float raNetPlant = raShortWavePlant + raLongWave;
247 
248  float epMax = 0.0f;
249  if (m_growCode[j] > 0)//land cover growing
250  {
251  //determine wind speed and height of wind speed measurement
252  //adjust to 100 cm (1 m) above canopy if necessary
253  float zz = 0.f; //height at which wind speed is determined(cm)
254  if (m_cht[j] <= 1.0f)
255  zz = 170.0f;
256  else
257  zz = m_cht[j] * 100.f + 100.f;
258  //wind speed at height zz(m/s)
259  float uzz = m_ws[j] * pow(zz/1000.f, 0.2f);
260 
261  //calculate canopy height in cm
262  float chz = 0.0f;
263  if (m_cht[j] < 0.01f)
264  chz = 1.0f;
265  else
266  chz = m_cht[j] * 100.0f;
267 
268  //calculate roughness length for momentum transfer
269  float zom = 0.0f;
270  if (chz <= 200.0f)
271  zom = 0.123f * chz;
272  else
273  zom = 0.058f * pow(chz, 1.19f);
274 
275  //calculate roughness length for vapor transfer
276  float zov = 0.1f * zom;
277 
278  //calculate zero-plane displacement of wind profile
279  float d = 0.667f * chz;
280 
281  //calculate aerodynamic resistance
282  float rv = log((zz - d) / zom) * log((zz - d) / zov);
283  rv = rv / (pow(0.41f,2) * uzz);
284 
285  //adjust stomatal conductivity for low vapor pressure
286  //this adjustment will lower maximum plant ET for plants
287  //sensitive to very low vapor pressure
288  float xx = vpd - 1.0f;//difference between vpd and vpthreshold
289  //amount of vapro pressure deficit over threshold value
290  float fvpd = 1.0f;
291  if (xx > 0.0f)
292  fvpd = max(0.1f, 1.0f - m_vpd2 * xx);
293  //fvpd = max(0.1f, 1.0 - vpd2(idplt(nro(j),icr(j),j)) * xx);
294  float gsi_adj = m_gsi * fvpd;
295  //gsi(idplt(nro(j),icr(j),j)) * fvpd;
296 
297  //calculate canopy resistance
298  float rc = 1.0f / gsi_adj; //single leaf resistance
299  //rc = rc / (0.5f * (laiday(j) + 0.01) * (1.4 - 0.4 * co2(hru_sub(j)) / 330.));
300  rc = rc / (0.5f * (m_lai[j] + 0.01f) * (1.4f - 0.4f * m_co2/330.f));
301 
302  //calculate maximum plant ET
303  epMax = (dlt * raNetPlant + gma * rho * vpd / rv) / (latentHeat * (dlt + gma * (1.f + rc / rv)));
304  }
305  epMax = max(0.0f, epMax);
306 
307  m_pet[j] = m_petFactor * min(petValue, epMax);
308  }
309 
310  return 0;
311 }

325 float PETPenmanMonteith::MaxSolarRadiation(int day,float lat)
326 {
327  lat = lat*3.1415926/180;
328  //Calculate Daylength
329  //calculate solar declination: equation 2.1.2 in SWAT manual
330  float sd = asin(0.4f * sin((day - 82.0f) / 58.09f)); //365/2pi = 58.09
331 
332  //calculate the relative distance of the earth from the sun
333  //the eccentricity of the orbit
334  //equation 2.1.1 in SWAT manual
335  float dd = 1.0f + 0.033f * cos(day / 58.09f);
336 
337  //daylength = 2 * Acos(-Tan(sd) * Tan(lat)) / omega
338  //where the angular velocity of the earth's rotation, omega, is equal
339  //to 15 deg/hr or 0.2618 rad/hr and 2/0.2618 = 7.6374
340  //equation 2.1.6 in SWAT manual
341 
342  float h = 0.0f;
343  float ch = -sin(lat) * tan(sd) / cos(lat);
344  if (ch > 1.f) //ch will be >= 1. if latitude exceeds +/- 66.5 deg in winter
345  h = 0.0f;
346  else if (ch >= -1.0f)
347  h = acos(ch);
348  else
349  h = 3.1416f; //latitude exceeds +/- 66.5 deg in summer
350 
351  float dayl = 7.6394f * h;
352 
353  //Calculate Potential (maximum) Radiation !!
354  //equation 2.2.7 in SWAT manual
355  float ys = sin(lat) * sin(sd);
356  float yc = cos(lat) * cos(sd);
357 
358  return 30.f * dd * (h * ys + yc * sin(h));
359 }
