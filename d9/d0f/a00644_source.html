34 int SoilTemperatureFINPL::Execute()
35 {
36  //check the data
37  CheckInputData();
38  if(this->m_soilTemp == NULL)
39  {
40  this->m_soilTemp = new float[this->m_size];
41  this->m_t1 = new float[this->m_size];
42  this->m_t2 = new float[this->m_size];
43 
44  #pragma omp parallel for
45  for (int i=0; i<m_size; ++i)
46  {
47  m_t1[i] = (m_tMax[i] + m_tMin[i])/2;
48  m_t2[i] = m_t1[i];
49  }
50  }
51 
52  #pragma omp parallel for
53  for (int i=0; i<m_size; ++i)
54  {
55  float t = (m_tMax[i] + m_tMin[i])/2;
56  float t1 = m_t1[i];
57  float t2 = m_t2[i];
58  float t10 = m_a0 + m_a1*t2 + m_a2*t1 + m_a3*t
59  + m_b1*sin(w*m_julianDay) + m_d1*cos(w*m_julianDay)
60  + m_b2*sin(2*w*m_julianDay) + m_d2*cos(2*w*m_julianDay);
61  m_soilTemp[i] = t10*m_kSoil10 + m_relativeFactor[i];
62  if (m_soilTemp[i] > 60.0f || m_soilTemp[i] < -90.0f)
63  {
64  ostringstream oss;
65  oss << "The calculated soil temperature at cell (" << i
66  << ") is out of reasonable range, whose value is " << m_soilTemp[i] << ".\nJulianDay: "
67  << m_julianDay << endl;
68  throw ModelException("STP_FP","Execute", oss.str());
69  }
70 
71  //save the temperature
72  m_t2[i] = m_t1[i];
73  m_t1[i] = t;
74  }
75 
76  return 0;
77 }
