113 void ImplicitKinematicWave::initalOutputs()
114 {
115  if(m_nCells <= 0) throw ModelException("IKW_OL","initalOutputs","The cell number of the input can not be less than zero.");
116 
117  if(m_q == NULL)
118  {
119  CheckInputData();
120  m_q = new float[m_nCells];
121 
122  m_sRadian = new float[m_nCells];
123  m_vel = new float[m_nCells];
124  m_flowWidth = new float[m_nCells];
125  m_flowLen = new float[m_nCells];
126  m_alpha = new float[m_nCells];
127  m_reInfil = new float[m_nCells];
128 
129  for (int i = 0; i < m_nCells; ++i)
130  {
131  m_q[i] = 0.0f;
132  m_reInfil[i] = 0.f;
133 
134  // flow width
135  m_flowWidth[i] = m_CellWidth;
136  int dir = (int)m_direction[i];
137  if (m_diagonal[dir] == 1)
138  m_flowWidth[i] = m_CellWidth/SQ2;
139  if(m_streamLink[i] > 0)
140  m_flowWidth[i] -= m_chWidth[i];
141 
142  float s0 = m_s0[i];
143  if (FloatEqual(s0, 0.0f))
144  s0 = MINI_SLOPE;
145  m_sRadian[i] = atan(s0);
146 
147  // flow length needs to be corrected by slope angle
148  float dx = m_CellWidth / cos(m_sRadian[i]);
149  if (m_diagonal[dir] == 1)
150  dx = SQ2*dx;
151  m_flowLen[i] = dx;
152  }
153  }
154 }

167 float ImplicitKinematicWave::GetNewQ(float qIn, float qLast, float surplus, float alpha, float dt, float dx)
168 {
169  /* Using Newton-Raphson Method */
170  float ab_pQ, dtX, C; //auxillary vars
171  int count;
172  float Qkx; //iterated discharge, becomes Qnew
173  float fQkx; //function
174  float dfQkx; //derivative
175  const float _epsilon = 1e-12f;
176  const float beta = 0.6f;
177 
178  /* if no input then output = 0 */
179  if ((qIn + qLast) <= -surplus*dx)//0)
180  {
181  //itercount = -1;
182  return(0);
183  }
184 
185  /* common terms */
186  ab_pQ = alpha*beta*pow(((qLast+qIn)/2),beta-1);
187  // derivative of diagonal average (space-time)
188 
189  dtX = dt/dx;
190  C = dtX*qIn + alpha*pow(qLast,beta) + dt*surplus;
191  //dt/dx*Q = m3/s*s/m=m2; a*Q^b = A = m2; surplus*dt = s*m2/s = m2
192  //C is unit volume of water
193  // first gues Qkx
194  Qkx = (dtX * qIn + qLast * ab_pQ + dt * surplus) / (dtX + ab_pQ);
195 
196  // VJ 050704, 060830 infil so big all flux is gone
197  //VJ 110114 without this de iteration cannot be solved for very small values
198  if (Qkx < MIN_FLUX)
199  {
200  //itercount = -2;
201  return(0);
202  }
203 
204  Qkx = max(Qkx, MIN_FLUX);
205 
206  count = 0;
207  do {
208  fQkx = dtX * Qkx + alpha * pow(Qkx, beta) - C; /* Current k */
209  dfQkx = dtX + alpha * beta * pow(Qkx, beta - 1); /* Current k */
210  Qkx -= fQkx / dfQkx; /* Next k */
211  Qkx = max(Qkx, MIN_FLUX);
212  count++;
213  //qDebug() << count << fQkx << Qkx;
214  } while(fabs(fQkx) > _epsilon && count < MAX_ITERS);
215 
216  if(Qkx != Qkx)
217  throw ModelException("IKW_OL", "GetNewQ", "Error in iteration!");
218 
219  //itercount = count;
220  return Qkx;
221 }

225 void ImplicitKinematicWave::OverlandFlow(int id)
226 {
227  const float beta = 0.6f;
228  const float _23 = 2.0f/3.0f;
229  float beta1 = 1/beta;
230 
231  float h = m_sr[id]/1000.f;
232 
233  float Perim = 2.f*h + m_flowWidth[id];
234  float r = 0;
235  if (Perim > 0)
236  r = h*m_flowWidth[id]/Perim;
237 
238  float sSin = sqrt(sin(m_sRadian[id]));
239  m_alpha[id] = pow(m_n[id]/sSin * pow(Perim, _23),beta);
240 
241  if (m_alpha[id] > 0)
242  m_q[id] = pow((m_flowWidth[id]*h)/m_alpha[id], beta1);
243  else
244  m_q[id] = 0;
245 
246  m_vel[id] = pow(r, _23)*sSin/m_n[id];
247 
248  float flowWidth = m_flowWidth[id];
249  float flowLen = m_flowLen[id];
250 
251  //sum the upstream overland flow
252  float qUp = 0.0f;
253  for (int k = 1; k <= (int)m_flowInIndex[id][0]; ++k)
254  {
255  int flowInID = (int)m_flowInIndex[id][k];
256  if (m_streamLink[flowInID] <= 0) // if the upstream cell is not a channel cell
257  qUp += m_q[flowInID];
258  }
259 
260  // if the channel width is greater than the cell width
261  if (m_streamLink[id] >= 0 && flowWidth <= 0)
262  {
263  m_q[id] = qUp;
264  m_sr[id] = 0.f;
265  return;
266  }
267 
268  // check whether overland flow routing is needed
269  if (qUp < MIN_FLUX && m_q[id] < MIN_FLUX)
270  {
271  m_sr[id] = 0.f;
272  m_q[id] = 0.f;
273  return;
274  }
275 
276  // calcluate infiltration surplus (m2/s)
277  float surplus = 0.f;
278  if (m_infilCapacitySurplus != NULL)
279  surplus = -m_infilCapacitySurplus[id]/1000.f * flowWidth / m_dtStorm;
280 
281  float qIn = m_q[id];
282  m_q[id] = GetNewQ(qUp, qIn, surplus, m_alpha[id], m_dtStorm, flowLen);
283 
284  float hNew = (m_alpha[id]*pow(m_q[id], 0.6f)) / flowWidth; // unit m
285  //float hTest = h + (qUp - m_q[id])*m_dtStorm/(flowWidth*flowLen);
286  m_sr[id] = hNew * 1000;
287 
288  float reInfil = (qUp - m_q[id])*m_dtStorm/(flowWidth*flowLen) + h - hNew;
289  reInfil *= 1000.f;
290 
291  //if (abs(reInfil) < 0.001f)
292  // reInfil = 0.f;
293  //else if(abs(reInfil - m_infilCapacitySurplus[id]) < 0.001f)
294  // reInfil = m_infilCapacitySurplus[id];
295 
296  //if (reInfil < 0.f || reInfil > m_infilCapacitySurplus[id])
297  // throw ModelException("IKW_OL", "OverlandFlow", "Reinfiltration exceeded range!");
298 
299  if (reInfil < 0.f)
300  reInfil = 0.f;
301 
302  m_infil[id] += reInfil;
303  if (m_accumuDepth != NULL)
304  m_accumuDepth[id] += reInfil;
305 
306  if (m_infilCapacitySurplus != NULL)
307  m_infilCapacitySurplus[id] -= reInfil;
308 
309  m_reInfil[id] = reInfil;
310 
311  // compute to channel flow
312  if (m_streamLink[id] > 0)
313  {
314  float fractiontochannel = min(m_dtStorm*m_vel[id]/(0.5f*flowWidth), 1.0f);
315  float Volume = m_sr[id]/1000.f * m_flowWidth[id] * flowLen;
316 
317  //if (id == m_idOutlet)// in catchment outlet cell, throw everything in channel
318  fractiontochannel = 1.0f;
319 
320  m_q[id] += fractiontochannel*Volume/m_dtStorm; // water diverted to the channel
321  m_sr[id] *= (1.f - fractiontochannel);
322 
323  }

327 int ImplicitKinematicWave::Execute()
328 {
329  initalOutputs();
330 
331 
332  for (int iLayer = 0; iLayer < m_nLayers; ++iLayer)
333  {
334  // There are not any flow relationship within each routing layer.
335  // So parallelization can be done here.
336  int nCells = (int)m_routingLayers[iLayer][0];
337  //omp_set_num_threads(2);
338  #pragma omp parallel for
339  for (int iCell = 1; iCell <= nCells; ++iCell)
340  {
341  int id = (int)m_routingLayers[iLayer][iCell];
342  OverlandFlow(id);
343  }
344  }
345 
346  return 0;
347 }
