119 void InterFlow_IKW::FlowInSoil(int id)
120 {
121  //sum the upstream overland flow
122  float qUp = 0.0f;
123  for (int k = 1; k <= (int)m_flowInIndex[id][0]; ++k)
124  {
125  int flowInID = (int)m_flowInIndex[id][k];
126  if (m_streamLink[id] > 0)
127  qUp += m_q[flowInID];
128  }
129 
130  float flowWidth = m_CellWidth;
131  // there is no land in this cell
132  if (m_streamLink[id] > 0)
133  {
134  flowWidth -= m_chWidth[id];
135  if (flowWidth <= 0)
136  {
137  m_q[id] = qUp;
138  m_h[id] = 0.f;
139  }
140  return;
141  }
142 
143  // adjust soil moisture
144  float s0 = m_s0[id];
145  float soilVolumn = m_rootDepth[id]/1000*m_CellWidth*flowWidth/cos(atan(s0)); //m3
146  m_soilMoistrue[id] += qUp * m_dt / soilVolumn;
147 
148  // the water exceeds the porosity is added to storage (return flow)
149  if(m_soilMoistrue[id] > m_porosity[id])
150  {
151  m_hReturnFlow[id] = (m_soilMoistrue[id] - m_porosity[id]) * m_rootDepth[id];
152  m_sr[id] += m_hReturnFlow[id];
153  m_soilMoistrue[id] = m_porosity[id];
154  }
155 
156  // if soil moisture is below the field capacity, no interflow will be generated
157  if (m_soilMoistrue[id] < m_fieldCapacity[id])
158  {
159  m_q[id] = 0.f;
160  m_h[id] = 0.f;
161  return;
162  }
163 
164  // calculate effective hydraulic conductivity (mm/h -> m/s)
165  //float k = m_ks[id]/1000/3600 * pow((m_soilMoistrue[id] - m_residual[id])/(m_porosity[id] - m_residual[id]), m_poreIndex[id]);
166  float k = m_ks[id]/1000/3600 * pow(m_soilMoistrue[id]/m_porosity[id], m_poreIndex[id]);
167  // calculate interflow (m3/s)
168  m_q[id] = m_landuseFactor * m_rootDepth[id]/1000 * s0 * k * m_CellWidth;
169 
170  // available water
171  float availableWater = (m_soilMoistrue[id] - m_fieldCapacity[id]) * soilVolumn;
172  float interFlow = m_q[id] * m_dt; // m3
173  if (interFlow > availableWater)
174  {
175  m_q[id] = availableWater / m_dt;
176  interFlow = availableWater;
177  }
178  m_h[id] = 1000 * interFlow / (m_CellWidth*m_CellWidth);
179 
180  // adjust soil moisture
181  m_soilMoistrue[id] -= interFlow / soilVolumn;
182 }

184 int InterFlow_IKW::Execute()
185 {
186 
187  initalOutputs();
188 
189  for (int iLayer = 0; iLayer < m_nLayers; ++iLayer)
190  {
191  // There are not any flow relationship within each routing layer.
192  // So parallelization can be done here.
193  int nCells = (int)m_routingLayers[iLayer][0];
194  //omp_set_num_threads(2);
195  #pragma omp parallel for
196  for (int iCell = 1; iCell <= nCells; ++iCell)
197  {
198  int id = (int)m_routingLayers[iLayer][iCell];
199  FlowInSoil(id);
200  }
201  }
202  return 0;
203 }
