87  float rechargeCap = m_Conductivity[i] / 3600.f * m_timestep * pow(moisture/m_Porosity[i], dcIndex); //Campbell, 1974
88  float availableWater = (m_Moisture[i] - m_FieldCapacity[i])*m_rootDepth[i];
89  if (rechargeCap >= availableWater)
90  rechargeCap = availableWater;
