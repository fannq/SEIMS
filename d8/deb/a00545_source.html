540 void MUSK_CH::GetCoefficients(float reachLength, float v0, MuskWeights& weights)
541 {
542  v0 = m_vScalingFactor * v0;
543  float K = (4.64f - 3.64f*m_co1) * reachLength / (5.f*v0/3.f);
544 
545  float min = 2.0f*K*m_x;
546  float max = 2.0f*K*(1-m_x);
547  float dt;
548  int n;
549  GetDt((float)m_dt, min, max, dt, n);
550  weights.dt = dt;
551 
552  //get coefficient
553  float temp = max + dt;
554  weights.c1 = (dt - min)/temp;
555  weights.c2 = (dt + min)/temp;
556  weights.c3 = (max - dt)/temp;
557  weights.c4 = 2*dt/temp;
558  weights.n = n;
559 
560  //make sure any coefficient is positive
561  if(weights.c1 < 0)
562  {
563  weights.c2 += weights.c1;
564  weights.c1 = 0.0f;
565  }
566  if(weights.c3 < 0)
567  {
568  weights.c2 += weights.c1;
569  weights.c3 = 0.0f;
570  }

573 void MUSK_CH::ChannelFlow(int i)
574 {
575  float st0 = m_chStorage[i];
576  float qiSub = 0.f;
577  if(m_qiSub != NULL)
578  qiSub = m_qiSub[i];
579  float qgSub = 0.f;
580  if(m_qgSub != NULL)
581  qgSub = m_qgSub[i];
582 
584  // first add all the inflow water
585  // 1. water from this subbasin
586  float qIn = m_qsSub[i] + qiSub + qgSub;
587 
588  // 2. water from upstream reaches
589  float qsUp = 0.f;
590  float qiUp = 0.f;
591  float qgUp = 0.f;
592  for (size_t j = 0; j < m_reachUpStream[i].size(); ++j)
593  {
594  int upReachId = m_reachUpStream[i][j];
595  qsUp += m_qsCh[upReachId];
596  qiUp += m_qiCh[upReachId];
597  qgUp += m_qgCh[upReachId];
598  }
599  qIn += qsUp + qiUp + qgUp;
600  qIn += m_qUpReach; // m_qUpReach is zero for not-parallel program and qsUp, qiUp and qgUp are zero for parallel computing
601 
602  // 3. water from bank storage
603  float bankOut = m_bankStorage[i] * (1 - exp(-m_aBank));
604 
605  m_bankStorage[i] -= bankOut;
606  qIn += bankOut/m_dt;
607 
608  // add inflow water to storage
609  m_chStorage[i] += qIn*m_dt;
610 
612  // then subtract all the outflow water
613  // 1. transmission losses to deep aquifer, which is lost from the system
614  // the unit of kchb is mm/hr
615  float seepage = m_Kchb/1000.f /3600.f * m_chWidth[i] * m_chLen[i] * m_dt;
616  if(qgSub < 0.001f)
617  {
618  if (m_chStorage[i] > seepage)
619  {
620  m_seepage[i] = seepage;
621  m_chStorage[i] -= seepage;
622  }
623  else
624  {
625  m_seepage[i] = m_chStorage[i];
626  m_chStorage[i] = 0.f;
627  m_qOut[i] = 0.f;
628  m_qsCh[i] = 0.f;
629  m_qiCh[i] = 0.f;
630  m_qgCh[i] = 0.f;
631  return;
632  }
633  }
634  else
635  {
636  m_seepage[i] = 0.f;
637  }
638 
639  // 2. calculate transmission losses to bank storage
640  float dch = m_chStorage[i]/(m_chWidth[i]*m_chLen[i]);
641  float bankInLoss = 2 * m_Kbank/1000.f/3600.f * dch * m_chLen[i] * m_dt ; // m3/s
642  bankInLoss = 0.f;
643  if (m_chStorage[i] > bankInLoss)
644  {
645  m_chStorage[i] -= bankInLoss;
646  }
647  else
648  {
649  bankInLoss = m_chStorage[i];
650  m_chStorage[i] = 0.f;
651  }
652  // water balance of the bank storage
653  // loss the water from bank storage to the adjacent unsaturated zone and groundwater storage
654  float bankOutGw = m_bankStorage[i] * (1 - exp(-m_bBank));
655  bankOutGw = 0.f;
656  m_bankStorage[i] = m_bankStorage[i] + bankInLoss - bankOutGw;
657  if(m_gwStorage != NULL)
658  m_gwStorage[i] += bankOutGw/m_area[i] * 1000.f; // updated groundwater storage
659 
660  if (FloatEqual(m_chStorage[i], 0.f))
661  {
662  m_qOut[i] = 0.f;
663  m_qsCh[i] = 0.f;
664  m_qiCh[i] = 0.f;
665  m_qgCh[i] = 0.f;
666  return;
667  }
668 
669  // 3. evaporation losses
670  float et = 0.f;
