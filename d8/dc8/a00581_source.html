121 int SUR_MR::Execute()
122 {
123  CheckInputData();
124 
125  initalOutputs();
126 
127  #pragma omp parallel for
128  for (int i = 0; i < m_nCells; i++)
129  {
130  //account for the effects of snow melt and soil temperature
131  float t = 10.f;
132  if (m_tMax != NULL && m_tMin != NULL)
133  t = (m_tMax[i] + m_tMin[i]) / 2;
134  float snowMelt = 0.f;
135  float snowAcc = 0.f;
136  if (m_snowMelt != NULL)
137  snowMelt = m_snowMelt[i];
138  if (m_snowAccu != NULL)
139  snowAcc = m_snowAccu[i];
140 
141  float sd = 0.f;
142  if (m_sd != NULL)
143  sd = m_sd[i];
144 
145  float hWater = 0.f;
146  // snow, without snow melt
147  if (t <= m_tSnow)
148  {
149  hWater = 0.0f;
150  }
151  // rain on snow, no snow melt
152  else if (t > m_tSnow && t <= m_t0 && snowAcc > m_pNet[i])
153  {
154  hWater = 0.0f;
155  }
156  else
157  {
158  hWater = m_pNet[i] + snowMelt + sd;
159  }
160 
161  if(hWater > 0.f)
162  {
163  float sm = m_soilMoisture[i][0]*m_upSoilDepth + m_soilMoisture[i][1]*(m_rootDepth[i] - m_upSoilDepth);
164  float por = m_porosity[i][0]*m_upSoilDepth + m_porosity[i][1]*(m_rootDepth[i] - m_upSoilDepth);
165 
166  float smFraction = sm / por;
167  smFraction = min(smFraction, 1.0f);
168 
169  // for frozen soil
170  if (m_soilTemp != NULL && m_soilTemp[i] <= m_tFrozen && smFraction >= m_sFrozen)
171  {
172  m_pe[i] = m_pNet[i] + snowMelt;
173  m_infil[i] = 0.0f;
174  }
175  //else if (m_soilMoisture[i] >= m_porosity[i])//for saturation overland flow
176  //{
177  // m_pe[i] = m_pNet[i] + snowMelt;
178  // m_infil[i] = 0.0f;
179  //}
180  else
181  {
182  float alpha = m_kRunoff - (m_kRunoff - 1) * hWater / m_pMax;
183  if (hWater >= m_pMax)
184  alpha = 1.0f;
185 
186  //runoff percentage
187  float runoffPercentage;
188  if (m_runoffCo[i] > 0.99f)
189  runoffPercentage = 1.0f;
190  else
191  runoffPercentage = m_runoffCo[i] * pow(smFraction, alpha);
192 
193  float surfq = hWater * runoffPercentage;
194 
195  m_infil[i] = hWater - surfq;
196  m_pe[i] = surfq;
197 
198  // check the output data
199  if (m_infil[i] != m_infil[i] || m_infil[i] < 0.0f)
200  {
201  //string datestr = getDate(&m_date);
202  ostringstream oss;
203  oss << "Cell id:" << i << "\tPrecipitation(mm) = " << m_pNet[i] << "\thwater = " << hWater
204  << "\tpercentage:" << runoffPercentage << "\tmoisture = " << sm
205  << "\tInfiltration(mm) = " << m_infil[i] << "\n";
206  throw ModelException("SUR_MR","Execute","Output data error: infiltration is less than zero. Where:\n"
207  + oss.str() + "Please contact the module developer. ");
208  }
209 
210  }
211  }
212  else
213  {
214  m_pe[i] = 0.0f;
215  m_infil[i] = 0.0f;
216  }
217  }
218  return 0;
219 }
