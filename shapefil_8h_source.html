00170 /* -------------------------------------------------------------------- */
00171 /*      SHPAPI_CALL                                                     */
00172 /*                                                                      */
00173 /*      The following two macros are present to allow forcing           */
00174 /*      various calling conventions on the Shapelib API.                */
00175 /*                                                                      */
00176 /*      To force __stdcall conventions (needed to call Shapelib         */
00177 /*      from Visual Basic and/or Dephi I believe) the makefile could    */
00178 /*      be modified to define:                                          */
00179 /*                                                                      */
00180 /*        /DSHPAPI_CALL=__stdcall                                       */
00181 /*                                                                      */
00182 /*      If it is desired to force export of the Shapelib API without    */
00183 /*      using the shapelib.def file, use the following definition.      */
00184 /*                                                                      */
00185 /*        /DSHAPELIB_DLLEXPORT                                          */
00186 /*                                                                      */
00187 /*      To get both at once it will be necessary to hack this           */
00188 /*      include file to define:                                         */
00189 /*                                                                      */
00190 /*        #define SHPAPI_CALL __declspec(dllexport) __stdcall           */
00191 /*        #define SHPAPI_CALL1 __declspec(dllexport) * __stdcall        */
00192 /*                                                                      */
00193 /*      The complexity of the situtation is partly caused by the        */
00194 /*      peculiar requirement of Visual C++ that __stdcall appear        */
00195 /*      after any "*"'s in the return value of a function while the     */
00196 /*      __declspec(dllexport) must appear before them.                  */
00197 /* -------------------------------------------------------------------- */
00198 
00199 #ifdef SHAPELIB_DLLEXPORT
00200 #  define SHPAPI_CALL __declspec(dllexport)
00201 #  define SHPAPI_CALL1(x)  __declspec(dllexport) x
00202 #endif
00203 
00204 #ifndef SHPAPI_CALL
00205 #  if defined(USE_GCC_VISIBILITY_FLAG)
00206 #    define SHPAPI_CALL     __attribute__ ((visibility("default")))
00207 #    define SHPAPI_CALL1(x) __attribute__ ((visibility("default")))     x
00208 #  else
00209 #    define SHPAPI_CALL
00210 #  endif
00211 #endif
00212 
00213 #ifndef SHPAPI_CALL1
00214 #  define SHPAPI_CALL1(x)      x SHPAPI_CALL
00215 #endif
