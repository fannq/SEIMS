00326 {
00327     lat = lat*3.1415926/180;
00328       //Calculate Daylength
00329       //calculate solar declination: equation 2.1.2 in SWAT manual
00330       float sd = asin(0.4f * sin((day - 82.0f) / 58.09f));  //365/2pi = 58.09
00331 
00332       //calculate the relative distance of the earth from the sun
00333       //the eccentricity of the orbit
00334       //equation 2.1.1 in SWAT manual
00335       float dd = 1.0f + 0.033f * cos(day / 58.09f);
00336 
00337       //daylength = 2 * Acos(-Tan(sd) * Tan(lat)) / omega
00338       //where the angular velocity of the earth's rotation, omega, is equal
00339       //to 15 deg/hr or 0.2618 rad/hr and 2/0.2618 = 7.6374
00340       //equation 2.1.6 in SWAT manual
00341 
00342       float h = 0.0f;
00343       float ch = -sin(lat) * tan(sd) / cos(lat);
00344       if (ch > 1.f) //ch will be >= 1. if latitude exceeds +/- 66.5 deg in winter
00345         h = 0.0f;
00346       else if (ch >= -1.0f)
00347         h = acos(ch);
00348       else
00349         h = 3.1416f; //latitude exceeds +/- 66.5 deg in summer
00350       
00351       float dayl = 7.6394f * h;
00352           
00353       //Calculate Potential (maximum) Radiation !!
00354       //equation 2.2.7 in SWAT manual
00355       float ys = sin(lat) * sin(sd);
00356       float yc = cos(lat) * cos(sd);
00357       
00358       return 30.f * dd * (h * ys + yc * sin(h));
