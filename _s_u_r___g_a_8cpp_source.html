00539 float SUR_GA::Calculate_CN(int cell)
00540 {
00541     float sw,s,CNday,xx;
00542 
00543     s = 0.;
00544     sw = this->m_soilMoisture[cell] * this->m_rootDepth[cell] ;       
00545     xx = this->m_w1[cell] - this->m_w2[cell] * sw;
00546     if (xx < -20.f)
00547     {
00548         xx = -20.;
00549     }
00550     if (xx > 20.f)
00551     {
00552         xx = 20.f;
00553     }
00554     // traditional CN method (function of soil water)
00555     if ((sw + exp(xx)) > 0.001f)
00556     {
00557         s = this->m_sMax[cell] * (1.f - sw /(sw + exp(xx)) );  //2:1.1.6
00558     }
00559 
00560     CNday = 25400.f / (s + 254.f);  //2:1.1.11
00561     return CNday;
00562 }

00564 void SUR_GA::initalW1W2()
00565 {
00566     this->m_w1 = new float[this->m_cellSize];
00567     this->m_w2 = new float[this->m_cellSize];
00568     this->m_sMax = new float[this->m_cellSize];
00569 
00570     for(int i=0;i<this->m_cellSize;i++)
00571     {
00572         float cnn = this->m_CN2[i];
00573         float fieldcap = this->m_fieldCap[i] * this->m_rootDepth[i];
00574         float wsat = this->m_porosity[i] * this->m_rootDepth[i];
00575         float c1,c3,c2,smx,s3,rto3,rtos,xx,wrt1,wrt2;
00576         c2 = 100.0f - cnn;
00577         c1 = cnn - 20.f * c2 / (c2 + exp(2.533f - 0.0636f * c2));   //CN1  2:1.1.4
00578         c1 = max( c1, 0.4f * cnn);
00579         c3 = cnn * exp(0.006729f * c2);                             //CN3  2:1.1.5
00580 
00581         //calculate maximum retention parameter value
00582         smx = 254.f * (100.f / c1 - 1.f);                           //2:1.1.2
00583 
00584         //calculate retention parameter value for CN3
00585         s3 = 254.f * (100.f / c3 - 1.f);
00586 
00587         rto3 = 1.f - s3 / smx;
00588         rtos = 1.f - 2.54f / smx;
00589 
00590         xx = log(fieldcap / rto3 - fieldcap);
00591         wrt2 = (xx - log(wsat / rtos - wsat)) / (wsat - fieldcap);    //soilWater :completely saturated  (= soil_por - sol_wp)  w1  2:1.1.8
00592         wrt1 = xx + (fieldcap * wrt2); //w2 2:1.1.7
00593 
00594         this->m_w1[i] = wrt1;
00595         this->m_w2[i] = wrt2;
00596         this->m_sMax[i] = smx;
00597     }
00598 }
