00099     this->initalOutputs();
00100 
00101     //the first time
00102     if(m_isInitial)
00103     {
00104         int count = 0;
00105         for(int i=0;i<this->m_cellSize;i++) 
00106         {
00107             if((this->m_tMin[i] + this->m_tMax[i]) / 2 < this->m_tsnow) {this->m_SA[i] = this->m_swe0; count++;}    //winter
00108             else                                                        this->m_SA[i] = 0.0f;                       // other seasons
00109         }
00110 
00111         m_swe =  this->m_swe0 * count / this->m_cellSize;
00112         m_isInitial = false;
00113     }
00114 
00115     if(this->m_lastSWE == -99.0f) this->m_lastSWE = this->m_swe;
00116     if(m_swe < 0.01)    //all cells have not snow, so snow redistribution is 0.
00117     {
00118         if(this->m_lastSWE >= 0.01)
00119         {
00120             for (int rw = 0; rw < this->m_cellSize; rw++) 
00121                 m_SR[rw] = 0.0f;
00122         }
00123 
00124         this->m_lastSWE = this->m_swe;
00125         return true;
00126     }
00127 
00128 
00129     //if some cells have snow, start to calculate snow redistribution.
00130     if(m_w == NULL) m_w = new float[this->m_cellSize];
00131     if(m_wt == NULL)   //get wt. wt is constant for the watershed, so save it in m_wt to save time.
00132     {
00133         m_wt = new float[this->m_cellSize];
00134         for (int rw = 0; rw < this->m_cellSize; rw++) 
00135             m_wt[rw] = 1.0f + 1 / this->m_k_slope * this->m_slope_wind[rw] + 1 / this->m_k_curvature * this->m_curva_wind[rw];          
00136     }
00137 
00138     float totalW = 0.0f;
00139     for ( int rw = 0; rw < this->m_cellSize; rw++)
00140     { 
00141         float wl = 1.0f;
00142         float shc = this->m_shc[rw] * 1000;
00143         if(this->m_swe < shc)
00144         {
00145             wl = shc / this->m_shc_crop +this->m_swe * ( 1 - shc / this->m_shc_crop) / shc ;
00146         }
00147 
00148         m_w[rw] = m_wt[rw] * wl;    //save w in the m_w array, this save time for following loop.
00149         totalW += m_w[rw];          
00150     }
00151 
00152     float weight = this->m_cellSize / totalW;       //weight
00153     float u = 0.0f;                                 //average wind speed
00154     for (int rw = 0; rw < this->m_wsSize; rw++) u+=this->m_ws[rw];
00155     u /= this->m_wsSize;
00156 
00157     for ( int rw = 0; rw < this->m_cellSize; rw++)
00158     {
00159         float snow = this->m_SA[rw];
00160         float tmean = (this->m_tMin[rw] + this->m_tMax[rw]) / 2; //if the temperature is higher than t0, the redistribution is 0
00161         if(tmean < this->m_tsnow) snow+= (1+this->m_kblow) * this->m_Pnet[rw];
00162         if(snow < 0.01)         // if snow is lower than a very small positive value (0.01), consider it is equal to 0.
00163         {
00164             this->m_SR[rw] = 0.0f;
00165             continue;
00166         }
00167 
00168         if(tmean >= this->m_t0)
00169         {
00170             this->m_SR[rw] = 0.0f;
00171             continue;
00172         }
00173 
00174         float wr = m_w[rw] * weight;
00175         float sap = m_swe * wr;
00176         float scp = sap - this->m_SA[rw];
00177 
00178         float ut = float(this->m_ut0 + 0.0033 * pow((this->m_tMin[rw] + this->m_tMax[rw])/2 - this->m_t_wind,2.0f));

00181         if(u-ut >= 0 && u-ut < this->m_u0)  ww = (u-ut) / this->m_u0;
00182

00184
