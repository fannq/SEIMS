00471 float SEDR_VCD::Q(float waterDepth, int id)
00472 {
00473     if(waterDepth < DEPTH_INTERVAL) return 0.0f;
00474 
00475     float area = crossSectionalArea(waterDepth, id);

00478     float wp = wettedPerimeter(waterDepth, id);

00485 float SEDR_VCD::crossSectionalArea(float waterDepth, int id)
00486 {
00487     if(waterDepth < DEPTH_INTERVAL) return 0.0f;
00488 
00489     if(m_depthcurrent[id] < waterDepth) //flood plain
00490     {
00491         float depth_flood = waterDepth - m_depthcurrent[id];
00492         //if(depth_flood > 9.0f * m_depthcurrent) //max flood depth
00493         //  throw ModelException("reachParameters","crossSectionalArea",
00494         //  "Two many water in reach!");    
00495 
00496         float area = (m_widthbottom[id] + m_depthcurrent[id] * this->m_sideslopeMain) * m_depthcurrent[id];
00497         area += (m_widthcurrent[id] * m_w_ratio + depth_flood * m_sideslopeMain) * depth_flood;
00498         return area;
00499     }
00500     else                            //not bankfull
00501     {
00502         return (m_widthbottom[id] + waterDepth * m_sideslopeMain) * waterDepth; 
00503     }
00504 }

00506 float SEDR_VCD::wettedPerimeter(float waterDepth, int id)
00507 {
00508     if(waterDepth < DEPTH_INTERVAL) return 0.0f;
00509 
00510     if(m_depthcurrent[id] < waterDepth) //flood plain
00511     {
00512         float depth_flood = waterDepth - m_depthcurrent[id];
00513         //if(depth_flood > 9.0f * m_depthcurrent) //max flood depth
00514         //  throw ModelException("reachParameters","crossSectionalArea",
00515         //  "Two many water in reach!");    
00516 
00517         float wettedPerimeter = m_widthbottom[id] + 
00518             2.0f * m_depthcurrent[id] * sqrt(1+pow(m_sideslopeMain,2.0f));      
00519         wettedPerimeter += 4 * m_widthcurrent[id] + 
00520             2.0f * depth_flood * sqrt(1+pow(m_sideslopeFloodplain,2.0f));
00521         return wettedPerimeter;
00522     }
00523     else                            //not bankfull
00524     {
00525         return m_widthbottom[id] + 
00526             2 * waterDepth * sqrt(1+pow(m_sideslopeMain,2.0f));     
00527     }
00528 }

00548 void SEDR_VCD::SedChannelRouting(int i)
00549 {
00550     // 1 .whether is no water in channel
00551     //float storage = m_T_CHSB[i][SEDIMENT_CHANNEL_ROUTING_RESULT_DISCHARGE_COLUMN_INDEX_STORAGE];
00552     float sedstorage = m_sedStorage[i];
00553     if (m_qchOut[i] < 0.0001f)
00554     {
00555         reset(i);
00556         m_sedStorage[i] = sedstorage;
00557         m_SedOut[i] = 0.f;
00558     }
00559     else
00560     {   // 2. sediment from upstream reaches
00561         float sedUp = 0;
00562         for (size_t j = 0; j < m_reachUpStream[i].size(); ++j)
00563         {
00564             int upReachId = m_reachUpStream[i][j];
00565             sedUp += m_SedOut[upReachId];
00566         }
00567         float allSediment = sedUp + m_sedtoCh[i] + m_sedStorage[i];
00568         //cout<<i<<"\t"<<sedUp <<"\t"<<m_sedtoCh[i]<<"\t"<<sedstorage<<endl;
00569         
00570         m_sedUps[i] = sedUp;
00571         m_sedSub[i] = m_sedtoCh[i];
00572         //get peak channel velocity (m/s)
00573         float peakVelocity = 0.0f;
00574         
00575         float peakFlowRate = m_qchOut[i] * m_prf;
