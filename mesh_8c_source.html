00237 idx_t FindCommonElements(idx_t qid, idx_t elen, idx_t *eind, idx_t *nptr, 
00238           idx_t *nind, idx_t *eptr, idx_t ncommon, idx_t *marker, idx_t *nbrs)
00239 {
00240   idx_t i, ii, j, jj, k, l, overlap;
00241 
00242   /* find all elements that share at least one node with qid */
00243   for (k=0, i=0; i<elen; i++) {
00244     j = eind[i];
00245     for (ii=nptr[j]; ii<nptr[j+1]; ii++) {
00246       jj = nind[ii];
00247 
00248       if (marker[jj] == 0) 
00249         nbrs[k++] = jj;
00250       marker[jj]++;
00251     }
00252   }
00253 
00254   /* put qid into the neighbor list (in case it is not there) so that it
00255      will be removed in the next step */
00256   if (marker[qid] == 0)
00257     nbrs[k++] = qid;
00258   marker[qid] = 0;
00259 
00260   /* compact the list to contain only those with at least ncommon nodes */
00261   for (j=0, i=0; i<k; i++) {
00262     overlap = marker[l = nbrs[i]];
00263     if (overlap >= ncommon || 
00264         overlap >= elen-1 || 
00265         overlap >= eptr[l+1]-eptr[l]-1)
00266       nbrs[j++] = l;
00267     marker[l] = 0;
00268   }
00269 
00270   return j;
00271 }

00348 idx_t FindCommonNodes(idx_t qid, idx_t nelmnts, idx_t *elmntids, idx_t *eptr, 
00349           idx_t *eind, idx_t *marker, idx_t *nbrs)
00350 {
00351   idx_t i, ii, j, jj, k;
00352 
00353   /* find all nodes that share at least one element with qid */
00354   marker[qid] = 1;  /* this is to prevent self-loops */
00355   for (k=0, i=0; i<nelmnts; i++) {
00356     j = elmntids[i];
00357     for (ii=eptr[j]; ii<eptr[j+1]; ii++) {
00358       jj = eind[ii];
00359       if (marker[jj] == 0) {
00360         nbrs[k++] = jj;
00361         marker[jj] = 1;
00362       }
00363     }
00364   }
00365 
00366   /* reset the marker */
00367   marker[qid] = 0;
00368   for (i=0; i<k; i++) {
00369     marker[nbrs[i]] = 0;
00370   }
00371 
00372   return k;
00373 }
