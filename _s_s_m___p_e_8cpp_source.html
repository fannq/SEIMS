00070 int SSM_PE::Execute()
00071 {
00072     this->CheckInputData();
00073 
00074     this->initalOutputs();
00075 
00076     if(m_isInitial)
00077     {
00078         int count = 0;
00079         for(int i=0;i<this->m_cellSize;i++) 
00080         {
00081             if((this->m_tMin[i] + this->m_tMax[i]) / 2 < this->m_tsnow) {this->m_SA[i] = this->m_swe0; count++;}    //winter
00082             else                                                        this->m_SA[i] = 0.0f;                       // other seasons
00083         }
00084 
00085         m_swe =  this->m_swe0 * count / this->m_cellSize;
00086         m_isInitial = false;
00087     }
00088 
00089     if(this->m_lastSWE == -99.0f) this->m_lastSWE = this->m_swe;
00090     if(m_swe < 0.01)    //all cells have not snow, so snow sublimation is 0.
00091     {
00092         if(this->m_lastSWE >= 0.01)
00093         {
00094             for (int rw = 0; rw < this->m_cellSize; rw++) 
00095                 m_SE[rw] = 0.0f;
00096         }
00097 
00098         this->m_lastSWE = this->m_swe;
00099         return true;
00100     }
00101 
00102     for (int rw = 0; rw < this->m_cellSize; rw++) 
00103     {
00104         float snow = this->m_SA[rw] + this->m_SR[rw];
00105         float tmean = (this->m_tMin[rw] + this->m_tMax[rw]) / 2;
00106         if(tmean < this->m_tsnow) snow+=(1+this->m_kblow)*this->m_Pnet[rw];
00107         if(snow < 0.01) this->m_SE[rw] = 0.0f;
00108         else
00109         {           
00110             if(tmean >= this->m_t0) this->m_SE[rw] = 0.0f;   //if temperature is higher than t0, the sublimation is 0.
00111             else
00112             {
00113                 float se = this->m_ksubli * this->m_PET[rw];
00114                 this->m_SE[rw] = min(snow,se);
00115             }
00116         }
00117     }
00118 
00119     this->m_lastSWE = this->m_swe;
00120     return true;
00121     return 0;
