00190 /*                           DBFWriteHeader()                           */
00191 /*                                                                      */
00192 /*      This is called to write out the file header, and field          */
00193 /*      descriptions before writing any actual data records.  This      */
00194 /*      also computes all the DBFDataSet field offset/size/decimals     */
00195 /*      and so forth values.                                            */
00196 /************************************************************************/
00197 
00198 static void DBFWriteHeader(DBFHandle psDBF)
00199 
00200 {
00201     unsigned char   abyHeader[XBASE_FLDHDR_SZ];
00202     int     i;
00203 
00204     if( !psDBF->bNoHeader )
00205         return;
00206 
00207     psDBF->bNoHeader = FALSE;
00208 
00209 /* -------------------------------------------------------------------- */
00210 /*  Initialize the file header information.             */
00211 /* -------------------------------------------------------------------- */
00212     for( i = 0; i < XBASE_FLDHDR_SZ; i++ )
00213         abyHeader[i] = 0;
00214 
00215     abyHeader[0] = 0x03;        /* memo field? - just copying   */
00216 
00217     /* write out a dummy date */
00218     abyHeader[1] = 95;          /* YY */
00219     abyHeader[2] = 7;           /* MM */
00220     abyHeader[3] = 26;          /* DD */
00221 
00222     /* record count preset at zero */
00223 
00224     abyHeader[8] = (unsigned char) (psDBF->nHeaderLength % 256);
00225     abyHeader[9] = (unsigned char) (psDBF->nHeaderLength / 256);
00226     
00227     abyHeader[10] = (unsigned char) (psDBF->nRecordLength % 256);
00228     abyHeader[11] = (unsigned char) (psDBF->nRecordLength / 256);
00229 
00230     abyHeader[29] = (unsigned char) (psDBF->iLanguageDriver);
00231 
00232 /* -------------------------------------------------------------------- */
00233 /*      Write the initial 32 byte file header, and all the field        */
00234 /*      descriptions.                                           */
00235 /* -------------------------------------------------------------------- */
00236     psDBF->sHooks.FSeek( psDBF->fp, 0, 0 );
00237     psDBF->sHooks.FWrite( abyHeader, XBASE_FLDHDR_SZ, 1, psDBF->fp );
00238     psDBF->sHooks.FWrite( psDBF->pszHeader, XBASE_FLDHDR_SZ, psDBF->nFields, 
00239                           psDBF->fp );
00240 
00241 /* -------------------------------------------------------------------- */
00242 /*      Write out the newline character if there is room for it.        */
00243 /* -------------------------------------------------------------------- */
00244     if( psDBF->nHeaderLength > 32*psDBF->nFields + 32 )
00245     {
00246         char    cNewline;
00247 
00248         cNewline = 0x0d;
00249         psDBF->sHooks.FWrite( &cNewline, 1, 1, psDBF->fp );
00250     }
00251 }

00292 static int DBFLoadRecord( DBFHandle psDBF, int iRecord )
00293 
00294 {
00295     if( psDBF->nCurrentRecord != iRecord )
00296     {

00299     if( !DBFFlushRecord( psDBF ) )

00303             psDBF->nRecordLength * (SAOffset) iRecord + psDBF->nHeaderLength;

00306         {
00307             char szMessage[128];

00309                      (long) nRecordOffset );
00310             psDBF->sHooks.Error( szMessage );
00311             return FALSE;
00312         }
00313 
00314     if( psDBF->sHooks.FRead( psDBF->pszCurrentRecord, 
00315                                  psDBF->nRecordLength, 1, psDBF->fp ) != 1 )
00316         {
00317             char szMessage[128];

00319                      psDBF->nRecordLength );
00320             psDBF->sHooks.Error( szMessage );
00321             return FALSE;
00322         }
00323 
00324     psDBF->nCurrentRecord = iRecord;
