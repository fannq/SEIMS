81 int clsPI_MSM::Execute()
82 {
83  //check input data
84  this->CheckInputData();
85 
86  //initial the state variable
87  if(this->m_st == NULL)
88  {
89  m_st = new float[this->m_cellSize];
90  this->m_evaporation = new float[this->m_cellSize];
91  this->m_interceptionLoss = new float[this->m_cellSize];
92  this->m_netPrecipitation = new float[this->m_cellSize];
93 
94  #pragma omp parallel for
95  for(int i = 0 ;i<this->m_cellSize;i++)
96  {
97  m_st[i] = this->m_Init_IS;
98  m_evaporation[i] = 0.f;
99  m_interceptionLoss[i] = 0.f;
100  m_netPrecipitation[i] = 0.f;
101  }
102  }
103 
104  int julian = JulianDay(m_date);
105  #pragma omp parallel for
106  for(int i = 0 ; i < this->m_cellSize; i ++)
107  {
108  if (m_P[i] > 0)
109  {
110  //interception storage capacity
111  float degree = 2 * PI * (julian - 87)/365;
112  float min = m_minSt[i];
113  float max = m_maxSt[i];
114  float capacity = min + (max - min)*pow(0.5 + 0.5*sin(degree),double(m_Pi_b));
115 
116  //interception
117  float availableSpace = capacity - m_st[i];
118  if (availableSpace < 0)
119  availableSpace = 0.f;
120 
121  if(availableSpace < m_P[i])
122  m_interceptionLoss[i] = availableSpace;
123  else
124  m_interceptionLoss[i] = m_P[i];
125 
126  //net precipitation
127  m_netPrecipitation[i] = m_P[i] - m_interceptionLoss[i];
128  m_st[i] += m_interceptionLoss[i];
129  }
130  else
131  {
132  m_interceptionLoss[i] = 0.f;
133  m_netPrecipitation[i] = 0.f;
134  }
135 
136  //evaporation
137  if(m_st[i] > m_PET[i])
138  m_evaporation[i] = m_PET[i];
139  else
140  m_evaporation[i] = m_st[i];
141 
142  m_st[i] -= m_evaporation[i];
143  }
144 
145  return 0;
146 }
