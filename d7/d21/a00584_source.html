135 int GWaterReservoir::Execute(void)
136 {
137  InitOutputs();
138  CheckInputData();
139  #pragma omp parallel for
140  for (int i = 0; i <= m_nReaches; i++)
141  m_percSubbasin[i] = 0.f;
142 
143  // get percolation for each subbasin
144  for (int i = 0; i < m_nCells; i++)
145  {
146 #ifdef MULTIPLY_REACHES
147  int subbasinId = (int)m_subbasin[i];
148 #else
149  int subbasinId = 1;
150 #endif
151  m_percSubbasin[subbasinId] += m_recharge[i];
152  }
153 
154  //float sum = 0.f;
155  #pragma omp parallel for //reduction(+:sum)
156  for (int i = 1; i <= m_nReaches; i++)
157  {
158  float percolation = m_percSubbasin[i] * (1 - m_deepCoefficient) / m_nCellsSubbasin[i];
159  // depth of groundwater runoff(mm)
160  float outFlowDepth = m_recessionCoefficient * pow(m_storage[i], m_recessionExponent);
161  // groundwater flow out of the subbasin at time t (m3/s)
162  m_qg[i] = outFlowDepth /1000 * m_nCellsSubbasin[i] * m_CellWidth * m_CellWidth / m_dt;
163  //sum = sum + m_qg[i];
164 
165  // water balance (mm)
166  m_storage[i] += percolation - outFlowDepth;
167  }
168  //m_qg[0] = sum;
169 
170  return 0;
