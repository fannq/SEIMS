00597   pos = gk_i32incset(nvtxs, 0, gk_i32malloc(nvtxs, "gk_graph_FindComponents: pos"));
00598 
00599 
00600   /* Find the connected componends */
00601   ncmps = -1;
00602   ntodo = nvtxs;     /* All vertices have not been visited */
00603   first = last = 0;  /* Point to the first and last vertices that have been touched
00604                         but not explored. 
00605                         These vertices are stored in cind[first]...cind[last-1]. */
00606   while (ntodo > 0) {
00607     if (first == last) { /* Find another starting vertex */
00608       cptr[++ncmps] = first;  /* Mark the end of the current CC */
00609 
00610       ASSERT(pos[todo[0]] != -1);
00611       i = todo[0];
00612 
00613       cind[last++] = i;
00614       pos[i] = -1;
00615     }
00616 
00617     i = cind[first++];  /* Get the first visited but unexplored vertex */
00618 
00619     /* Remove i from the todo list and put the last item in the todo 
00620        list at the position that i was so that the todo list will be
00621        consequtive. The pos[] array is updated accordingly to keep track
00622        the location of the vertices in the todo[] list. */
00623     k = pos[i];
00624     j = todo[k] = todo[--ntodo];
00625     pos[j] = k;
00626 
00627     for (j=xadj[i]; j<xadj[i+1]; j++) {
00628       k = adjncy[j];
00629       if (pos[k] != -1) {
00630         cind[last++] = k;
00631         pos[k] = -1;
00632       }
00633     }
00634   }
00635   cptr[++ncmps] = first;
00636 
00637   if (mustfree_ccsr)
00638     gk_free((void **)&cptr, &cind, LTERM);
00639 
00640   gk_free((void **)&pos, &todo, LTERM);
00641 
00642   return (int) ncmps;
00643 }

00664 /*************************************************************************/
00665 void gk_graph_ComputeBFSOrdering(gk_graph_t *graph, int v, int32_t **r_perm,
00666           int32_t **r_iperm)
00667 {
00668   ssize_t j, *xadj;
00669   int i, k, nvtxs, first, last;
00670   int32_t *adjncy, *cot, *pos;
00671 
00672   if (graph->nvtxs <= 0)
00673     return;
00674 
00675   nvtxs  = graph->nvtxs;
00676   xadj   = graph->xadj;
00677   adjncy = graph->adjncy;
00678 
00679   /* This array will function like pos + touched of the CC method */
00680   pos = gk_i32incset(nvtxs, 0, gk_i32malloc(nvtxs, "gk_graph_ComputeBFSOrdering: pos"));
00681 
00682   /* This array ([C]losed[O]pen[T]odo => cot) serves three purposes. 
00683      Positions from [0...first) is the current iperm[] vector of the explored vertices; 
00684      Positions from [first...last) is the OPEN list (i.e., visited vertices);
00685      Positions from [last...nvtxs) is the todo list. */
00686   cot = gk_i32incset(nvtxs, 0, gk_i32malloc(nvtxs, "gk_graph_ComputeBFSOrdering: cot"));
00687 
00688 
00689   /* put v at the front of the todo list */
00690   pos[0] = cot[0] = v;
00691   pos[v] = cot[v] = 0;
00692 
00693   /* Find the connected componends induced by the partition */
00694   first = last = 0;
00695   while (first < nvtxs) {
00696     if (first == last) { /* Find another starting vertex */
00697       k = cot[last];
00698       ASSERT(pos[k] != -1);
00699       pos[k] = -1; /* mark node as being visited */
00700       last++;
00701     }
00702 
00703     i = cot[first++];  /* the ++ advances the explored vertices */
00704     for (j=xadj[i]; j<xadj[i+1]; j++) {
00705       k = adjncy[j];
00706       /* if a node has already been visited, its perm[] will be -1 */
00707       if (pos[k] != -1) {
00708         /* pos[k] is the location within iperm of where k resides (it is in the 'todo' part); 
00709            It is placed in that location cot[last] (end of OPEN list) that we 
00710            are about to overwrite and update pos[cot[last]] to reflect that. */
00711         cot[pos[k]]    = cot[last]; /* put the head of the todo list to 
00712                                        where k was in the todo list */
00713         pos[cot[last]] = pos[k];    /* update perm to reflect the move */
00714 
00715         cot[last++] = k;  /* put node at the end of the OPEN list */
00716         pos[k]      = -1; /* mark node as being visited */
00717       }
00718     }
00719   }
00720 
00721   /* time to decide what to return */
00722   if (r_perm != NULL) {
00723     /* use the 'pos' array to build the perm array */
00724     for (i=0; i<nvtxs; i++)
00725       pos[cot[i]] = i;
00726 
00727     *r_perm = pos;
00728     pos = NULL;
00729   }
00730 
00731   if (r_iperm != NULL) {
00732     *r_iperm = cot;
00733     cot = NULL;
00734   }
00735 
00736 
00737   /* cleanup memory */
00738   gk_free((void **)&pos, &cot, LTERM);
00739 
00740 }
