190  p->Name = GetStringFromBSONITER(&iter);
191  if (bson_iter_init_find(&iter,info,PARAM_FLD_UNIT))
192  p->Units = GetStringFromBSONITER(&iter);
193  if (bson_iter_init_find(&iter,info,PARAM_FLD_VALUE))
194  p->Value = GetFloatFromBSONITER(&iter);
195  if (bson_iter_init_find(&iter,info,PARAM_FLD_CHANGE))
196  p->Change = GetStringFromBSONITER(&iter);
197  if (bson_iter_init_find(&iter,info,PARAM_FLD_IMPACT))
198  p->Impact = GetFloatFromBSONITER(&iter);
199  if (bson_iter_init_find(&iter,info,PARAM_FLD_MAX))

201  if (bson_iter_init_find(&iter,info,PARAM_FLD_MIN))
202  p->Min = GetFloatFromBSONITER(&iter);
203  if (bson_iter_init_find(&iter,info,PARAM_FLD_USE))
204  p->Use = GetStringFromBSONITER(&iter);

206  }

208  mongoc_collection_destroy(collection);

312 void ModuleFactory::ReadDLL(string& id, string& dllID)
313 {
314  // the dll file is already read, return
315  if (m_instanceFuncs.find(id) != m_instanceFuncs.end())
316  return;
317 
318  // check if the dll file exists
319  string moduleFileName = trim(m_modulePath) + string(dllID) + string(Tag_ModuleExt);
320  utils util;
321  if(!util.FileExists(moduleFileName)) throw ModelException("ModuleFactory", "ReadDLL", moduleFileName + " does not exist or has no read permission!");
322 
323  //load library
324 #ifndef linux
325  HINSTANCE handle = LoadLibrary(TEXT(moduleFileName.c_str()));
326  if (handle == NULL) throw ModelException("ModuleFactory", "ReadDLL", "Could not load " + moduleFileName);
327  m_instanceFuncs[id] = InstanceFunction(GetProcAddress(HMODULE(handle), "GetInstance"));
328  m_metadataFuncs[id] = MetadataFunction(GetProcAddress(HMODULE(handle), "MetadataInformation"));
329 #else
330  void* handle = dlopen(moduleFileName.c_str(), RTLD_LAZY);
331  if (handle == NULL)
332  {
333  cout << dlerror() << endl;
334  throw ModelException("ModuleFactory", "ReadDLL", "Could not load " + moduleFileName);
335  }
336  m_instanceFuncs[id] = InstanceFunction(dlsym(handle, "GetInstance"));
337  m_metadataFuncs[id] = MetadataFunction(dlsym(handle, "MetadataInformation"));
338 #endif
339  m_dllHandles.push_back(handle);
340  if(m_instanceFuncs[id] == NULL)
341  throw ModelException("ModuleFactory", "ReadDLL", moduleFileName + " does not implement API function: GetInstance");
342  if(m_metadataFuncs[id] == NULL)
343  throw ModelException("ModuleFactory", "ReadDLL", moduleFileName + " does not implement API function: MetadataInformation");

736 bool ModuleFactory::LoadSettingsFromFile(const char* filename, vector< vector<string> >& settings)
737 {
738  bool bStatus = false;
739  ifstream myfile;
740  string line;
741  utils utl;
742  try
743  {
744  // open the file
745  myfile.open(filename, ios::in);
746  if (myfile.is_open())
747  {
748  while (!myfile.eof())
749  {
750  if (myfile.good())
751  {
752  getline(myfile, line);
753  line = trim(line);
754  if ((line.size() > 0) && (line[0] != '#')) // ignore comments and empty lines
755  {
756  // parse the line into separate item
757  vector<string> tokens = utl.SplitString(line, '|');
758  // is there anything in the token list?
759  if (tokens.size() > 0)
760  {
761  for (size_t i=0; i<tokens.size(); i++)
762  {
763  //utl.TrimSpaces(tokens[i]);
764  tokens[i] = trim(tokens[i]);
765  }
766  // is there anything in the first item?
767  if (tokens[0].size() > 0)
768  {
769  // there is something to add so resize the header list to append it
770  int sz = settings.size(); // get the current number of rows
771  settings.resize(sz+1); // resize with one more row
772 
773  settings[sz] = tokens;
774 
775  bStatus = true; // consider this a success
776  } // if there is nothing in the first item of the token list there is nothing to add to the header list
777  }
778  }
779  }
780  }
781  bStatus = true;
782  myfile.close();
783  }
784  }
785  catch (...)
786  {
787  myfile.close();
788  throw ModelException("ModuleFactory","LoadSettingsFromFile","Failed, Please check the format of " + string(File_Config) + ".\n");
789  }
790  return bStatus;
