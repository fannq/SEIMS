37 bool SettingsInput::readDate()
38 {
39  //read start and end time
40  utils utl;
41  m_startDate = utl.ConvertToTime2(Value(Tag_StartTime), "%d-%d-%d %d:%d:%d", true);
42  m_endDate = utl.ConvertToTime2(Value(Tag_EndTime), "%d-%d-%d %d:%d:%d", true);
43 
44  if (m_startDate == -1 || m_endDate == -1) return false;
45 
46  // make sure the start and end times are in the proper order
47  if (m_endDate < m_startDate)
48  {
49  time_t tmp = m_startDate;
50  m_startDate = m_endDate;
51  m_endDate = tmp;
52  }
53 
54  m_mode = GetUpper(Value(Tag_Mode));
55 
56  //read interval
57  vector<string> dtList = utl.SplitString(Value(Tag_Interval), ',');
58  m_dtHs = atoi(dtList[0].c_str());
59  m_dtCh = m_dtHs;
60  if (dtList.size() > 1)
61  m_dtCh = atoi(dtList[1].c_str());
62  // convert the time interval to seconds to conform to time_t struct
63  if (StringMatch(m_mode, Tag_Mode_Daily))
64  {
65  m_dtHs = 86400; // 86400 secs is 1 day
66  m_dtCh = 86400;
67  }
68  return true;

127 void SettingsInput::ReadSiteList()
128 {
129  bson_t *query;
130  query = bson_new();
131 
132  // subbasin id
133  BSON_APPEND_INT32(query, Tag_SubbasinId, m_subbasinID);
134  // mode
135  BSON_APPEND_UTF8(query, Tag_Mode, m_mode.c_str());
136 
137  //string siteListTable = DB_TAB_SITELIST;
138  bool stormMode = false;
139  if (StringMatch(m_mode, Tag_Mode_Storm))
140  stormMode = true;
141  mongoc_cursor_t *cursor;
142  mongoc_collection_t *collection;
143  const bson_t *doc;
144  collection = mongoc_client_get_collection(m_conn,m_dbName.c_str(),DB_TAB_SITELIST);
145  cursor = mongoc_collection_find(collection,MONGOC_QUERY_NONE,0,0,0,query,NULL,NULL);
146  while(mongoc_cursor_next(cursor,&doc)){
147  bson_iter_t iter;
148  if (bson_iter_init (&iter, doc) && bson_iter_find (&iter,MONG_SITELIST_DB)){
149  m_dbHydro = GetStringFromBSONITER(&iter);
150  }
151  else
152  throw ModelException("SettingsInput", "ReadSiteList", "The DB field does not exist in SiteList table.");
153  string siteList = "";
154  if (bson_iter_init (&iter, doc) && bson_iter_find (&iter,SITELIST_TABLE_M))
155  {
156  siteList = GetStringFromBSONITER(&iter);
157  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_MeanTemperature, m_startDate, m_endDate);
158  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_MaximumTemperature, m_startDate, m_endDate);
159  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_MinimumTemperature, m_startDate, m_endDate);
160  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_WindSpeed, m_startDate, m_endDate);
161  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_SolarRadiation, m_startDate, m_endDate);
162  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_RelativeAirMoisture, m_startDate, m_endDate);
163  }
164 
165  if (bson_iter_init (&iter, doc) && bson_iter_find (&iter,SITELIST_TABLE_P)) {
166  siteList = GetStringFromBSONITER(&iter);
167  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_Precipitation, m_startDate, m_endDate, stormMode);
168  }
169 
170  if (bson_iter_init (&iter, doc) && bson_iter_find (&iter,SITELIST_TABLE_PET)) {
171  siteList = GetStringFromBSONITER(&iter);
172  m_inputStation->ReadSitesData(m_dbHydro, siteList, DataType_PotentialEvapotranspiration, m_startDate, m_endDate, stormMode);
173  }
174  }
175 
176  bson_destroy(query);
177  mongoc_cursor_destroy(cursor);
178  mongoc_collection_destroy(collection);
179 }
