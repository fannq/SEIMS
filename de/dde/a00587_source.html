163 int Denitrification::Execute()
164 {
165  //check the data
166  CheckInputData();
167 
168  initalOutputs();
169 
170  #pragma omp parallel for
171  for(int i=0; i < m_nLayers; i++)
172  {
173  for(int j=0; j < m_size; j++)
174  {
175  float denWF = 1.1f;
176  if (m_denWF != NULL)
177  denWF = m_denWF[i][j];
178  float orgCar = 1.12f;
179  if (m_orgCar != NULL)
180  orgCar = m_orgCar[i][j];
181 
182  if(m_WF[i][j] < denWF)
183  m_denLostN[i][j] = 0.0f;
184  else
185  m_denLostN[i][j] = m_Nitrate[i][j] * (1 - exp(- m_denRC * m_TF[i][j] * orgCar));
186 
187  m_Nitrate[i][j] -= m_denLostN[i][j];
188  }
189  }
190 
191  return 0;
192 }
