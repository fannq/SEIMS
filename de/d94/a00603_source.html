129 int PETPriestleyTaylor::Execute()
130 {
131  if(!this->CheckInputData()) return false;
132  if(this->m_pet == NULL)
133  {
134  this->m_pet = new float[this->m_size];
135  }
136  int d = JulianDay(this->m_date);
137  #pragma omp parallel for
138  for (int i = 0; i < m_size; ++i)
139  {
142  float raShortWave = m_sr[i] * (1.0f - 0.23f);
143  float tMean = (m_tMax[i] + m_tMin[i]) / 2;
144  //if the mean T < T_snow, consider the snow depth is larger than 0.5mm.
145  if(tMean < this->m_tSnow)
146  raShortWave = m_sr[i] * (1.0f - 0.8f);
147 
149  float srMax = MaxSolarRadiation(d,this->m_latitude[i]);
150 
153  float satVaporPressure = SaturationVaporPressure(tMean);
154  float actualVaporPressure = m_rhd[i] * satVaporPressure;
155  float rbo = -(0.34f - 0.139f * sqrt(actualVaporPressure));
156  //cloud cover factor
157  float rto = 0.0f;
158  if (srMax >= 1.0e-4)
159  rto = 0.9f * (m_sr[i] / srMax) + 0.1f;
160  //net long-wave radiation
161  float tk = tMean + 273.15f;
162  float raLongWave = rbo * rto * 4.9e-9f * pow(tk, 4);
163 
164  float raNet = raShortWave + raLongWave;
165 
167  //calculate the slope of the saturation vapor pressure curve
168  float dlt = 4098.f* satVaporPressure / pow((tMean + 237.3f), 2);
169 
170  //calculate latent heat of vaporization(MJ/kg, from swat)
171  float latentHeat = LatentHeatVapor(tMean);
172 
173  //calculate mean barometric pressure(kPa)
174  float pb = MeanBarometricPressure(m_elev[i]);
175  //psychrometric constant(kPa/deg C)
176  float gma = 1.013e-3f * pb / (0.622f * latentHeat);
177  float pet_alpha = 1.28f;
178 
179  float petValue = pet_alpha * (dlt / (dlt + gma)) * raNet / latentHeat;
180  m_pet[i] = m_petFactor * max(0.0f, petValue);
181  }
182  return 0;
183 }
