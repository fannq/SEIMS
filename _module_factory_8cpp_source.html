00818                 if (myfile.good())

00820                     getline(myfile, line);

00824                     {
00825                         // parse the line into separate item
00826                         vector<string> tokens = utl.SplitString(line, '|');
00827                         // is there anything in the token list?
00828                         if (tokens.size() > 0)
00829                         {
00830                             for (size_t i=0; i<tokens.size(); i++)
00831                             {
00832                                 //utl.TrimSpaces(tokens[i]);
00833                                 tokens[i] = trim(tokens[i]);
00834                             }
00835                             // is there anything in the first item?
00836                             if (tokens[0].size() > 0)
00837                             {
00838                                 // there is something to add so resize the header list to append it
00839                                 int sz = settings.size(); // get the current number of rows
00840                                 settings.resize(sz+1);      // resize with one more row
00841 
00842                                 settings[sz] = tokens;
00843 
00844                                 bStatus = true; // consider this a success
00845                             } // if there is nothing in the first item of the token list there is nothing to add to the header list
00846                         }
00847                     }
00848                 }
00849             }
00850             bStatus = true;
00851             myfile.close();
00852         }
00853     }
00854     catch (...)
00855     {
00856         myfile.close();
