00048 int Percolation::Execute()
00049 {
00050 
00051     if(m_recharge == NULL) 
00052     {
00053         CheckInputData();
00054         m_recharge = new float[this->m_size];   
00055     }
00056     
00057 #pragma omp parallel for
00058     for(int i = 0; i < m_size; i++)
00059     {
00060         //if(this->m_SoilT[i] <= this->m_ForzenT)   //if the soil temperature is lower than tFrozen, then PERC = 0.
00061         //{
00062         //  m_recharge[i] = 0.0f;
00063         //  continue;
00064         //}
00065 
00066         float moisture = m_Moisture[i] ; 
00067 
00068         //float temp = m_Porosity[i] - m_Residual[i];
00069         //if(temp <= 0.0f) 
00070         //  temp = 0.001f;
00071 
00072         m_recharge[i] = 0.f;
00073         if(moisture > m_FieldCapacity[i])
00074         {
00075             // the water exceeds the porosity is added to percolation directly
00076             if(moisture > m_Porosity[i]) 
00077             {
00078                 m_recharge[i] += (moisture - m_Porosity[i]) * m_rootDepth[i];
00079                 m_Moisture[i] = m_Porosity[i];
00080             }
00081 
00082             // recharge capacity (mm)
00083             float dcIndex = 3.f + 2.f/m_Poreindex[i]; // pore disconnectedness index
00084             //float rechargeCap = m_Conductivity[i] / 3600.f * m_timestep * pow((moisture - m_Residual[i])/temp, dcIndex); 
00085             float rechargeCap = m_Conductivity[i] / 3600.f * m_timestep * pow(moisture/m_Porosity[i], dcIndex); //Campbell, 1974
00086             float availableWater = (m_Moisture[i] - m_FieldCapacity[i])*m_rootDepth[i];
00087             if (rechargeCap >= availableWater)
00088                 rechargeCap = availableWater;
00089             
00090             m_recharge[i] += rechargeCap;
00091             m_Moisture[i] -= m_recharge[i]/m_rootDepth[i];
00092         }
00093     }
00094 
00095     return true;
00096 
00097 }
