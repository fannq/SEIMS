00023 void BucketSortKeysInc(ctrl_t *ctrl, idx_t n, idx_t max, idx_t *keys, 
00024          idx_t *tperm, idx_t *perm)
00025 {

00027   idx_t *counts;

00031   counts = iset(max+2, 0, iwspacemalloc(ctrl, max+2));
00032 
00033   for (i=0; i<n; i++)
00034     counts[keys[i]]++;
00035   MAKECSR(i, max+1, counts);
