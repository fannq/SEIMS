158 float* SurrunoffTransfer::CalculateEnrRatio(){
159  //enrichment ratio
160  float* enratio;
161  for(int i = 0; i < m_nCells; i++) {
162  if (m_sedimentYield[i] < 1e-4) {
163  m_sedimentYield[i] = 0;
164  }
165  // CREAMS method for calculating enrichment ratio
166  float cy = 0;
167  // Calculate sediment calculations, equation 4:2.2.3 in SWAT Theory 2009, p272
168  cy = 0.1 * m_sedimentYield[i] / (m_cellWidth * m_cellWidth * 0.0001 * m_surfr[i] + 1e-6);
169  if (cy > 1e-6) {
170  enratio[i] = 0.78 * pow(cy, -0.2468f);
171  } else {
172  enratio[i] = 0;
173  }
174  if (enratio[i] > 3.5) {
175  enratio[i] = 3.5;
176  }
177  }
178  return enratio;

180 void SurrunoffTransfer::OrgnRemoveinSr(){
181  //enrichment ratio (enratio)
182  float* enratio = CalculateEnrRatio();
183  for(int i = 0; i < m_nCells; i++) {
184  for(int k = 0; k < m_nSolLyrs; k++) {
185  //amount of organic N in first soil layer (orgninfl)
186  float orgninfl = 0;
187  //conversion factor (wt)
188  float wt = 0;
189  orgninfl = m_sol_orgn[i][0] + m_sol_aorgn[i][0] + m_sol_fon[i][0];
190  wt = m_sol_bd[i][0] * m_sol_z[i][0] / 100;
191  //concentration of organic N in soil (concn)
192  float concn = 0;
193  concn = orgninfl * enratio[i] / wt;
194  //Calculate the amount of organic nitrogen transported with sediment to the stream, equation 4:2.2.1 in SWAT Theory 2009, p271
195  m_sedorgn[i] = 0.001 * concn * m_sedimentYield[i] / (m_cellWidth * m_cellWidth * m_nCells);
196  //update soil nitrogen pools
197  if(orgninfl > 1e-6) {
198  m_sol_aorgn[i][0] = m_sol_aorgn[i][0] - m_sedorgn[i] * (m_sol_aorgn[i][0] / orgninfl);
199  m_sol_orgn[i][0] = m_sol_orgn[i][0] - m_sedorgn[i] * (m_sol_orgn[i][0] / orgninfl);
200  m_sol_fon[i][0] = m_sol_fon[i][0] - m_sedorgn[i] * (m_sol_fon[i][0] / orgninfl);
201  if (m_sol_aorgn[i][0] < 0) {
202  m_sedorgn[i] = m_sedorgn[i] + m_sol_aorgn[i][0];
203  m_sol_aorgn[i][0] = 0;
204  }
205  if (m_sol_orgn[i][0] < 0) {
206  m_sedorgn[i] = m_sedorgn[i] + m_sol_orgn[i][0];
207  m_sol_orgn[i][0] = 0;

218 void SurrunoffTransfer::OrgpAttachedtoSed(){
219  float* enratio = CalculateEnrRatio();
220  for(int i = 0; i < m_nCells; i++) {
221  for(int k = 0; k < m_nSolLyrs; k++) {
222  //amount of phosphorus attached to sediment in soil (sol_attp)
223  float sol_attp = 0;
224  //fraction of active mineral/organic/stable mineral phosphorus in soil (sol_attp_o, sol_attp_a, sol_attp_s)
225  float sol_attp_o = 0;
226  float sol_attp_a = 0;
227  float sol_attp_s = 0;
228  //Calculate sediment
229  sol_attp = m_sol_orgp[i][0] + m_sol_fop[i][0] + m_sol_mp[i][0] + m_sol_actp[i][0] + m_sol_stap[i][0];
230  if (sol_attp > 1e-3) {
231  sol_attp_o = (m_sol_orgp[i][0] + m_sol_fop[i][0]+ m_sol_mp[i][0]) / sol_attp;
232  sol_attp_a = m_sol_actp[i][0] / sol_attp;
233  sol_attp_s = m_sol_stap[i][0] / sol_attp;
234  }
235  //conversion factor (mg/kg => kg/ha) (wt)
236  float wt = m_sol_bd[i][0] * m_sol_z[i][0] / 100;
237  //concentration of organic P in soil (concp)
238  float concp = 0;
239  concp = sol_attp * enratio[i] / wt;
240  //total amount of P removed in sediment erosion (sedp)
241  float sedp = 0.001 * concp * m_sedimentYield[i] / (m_cellWidth * m_cellWidth);
242  m_sedorgp[i] = sedp * sol_attp_o;
243  m_sedminpa[i] = sedp * sol_attp_a;
244  m_sedminps[i] = sedp * sol_attp_s;
245  //modify phosphorus pools
246  //total amount of P in mineral sediment pools prior to sediment removal (psedd)
247  float psedd = 0;
248  //total amount of P in organic pools prior to sediment removal (porgg)
249  float porgg = 0;
250  psedd = m_sol_actp[i][0] + m_sol_stap[i][0];
251  porgg = m_sol_orgp[i][0] + m_sol_fop[i][0];
252  if (porgg > 1e-3) {
253  m_sol_orgp[i][0] = m_sol_orgp[i][0] - m_sedorgp[i] * (m_sol_orgp[i][0] / porgg);
254  m_sol_fop[i][0] = m_sol_fop[i][0] - m_sedorgp[i] * (m_sol_fop[i][0] / porgg);
255  m_sol_mp[i][0] = m_sol_mp[i][0] - m_sedorgp[i] * (m_sol_mp[i][0] / porgg);
256  }
257  m_sol_actp[i][0] = m_sol_actp[i][0] - m_sedminpa[i];
258  m_sol_stap[i][0] = m_sol_stap[i][0] - m_sedminps[i];
259  if (m_sol_orgp[i][0] < 0) {
260  m_sedorgp[i] = m_sedorgp[i] + m_sol_orgp[i][0];
261  m_sol_orgp[i][0] = 0;
262  }
