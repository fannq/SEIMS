202 void SedBonNutrient_OL::concentration(int id)
203 {
204  float Fcl = m_clayFrac[id];
205  float K_NH4 = 1.34f + 0.038f*Fcl; // partition coefficient
206  float K_P = 100 + 2.5f * Fcl; // partition coefficient
207  // sendiment bound nutrient concentration
208  //concentration at the end of the time step
209  float Q = m_CELLQ[id]; //m3/s
210  float h = m_CELLH[id] * m_cellA[id]/1000; //m3
211  float f = m_infil[id] * m_cellA[id]/1000; //m3
212  float S = m_sd[id] * m_cellA[id]/1000; //m3
213  float p = m_Porosity[id];
214  if(S!=0)
215  {
216  if (h==0)
217  {
218  h = m_whtoCh[id];
219  }
220  m_X2 = (h + f) / S;
221  m_X3 = h / S;
222  }
223  else
224  {
225  m_X2 = 0;
226  m_X3 = 0;
227  }
228  float temp1 = Q * m_dt + f + S; //m3
229  float temp2 = 2.65f*(1-p);
230  float X_NH4 = temp1 /(p + temp2 * K_NH4);
231  float Cfin_NH4 = m_PreConcNH4[id] * exp(-X_NH4); //kg/kg-soil
232 
233  float X_P = temp1 /(p + temp2 * K_P);
234  float Cfin_P = m_PreConcP[id] * exp(-X_P); //kg/kg-soil
235 
236  float V = m_soilMoisture[id] * 1000; //kg/m3
237  float M = m_Density[id]; //kg/m3
238  float P0_NH4 = K_NH4 * M * Cfin_NH4 / V; //kg/kg-soil
239  float P0_P = K_NH4 * M * Cfin_NH4 / V; //kg/kg-soil
240  // newly generated sediment-bound nutrient (kg/s)
241  float Sed = m_SedKG[id] / m_dt; //newly generated sediment (kg/s)
242  m_CP_Sed[id] = P0_P * Sed;
243  m_CNH4_Sed[id] = P0_NH4 * Sed;
244 }

247  void SedBonNutrient_OL::CalcuVelocityChannelFlow()
248 {
249  const float beta = 0.6f;
250  const float _23 = 2.0f/3;
251  float Perim,R,S,n;
252 
253  for (int i=0; i<m_nCells; i++)
254  {
255  if (m_whtoCh[i] > 0.0001f)
256  {
257  Perim = 2 * m_whtoCh[i]/1000 + m_chWidth[i]; // mm to m -> /1000
258  if (Perim > 0)
259  R = m_whtoCh[i]/1000 * m_chWidth[i] / Perim;
260  else
261  R = 0.0f;
262  S = sin(atan(m_Slope[i])); //sine of the slope
263  S = max(0.001f, S);
264  n = m_ManningN[i]; //manning n
265  m_ChV[i] = pow(R, _23) * sqrt(S)/n;
266  }
267  else
268  m_ChV[i] = 0;
269  }

272 void SedBonNutrient_OL::OverlandFlow(int id)
273 {
274  concentration(id);
275  //sum the upstream overland flow
276  float flowwidth = m_flowWidth[id];
277  float PUp = 0.0f;
278  float NH4Up = 0.0f;
279  for (int k = 1; k <= (int)m_flowInIndex[id][0]; ++k)
280  {
281  int flowInID = (int)m_flowInIndex[id][k];
282  PUp += m_CP_Sed[flowInID];
283  NH4Up += m_CNH4_Sed[flowInID];
284  }
285  // if the channel width is greater than the cell width
286  if (m_streamLink[id] >= 0 && flowwidth <=0)
287  {
288  m_SedBonPToCh[id] = PUp * m_dt; // convert to kg
289  m_SedBonAmmoniumToCh[id] = NH4Up * m_dt;
290  return;
291  }
292 
293  //An initial test is done on the discharge at the end of the time interval. When the discharge is equal to zero
294  if (m_CP_Sed[id]==0)
295  {
296  float Pinflow = PUp;
297  float NH4inflow = NH4Up ;
298  //The amount of sediment bound nutrient in transit
299  float Ptransit = (m_PPT_P[id] + Pinflow)/2;
300  float NH4transit = (m_PPT_NH4[id] + NH4inflow)/2;
301  //The outflow of sediment bound nutrient kg/s
302  m_SedBonP[id] = 0;
303  m_SedBonAmmonium[id] = 0;
304  //The new value of PPT
305  m_PPT_P[id] = Pinflow;
306  m_PPT_NH4[id] = NH4inflow;
307  }
308  //When the discharge is positive
309  else
310  {
311  float Pinflow = PUp + m_CP_Sed[id];
312  float NH4inflow = NH4Up + m_CNH4_Sed[id];
313  //The amount of sediment bound nutrient in transit
314  float Ptransit = (m_PPT_P[id] + Pinflow)/(1 + m_X1);
315  float NH4transit = (m_PPT_NH4[id] + NH4inflow)/(1 + m_X1);
316  //The outflow of sediment bound nutrient kg/s
317  float Q = m_CELLQ[id]; //m3/s
318  float S = m_sd[id] * m_cellA[id]/1000; //m3
319  m_SedBonP[id] = (Ptransit * Q) / S;
320  m_SedBonAmmonium[id] = (NH4transit * Q) / S;
321  //The new value of PPT
322  float ratio = max(0.0f, 1-m_X1);
323  m_PPT_P[id] = Pinflow + m_SedBonP[id]*ratio;
324  m_PPT_NH4[id] = NH4inflow + m_SedBonAmmonium[id]*ratio;
325  }
326  // compute nutrients to channel
327  float fractiontochannel = 0.0f;
328  if (m_chWidth[id] > 0)
329  {
330  float tem = m_ChV[id] * m_dt;
331  fractiontochannel = 2*tem / (m_CellWidth - m_chWidth[id]);
332  fractiontochannel = min(fractiontochannel, 1.0f);
333  // convert to kg
334  m_SedBonPToCh[id] = m_SedBonP[id] * m_dt * fractiontochannel;
335  m_SedBonAmmoniumToCh[id] = m_SedBonAmmonium[id] * m_dt * fractiontochannel;
336  }

339 int SedBonNutrient_OL::Execute()
340 {
341  //check the data
342  CheckInputData();
343 
344  initalOutputs();
345 
346  for (int iLayer = 0; iLayer < m_nLayers; ++iLayer)
347  {
348  // There are not any flow relationship within each routing layer.
349  // So parallelization can be done here.
350  int nCells = (int)m_routingLayers[iLayer][0];
351  //omp_set_num_threads(2);
352  #pragma omp parallel for
353  for (int iCell = 1; iCell <= nCells; ++iCell)
354  {
355  int id = (int)m_routingLayers[iLayer][iCell];
356  OverlandFlow(id);
357  }
358  }
359  return 0;
360 }
