83 int DepressionFSDaily::Execute()
84 {
85  //check the data
86  CheckInputData();
87 
88  initalOutputs();
89 
90  #pragma omp parallel for
91  for (int i = 0; i < m_size; ++i)
92  {
94  // runoff
95  if (m_depCap[i] < 0.001f)
96  {
97  m_sr[i] = m_pe[i];
98  m_sd[i] = 0.f;
99  }
100  else if (m_pe[i] > 0)
101  {
102  float pc = m_pe[i] - m_depCap[i] * log(1.f - m_sd[i]/m_depCap[i]);
103  float deltaSd = m_pe[i]*exp(-pc/m_depCap[i]);
104  if (deltaSd > m_depCap[i] - m_sd[i])
105  deltaSd = m_depCap[i] - m_sd[i];
106  m_sd[i] += deltaSd;
107  m_sr[i] = m_pe[i] - deltaSd;
108  }
109  else
110  {
111  m_sd[i] += m_pe[i];
112  m_sr[i] = 0.f;
113  }
114 
116  // evaporation
117  if (m_sd[i] > 0)
118  {
119  // evaporation from depression storage
120  if (m_pet[i] - m_ei[i] < m_sd[i])
121  {
122  m_ed[i] = m_pet[i] - m_ei[i];
123  }
124  else
125  {
126  m_ed[i] = m_sd[i];
127  }
128  m_sd[i] -= m_ed[i];
129  }
130  else
131  {
132  m_ed[i] = 0.0f;
133  m_sd[i] = 0.0f;
134  }
135  }
136 
137  return true;
138 }
