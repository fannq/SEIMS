105 void SOL_WB::setValueToSubbasin()
106 {
107 
108  if(m_subbasinList != NULL)
109  {
110  if(m_soilWaterBalance == NULL)
111  {
112  m_soilWaterBalance = new float*[m_subbasinSelectedCount];
113  for(int i=0;i<m_subbasinSelectedCount;i++)
114  {
115  m_soilWaterBalance[i] = new float[18];
116  }
117  }
118 
119  int index = 0;
120  float ri, sm;
121  map<int,subbasin*>::iterator it;
122  for(it=m_subbasinList->begin();it!=m_subbasinList->end();it++)
123  {
124  it->second->clear();
125  vector<int>* cells = it->second->getCells();
126  vector<int>::iterator itCells;
127  for(itCells=cells->begin(); itCells<cells->end(); itCells++)
128  {
129  int cell = *itCells;
130  float depth[2];
131  depth[0] = m_upSoilDepth;
132  depth[1] = m_Rootdepth[cell] - m_upSoilDepth;
133 
134  ri = 0.f;
135  sm = 0.f;
136  for(int i = 0; i < m_nSoilLayers; i++)
137  {
138  if(m_RI != NULL)
139  ri += m_RI[cell][i];
140  sm += m_sm[cell][i] * depth[i];
141  }
142  sm /= m_Rootdepth[cell];
143 
144  it->second->addP(m_Precipitation[cell]);
145  it->second->addNetP(m_pNet[cell]);
146  it->second->addDepressionET(m_ED[cell]);
147  it->second->addInfiltration(m_Infil[cell]);
148  it->second->addInterception(m_Interception[cell]);
149  it->second->addInterceptionET(m_EI[cell]);
150  it->second->addNetPercolation(m_Percolation[cell][m_nSoilLayers-1] - m_Revap[cell]);
151  //it->second->addR(m_RS[cell] + m_RI[cell] + m_RG[cell]);
152  it->second->addRevap(m_Revap[cell]);
153  //it->second->addRG(m_RG[cell]);
154 
155  it->second->addRI(ri);
156  it->second->addRS(m_RS[cell]);
157  it->second->addSMOI(sm);
158  //it->second->addTotalET(m_EI[cell] + m_ED[cell] + m_ES[cell] + m_SE[cell]);
159  it->second->addTotalET(m_EI[cell] + m_ED[cell] + m_ES[cell]);
160  it->second->addSoilET(m_ES[cell]);
161  it->second->addT(m_tMax[cell] / 2 + m_tMin[cell] / 2);
162  it->second->addSoilT(m_SoilT[cell]);
163  it->second->addMoistureDepth(m_Rootdepth[cell]*sm);
164  }
165 
166  m_soilWaterBalance[index][0] = it->second->getAverage("P");
167  m_soilWaterBalance[index][1] = it->second->getAverage("T");
168  m_soilWaterBalance[index][2] = it->second->getAverage("Soil_T");
169  m_soilWaterBalance[index][3] = it->second->getAverage("NetP");
170  m_soilWaterBalance[index][4] = it->second->getAverage("InterceptionET");
171  m_soilWaterBalance[index][5] = it->second->getAverage("DepressionET");
172  m_soilWaterBalance[index][6] = it->second->getAverage("Infiltration");
173  m_soilWaterBalance[index][7] = it->second->getAverage("Total_ET");
174  m_soilWaterBalance[index][8] = it->second->getAverage("Soil_ET");
175  m_soilWaterBalance[index][9] = it->second->getAverage("Net_Percolation");
176  m_soilWaterBalance[index][10] = it->second->getAverage("Revap");
177  m_soilWaterBalance[index][11] = it->second->getAverage("RS");
178  m_soilWaterBalance[index][12] = it->second->getAverage("RI");
179 
180  int id = it->second->getId();
181  if(id >= m_subbasinTotalCount)
182  throw ModelException("SOL_WB","setValueToSubbasin","The groundwater runoff for subbasin " + SOL_WB::toString(float(id)) + " does not exist. Please check the input variable named T_RG from groundwater module.");
183  m_soilWaterBalance[index][13] = m_RG[id];
184  m_soilWaterBalance[index][14] = m_soilWaterBalance[index][11] + m_soilWaterBalance[index][12] + m_soilWaterBalance[index][13];
185  m_soilWaterBalance[index][15] = it->second->getAverage("S_MOI");
186  m_soilWaterBalance[index][16] = it->second->getAverage("MoistureDepth");
187 
188  index++;
189  }
190  }

335 void SOL_WB::getSubbasinList(int cellCount, float* subbasinGrid, int subbasinSelectedCount, float* subbasinSelected)
336 {
337  if(m_subbasinList != NULL) return;
338  if(subbasinSelected == NULL) return;
339 
340  map<int,bool> selected;
341  bool isAllNeedStatistc = false;
342  for(int i = 0;i < subbasinSelectedCount; i++)
343  {
344  int subid = int(subbasinSelected[i]);
345  selected[subid] = true;
346  if(subid == 0)
347  isAllNeedStatistc = true;
348  }
349 
350  //map<int,subbasin*> list;
351  m_subbasinList = new map<int,subbasin*>();
352  if(isAllNeedStatistc) (*m_subbasinList)[0] = new subbasin(0);
353  for(int i = 0;i < cellCount; i++)
354  {
355  if(isAllNeedStatistc)
356  (*m_subbasinList)[0]->addCell(i);
357 
358  int subid = int(subbasinGrid[i]);
359  if(!selected[subid]) continue;
360 
361  map<int,subbasin*>::iterator it = (*m_subbasinList).find(subid);
362  if(it == (*m_subbasinList).end()) (*m_subbasinList)[subid] = new subbasin(subid);
363  (*m_subbasinList)[subid]->addCell(i);
364  }
365 }
