69 void MUSLE_I30::initalOutputs()
70 {
71  if(m_sedimentYield == NULL) m_sedimentYield = new float[this->m_cellSize];
72  if(m_usle_ls == NULL)
73  {
74  float constant = pow(22.13f,0.4f);
75  m_usle_ls = new float[this->m_cellSize];
76  for(int i=0;i<this->m_cellSize;i++)
77  {
78  float lambda_i1 = this->m_flowacc[i] * this->m_cellWidth;
79  float lambda_i = lambda_i1 + this->m_cellWidth;
80  float L = pow(lambda_i,1.4f) - pow(lambda_i1,1.4f);
81  L /= this->m_cellWidth * constant;
82 
83  float S = pow(this->m_slope[i] / 100.0f / 0.0896f,1.3f);
84 
85  this->m_usle_ls[i] = L * S;//equation 3 in memo, LS factor
86  }
87  }
88  if(m_slopeForPq == NULL)
89  {
90  m_slopeForPq = new float[this->m_cellSize];
91  for(int i=0;i<this->m_cellSize;i++)
92  {
93  m_slopeForPq[i] = pow(this->m_slope[i] / 100.0f * 1000.0f,0.16f);
94  }
95  }
96  m_cellAreaKM = pow(this->m_cellWidth/1000.0f,2.0f);
97 
98 
99  //initial average half-hour rainfall fraction for each month
100  if(m_alpha_month == NULL)
101  {
102  m_alpha_month = new float[12];
103 
104  //first smooth the extreme half-hour rainfall values, eq.1:3.2.1 p65
105  for(int i=0;i<12;i++)
106  {
107  if(i == 0) m_alpha_month[i] = this->m_p_stat[11][0] + this->m_p_stat[0][0] + this->m_p_stat[1][0];
108  else if(i == 11) m_alpha_month[i] = this->m_p_stat[10][0] + this->m_p_stat[11][0] + this->m_p_stat[0][0];
109  else m_alpha_month[i] = this->m_p_stat[i-1][0] + this->m_p_stat[i][0] + this->m_p_stat[i+1][0];
110 
111  m_alpha_month[i] /= 3.0f;
112  }
113 
114  //get alpha, eq.1:3.2.2 p66
115  for(int i=0;i<12;i++)
116  {
117  float alpha = 0.0f;
118  float mu = this->m_p_stat[i][1] / this->m_p_stat[i][2];
119  alpha = log(0.5f / this->m_rain_yrs / this->m_p_stat[i][2]);
120  alpha *= mu;
121  alpha = exp(m_alpha_month[i] / alpha);
122  alpha = this->m_adj_pkr * (1 - alpha);
123 
124  m_alpha_month[i] = alpha;
125  }
126  }
127 
128  //rndseed(6,j) = 1094585182 gcycl.f Line73
129  m_rndseed = 1094585182;
130 }

132 float MUSLE_I30::getPeakRunoffRate(int cell)
133 {
134  float p = this->m_p[cell];
135  float snomlt = this->m_snowMelt[cell];
136  if(p>snomlt) p -= snomlt; //minus snow melt
137  else p = 0.0f;
138 
139  struct tm datestruture;
140  localtime_s(&datestruture,&m_Date); //get month
141  float max = 1.0f - exp(-125.0f / (p + 5)); //eq.1:3.2.3 p66
142  float a15 = triangularDistribution(0.02083f,m_alpha_month[datestruture.tm_mon],max,&m_rndseed); //eq.1:3.2.4-5 p67
143 
144  float t = this->m_t_concentration[cell];
145  if(t<0.1f) t = 0.1f; //make sure t >= 0.1
146  float altc = 1.0f - exp(2.0f * t * log(1.0f-a15)); //eq.2:1.3.19 p111
147  float peakr = altc * this->m_surfaceRunoff[cell] * m_cellAreaKM / 3.6f / t;//eq.2:1.3.20 p111
148 
149  return peakr;
150 }

163 float MUSLE_I30::triangularDistribution(float min, float mean, float max, int* seed)
164 {
165  float at1 = min;
166  float at2 = mean;
167  float at3 = max;
168  int* at4i = seed;
169  float u3, rn, y, b1, b2, x1, xx, yy, amn;
170  float atri;
171 
172  u3 = 0.0f;
173  rn = 0.0f;
174  y = 0.0f;
175  b1 = 0.0f;
176  b2 = 0.0f;
177  x1 = 0.0f;
178  xx = 0.0f;
179  yy = 0.0f;
180  amn = 0.0f;
181 
182  u3 = at2 - at1;
183  rn = aunif(at4i);
184  y = 2.0f / (at3 - at1);
185  b2 = at3 - at2;
186  b1 = rn / y;
187  x1 = y * u3 / 2.0f;
188 
189  if (rn <= x1)
190  {
191  xx = 2.0f * b1 * u3;
192  if (xx <= 0.0f)
193  yy = 0.0f;
194  else
195  yy = sqrt(xx);
196 
197  atri = yy + at1;
198  }
199  else
200  {
201  xx = b2 * b2 - 2.0f * b2 * (b1 - 0.5f * u3);
202  if (xx <= 0.0f)
203  yy = 0.0f;
204  else
205  yy = sqrt(xx);
206 
207  atri = at3 - yy;
208  }
209 
210  amn = (at3 + at2 + at1) / 3.0f;
211  atri = atri * at2 / amn;
212 
213  if (atri >= 1.0) return 0.99f;
214  if (atri <= 0.0) return 0.001f;
215  return atri;
216 }

218 int MUSLE_I30::Execute()
219 {
220  this->CheckInputData();
221 
222  this->initalOutputs();
223 
224  for(int i=0;i<this->m_cellSize;i++)
225  {
226  if(this->m_surfaceRunoff[i] < 0.0001f) this->m_sedimentYield[i] = 0.0f;
227  else
228  {
229  float q = getPeakRunoffRate(i); //equation 2 in memo, peak flow
230  float Y = 11.8f * pow(this->m_surfaceRunoff[i] * m_cellAreaKM * 1000.0f * q, 0.56f)
231  * this->m_usle_k[i] * this->m_usle_ls[i] * this->m_usle_c[i] * this->m_usle_p[i]; //equation 1 in memo, sediment yield
232 
233  if(this->m_snowAccumulation[i] > 0.0001f) Y /= exp(3.0f * this->m_snowAccumulation[i] / 25.4f); //equation 4 in memo, the snow pack effect
234  this->m_sedimentYield[i] = Y;
235  }
236  }
237  return 0;
238 }
