77 float PETHargreaves::MaxSolarRadiation(int day,float lat)
78 {
79  lat = lat*3.1415926/180;
80  //Calculate Daylength
81  //calculate solar declination: equation 2.1.2 in SWAT manual
82  float sd = asin(0.4f * sin((day - 82.0f) / 58.09f)); //365/2pi = 58.09
83 
84  //calculate the relative distance of the earth from the sun
85  //the eccentricity of the orbit
86  //equation 2.1.1 in SWAT manual
87  float dd = 1.0f + 0.033f * cos(day / 58.09f);
88 
89  //daylength = 2 * Acos(-Tan(sd) * Tan(lat)) / omega
90  //where the angular velocity of the earth's rotation, omega, is equal
91  //to 15 deg/hr or 0.2618 rad/hr and 2/0.2618 = 7.6374
92  //equation 2.1.6 in SWAT manual
93 
94  float h = 0.0f;
95  float ch = -sin(lat) * tan(sd) / cos(lat);
96  if (ch > 1.f) //ch will be >= 1. if latitude exceeds +/- 66.5 deg in winter
97  h = 0.0f;
98  else if (ch >= -1.0f)
99  h = acos(ch);
100  else
101  h = 3.1416f; //latitude exceeds +/- 66.5 deg in summer
102 
103  float dayl = 7.6394f * h;
104 
105  //Calculate Potential (maximum) Radiation !!
106  //equation 2.2.7 in SWAT manual
107  float ys = sin(lat) * sin(sd);
108  float yc = cos(lat) * cos(sd);
109 
110  return 30.f * dd * (h * ys + yc * sin(h));
111 }

113 int PETHargreaves::Execute()
114 {
115  if(!this->CheckInputData()) return false;
116  if(this->m_pet == NULL)
117  {
118  this->m_pet = new float[this->m_size];
119  }
120 
121  int d = JulianDay(this->m_date);
122 #pragma omp parallel for
123  for (int i = 0; i < m_size; ++i)
124  {
125  float tMean = (m_tMax[i] + m_tMin[i])/2;
126  float srMax = this->MaxSolarRadiation(d, m_latitude[i]);
127 
128  //calculate latent heat of vaporization(from swat)
129  float latentHeat = 2.501f - 0.002361f * tMean;
130  // extraterrestrial radiation(from swat)
131  // 37.59 is coefficient in equation 2.2.6 !!extraterrestrial
132  // 30.00 is coefficient in equation 2.2.7 !!max at surface
133  float h0 = srMax * 37.59f / 30.0f;
134  float petValue = 0.0023f * h0 * pow(m_tMax[i]-m_tMin[i], 0.5f)
135  * (tMean + 17.8f) / latentHeat;
136  m_pet[i] = m_petFactor * max(0.0f, petValue);
137 
138  //if (FloatEqual(m_pet[i], 0.f))
139  //{
140  // int a = 0.f;
141  //}
142  }
143  return 0;
144 }
