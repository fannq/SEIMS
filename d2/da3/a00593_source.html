165 float ImplicitKinematicWave_OL::GetNewQ(float qIn, float qLast, float surplus, float alpha, float dt, float dx)
166 {
167  /* Using Newton-Raphson Method */
168  float ab_pQ, dtX, C; //auxillary vars
169  int count;
170  float Qkx; //iterated discharge, becomes Qnew
171  float fQkx; //function
172  float dfQkx; //derivative
173  const float _epsilon = 1e-12f;
174  const float beta = 0.6f;
175 
176  /* if no input then output = 0 */
177  if ((qIn + qLast) <= -surplus*dx)//0)
178  {
179  //itercount = -1;
180  return(0);
181  }
182 
183  /* common terms */
184  ab_pQ = alpha*beta*pow(((qLast+qIn)/2),beta-1);
185  // derivative of diagonal average (space-time)
186 
187  dtX = dt/dx;
188  C = dtX*qIn + alpha*pow(qLast,beta) + dt*surplus;
189  //dt/dx*Q = m3/s*s/m=m2; a*Q^b = A = m2; surplus*dt = s*m2/s = m2
190  //C is unit volume of water
191  // first gues Qkx
192  Qkx = (dtX * qIn + qLast * ab_pQ + dt * surplus) / (dtX + ab_pQ);
193 
194  // VJ 050704, 060830 infil so big all flux is gone
195  //VJ 110114 without this de iteration cannot be solved for very small values
196  if (Qkx < MIN_FLUX)
197  {
198  //itercount = -2;
199  return(0);
200  }
201 
202  Qkx = max(Qkx, MIN_FLUX);
203 
204  count = 0;
205  do {
206  fQkx = dtX * Qkx + alpha * pow(Qkx, beta) - C; /* Current k */
207  dfQkx = dtX + alpha * beta * pow(Qkx, beta - 1); /* Current k */
208  Qkx -= fQkx / dfQkx; /* Next k */
209  Qkx = max(Qkx, MIN_FLUX);
210  count++;
211  //qDebug() << count << fQkx << Qkx;
212  } while(fabs(fQkx) > _epsilon && count < MAX_ITERS);
213 
214  if(Qkx != Qkx)
215  throw ModelException(MID_IKW_OL, "GetNewQ", "Error in iteration!");
216 
217  //itercount = count;
218  return Qkx;
219 }

223 void ImplicitKinematicWave_OL::OverlandFlow(int id)
224 {
225  const float beta = 0.6f;
226  const float _23 = 2.0f/3.0f;
227  float beta1 = 1/beta;
228 
229  float h = m_sr[id]/1000.f;
230 
231  float Perim = 2.f*h + m_flowWidth[id];
232  float r = 0;
233  if (Perim > 0)
234  r = h*m_flowWidth[id]/Perim;
235 
236  float sSin = sqrt(sin(m_sRadian[id]));
237  m_alpha[id] = pow(m_n[id]/sSin * pow(Perim, _23),beta);
238 
239  if (m_alpha[id] > 0)
240  m_q[id] = pow((m_flowWidth[id]*h)/m_alpha[id], beta1);
241  else
242  m_q[id] = 0;
243 
244  m_vel[id] = pow(r, _23)*sSin/m_n[id];
245 
246  float flowWidth = m_flowWidth[id];
247  float flowLen = m_flowLen[id];
248 
249  //sum the upstream overland flow
250  float qUp = 0.0f;
251  for (int k = 1; k <= (int)m_flowInIndex[id][0]; ++k)
252  {
253  int flowInID = (int)m_flowInIndex[id][k];
254  if (m_streamLink[flowInID] <= 0) // if the upstream cell is not a channel cell
255  qUp += m_q[flowInID];
256  }
257 
258  // if the channel width is greater than the cell width
259  if (m_streamLink[id] >= 0 && flowWidth <= 0)
260  {
261  m_q[id] = qUp;
262  m_sr[id] = 0.f;
263  return;
264  }
265 
266  // check whether overland flow routing is needed
267  if (qUp < MIN_FLUX && m_q[id] < MIN_FLUX)
268  {
269  m_sr[id] = 0.f;
270  m_q[id] = 0.f;
271  return;
272  }
273 
274  // calcluate infiltration surplus (m2/s)
275  float surplus = 0.f;
276  if (m_infilCapacitySurplus != NULL)
277  surplus = -m_infilCapacitySurplus[id]/1000.f * flowWidth / m_dtStorm;
278 
279  float qIn = m_q[id];
280  m_q[id] = GetNewQ(qUp, qIn, surplus, m_alpha[id], m_dtStorm, flowLen);
281 
282  float hNew = (m_alpha[id]*pow(m_q[id], 0.6f)) / flowWidth; // unit m
283  //float hTest = h + (qUp - m_q[id])*m_dtStorm/(flowWidth*flowLen);
284  m_sr[id] = hNew * 1000;
285 
286  float reInfil = (qUp - m_q[id])*m_dtStorm/(flowWidth*flowLen) + h - hNew;
287  reInfil *= 1000.f;
288 
289  //if (abs(reInfil) < 0.001f)
290  // reInfil = 0.f;
291  //else if(abs(reInfil - m_infilCapacitySurplus[id]) < 0.001f)
292  // reInfil = m_infilCapacitySurplus[id];
293 
294  //if (reInfil < 0.f || reInfil > m_infilCapacitySurplus[id])
295  // throw ModelException(MID_IKW_OL, "OverlandFlow", "Reinfiltration exceeded range!");
296 
297  if (reInfil < 0.f)
298  reInfil = 0.f;
299 
300  m_infil[id] += reInfil;
301  if (m_accumuDepth != NULL)
302  m_accumuDepth[id] += reInfil;
303 
304  if (m_infilCapacitySurplus != NULL)
305  m_infilCapacitySurplus[id] -= reInfil;
306 
307  m_reInfil[id] = reInfil;
308 
309  // compute to channel flow
310  if (m_streamLink[id] > 0)
311  {
312  float fractiontochannel = min(m_dtStorm*m_vel[id]/(0.5f*flowWidth), 1.0f);
313  float Volume = m_sr[id]/1000.f * m_flowWidth[id] * flowLen;
314 
315  //if (id == m_idOutlet)// in catchment outlet cell, throw everything in channel
316  fractiontochannel = 1.0f;
317 
318  m_q[id] += fractiontochannel*Volume/m_dtStorm; // water diverted to the channel
319  m_sr[id] *= (1.f - fractiontochannel);
320 
321  }

326 {

328 
329 
330  for (int iLayer = 0; iLayer < m_nLayers; ++iLayer)
331  {

333  // So parallelization can be done here.
334  int nCells = (int)m_routingLayers[iLayer][0];
335  //omp_set_num_threads(2);
