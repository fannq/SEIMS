19 def GetProperties(s, c, om):
20  #wilting point
21  wpt = -0.024*s + 0.487*c + 0.006*om + 0.005*s*om - 0.013*c*om + 0.068*s*c + 0.031
22  wp = 1.14*wpt - 0.02
23 
24  #field capacity
25  fct = -0.251*s + 0.195*c + 0.011*om + 0.006*s*om - 0.027*c*om + 0.452*s*c + 0.299
26  fc = fct + 1.283*fct*fct - 0.374*fct - 0.015
27 
28  s33t = 0.278*s + 0.034*c + 0.022*om - 0.018*s*om - 0.027*c*om - 0.584*s*c + 0.078
29  s33 = 1.636*s33t - 0.107
30 
31  sat = fc + s33 - 0.097*s + 0.043
32 
33  pn = 2.65*(1.0 - sat)
34 
35  # df means density effects
36  p_df = pn
37  sat_df = 1 - p_df/2.65
38  fc_df = fc - 0.2*(sat - sat_df)
39 
40  # pore disconnectedness index
41  b = (log(1500.) - log(33.)) / (log(fc) - log(wp))
42  lamda = 1.0/b
43 
44  # saturated conductivity
45  #print s, c, sat, fc, 3-lamda
46  ks = 1930*pow(sat-fc, 3-lamda)
47 
48  #print wpt, fct, s33, fc, sat
49  return wp, fc_df, sat_df, p_df, ks, lamda
