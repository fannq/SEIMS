263 def GetProperties(s, c, om):
264  ## wilting point (SOL_WP)
265  wpt = -0.024*s + 0.487*c + 0.006*om + 0.005*s*om - 0.013*c*om + 0.068*s*c + 0.031
266  wp = 1.14*wpt - 0.02
267 
268  ## bulk density according to field capacity
269  fct = -0.251*s + 0.195*c + 0.011*om + 0.006*s*om - 0.027*c*om + 0.452*s*c + 0.299
270  fc = fct + 1.283*fct*fct - 0.374*fct - 0.015
271 
272  s33t = 0.278*s + 0.034*c + 0.022*om - 0.018*s*om - 0.027*c*om - 0.584*s*c + 0.078
273  s33 = 1.636*s33t - 0.107
274  sat = fc + s33 - 0.097*s + 0.043
275  pn = 2.65*(1.0 - sat)
276 
277  ## field capacity (SOL_FC) with density effects (df)
278  p_df = pn
279  sat_df = 1 - p_df/2.65 ## porosity
280  fc_df = fc - 0.2*(sat - sat_df)
281 
282  ## available water capacity (SOL_AWC)
283  awc = fc_df - wp
284 
285  # pore disconnectedness index
286  b = (log(1500.) - log(33.)) / (log(fc) - log(wp))
287  lamda = 1.0/b
288 
289  # saturated conductivity
290  #print s, c, sat, fc, 3-lamda
291  ks = 1930*pow(sat-fc, 3-lamda)
292 
293  #print wp, fc_df, awc,
294  return wp, fc_df, sat_df, p_df, ks, lamda
