53 void Common::getScurveShapeParameter(float xMid, float xEnd, float yMid, float yEnd, float* shape1, float* shape2)
54 {
55  float xx = log(yMid/xMid - yMid);
56  *shape2 = (xx - log(yEnd/xEnd - yEnd)) / (yEnd - yMid);
