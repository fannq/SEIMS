100 float ImplicitKinematicWave_CH::GetNewQ(float qIn, float qLast, float surplus, float alpha, float dt, float dx)
101 {
102  /* Using Newton-Raphson Method */
103  float ab_pQ, dtX, C; //auxillary vars
104  int count;
105  float Qkx; //iterated discharge, becomes Qnew
106  float fQkx; //function
107  float dfQkx; //derivative
108  const float _epsilon = 1e-12f;
109  const float beta = 0.6f;
110 
111  /* if no input then output = 0 */
112  if ((qIn + qLast) <= -surplus*dx)//0)
113  {
114  //itercount = -1;
115  return(0);
116  }
117 
118  /* common terms */
119  ab_pQ = alpha*beta*pow(((qLast+qIn)/2),beta-1);
120  // derivative of diagonal average (space-time)
121 
122  dtX = dt/dx;
123  C = dtX*qIn + alpha*pow(qLast,beta) + dt*surplus;
124  //dt/dx*Q = m3/s*s/m=m2; a*Q^b = A = m2; surplus*dt = s*m2/s = m2
125  //C is unit volume of water
126  // first gues Qkx
127  Qkx = (dtX * qIn + qLast * ab_pQ + dt * surplus) / (dtX + ab_pQ);
128 
129  // VJ 050704, 060830 infil so big all flux is gone
130  //VJ 110114 without this de iteration cannot be solved for very small values
131  if (Qkx < MIN_FLUX)
132  {
133  //itercount = -2;
134  return(0);
135  }
136 
137  Qkx = max(Qkx, MIN_FLUX);
138 
139  count = 0;
140  do {
141  fQkx = dtX * Qkx + alpha * pow(Qkx, beta) - C; /* Current k */
142  dfQkx = dtX + alpha * beta * pow(Qkx, beta - 1); /* Current k */
143  Qkx -= fQkx / dfQkx; /* Next k */
144  Qkx = max(Qkx, MIN_FLUX);
145  count++;
146  //qDebug() << count << fQkx << Qkx;
147  } while(fabs(fQkx) > _epsilon && count < MAX_ITERS_CH);
148 
149  if(Qkx != Qkx)
150  throw ModelException(MID_IKW_CH, "GetNewQ", "Error in iteration!");
151 
152  //itercount = count;
153  return Qkx;
154 }

320 void ImplicitKinematicWave_CH::ChannelFlow(int iReach, int iCell, int id, float qgEachCell)
321 {
322  float qUp = 0.f;
323 
324  if (iReach == 0 && iCell == 0)
325  {
326  qUp = m_qUpReach;
327  }
328 
329  // inflow from upstream channel
330  if (iCell == 0)// inflow of this cell is the last cell of the upstream reach
331  {
332  for (size_t i = 0; i < m_reachUpStream[iReach].size(); ++i)
333  {
334  int upReachId = m_reachUpStream[iReach][i];
335  if (upReachId >= 0)
336  {
337  int upCellsNum = m_reachs[upReachId].size();
338  int upCellId = m_reachs[upReachId][upCellsNum-1];
339  qUp += m_qCh[upReachId][upCellsNum-1];
340  }
341  }
342  //cout << qUp << "\t";
343  }
344  else
345  {
346  qUp = m_qCh[iReach][iCell-1] ;
347  }
348 
349  float dx = m_flowLen[iReach][iCell];
350 
351  float qLat = m_prec[id]/1000.f * m_chWidth[id] * dx / m_dt;
352  qLat += qgEachCell;
353 
354  //if (m_qs != NULL)
355  qLat += m_qs[id];
356  if (m_qi != NULL)
357  qLat += m_qi[id];
358 
359  if (qLat < MIN_FLUX && qUp < MIN_FLUX)
360  {
361  m_hCh[iReach][iCell] = 0.f;
362  m_qCh[iReach][iCell] = 0.f;
363  return;
364  }
365 
366  qUp += qLat;
367 
368  float Perim = 2.f*m_hCh[iReach][iCell] + m_chWidth[id];
369 
370  float sSin = sqrt(sin(m_sRadian[id]));
371  float alpha = pow((m_manningScalingFactor*m_reachN[iReach])/sSin * pow(Perim, _23), 0.6f);
372 
373  float qIn = m_qCh[iReach][iCell];
374 
375  m_qCh[iReach][iCell] = GetNewQ(qUp, qIn, 0.f, alpha, m_dt, dx);
376 
377  float hTest = m_hCh[iReach][iCell] + (qUp - m_qCh[iReach][iCell])*m_dt/m_chWidth[id]/dx;
378  float hNew = (alpha*pow(m_qCh[iReach][iCell], 0.6f)) / m_chWidth[id]; // unit m
379  m_hCh[iReach][iCell] = (alpha*pow(m_qCh[iReach][iCell], 0.6f)) / m_chWidth[id]; // unit m
380 }
