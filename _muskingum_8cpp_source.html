00115 void Muskingum::GetCofficients(float reachLength, float waterDepth, float s0, float v0, MuskWeights& weights)
00116 {
00117     
00118     //float x = 0.5f * (1 - waterDepth/(m_beta*reachLength*s0));
00119     //if (x < 0)
00120     //  x = 0.01f;
00121     float x = m_msk_x;
00122     //float v = sqrt(s0) * pow(waterDepth, TWO_THIRDS) / manning_n;
00123     //if (v < 0.2f)

00126     //  v = 3.f;
00127     //float K = reachLength / (m_beta*v);
00128     v0 = m_vScalingFactor * v0;
00129     float K = (4.64f - 3.64f*0.7f) * reachLength / (m_beta*v0);
00130     //get delta t
00131     float min = 2.0f*K*x;

00133     float delta_t = GetDelta_t(m_dt, min, max);
00134     weights.dt = delta_t;
00135 
00136     //get coefficient
00137     float temp = max + delta_t;
00138     weights.c1 = (delta_t - min)/temp;
00139     weights.c2 = (delta_t + min)/temp;
00140     weights.c3 = (max - delta_t)/temp;
00141     weights.c4 = 2*delta_t/temp;
00142 
00143     //make sure any coefficient is positive
00144     if(weights.c1 < 0) 
00145     {
00146         weights.c2 += weights.c1;
00147         weights.c1 = 0.0f;
00148     }
00149     if(weights.c3 < 0)
00150     {
00151         weights.c2 += weights.c1;
00152         weights.c3 = 0.0f;

00304 void Muskingum::ChannelFlow(int iReach, int iCell, int id, float qgEachCell)
00305 {
00306     float qUpNew = 0.f;
00307     
00308     if (iReach == 0 && iCell == 0)
00309     {
00310         qUpNew = m_qUpReach;
00311     }
00312 
00313 
00314     // inflow from upstream channel
00315     if (iCell == 0)// inflow of this cell is the last cell of the upstream reach
00316     {
00317         for (size_t i = 0; i < m_reachUpStream[iReach].size(); ++i)
00318         {
00319             int upReachId = m_reachUpStream[iReach][i];
00320             if (upReachId >= 0)
00321             {
00322                 int upCellsNum = m_reachs[upReachId].size();
00323                 int upCellId = m_reachs[upReachId][upCellsNum-1];
00324                 qUpNew += m_qCh[upReachId][upCellsNum-1];
00325             }
00326         }
00327     }
00328     else
00329     {
00330         qUpNew = m_qCh[iReach][iCell-1];
00331     }
00332     
00333     float dx = m_flowLen[iReach][iCell];
00334     float area = dx*m_chWidth[id];
00335 
00336     // lateral flow 
00337     float qLat = m_prec[id]/1000.f*area/m_dt;
00338 
00339     if(m_qs != NULL)
00340         qLat += m_qs[id];
00341 
00342     if(m_qi != NULL)
00343         qLat += m_qi[id];
00344 
00345     qLat += qgEachCell;
00346 
00347     //if (iReach == 364)
00348     //  cout << qLat << qUpNew << endl;
00349     qUpNew += qLat;
00350     
00351     float waterDepth = m_chStorage[iReach][iCell]/area;
00352 
00353     MuskWeights weights;
00354     GetCofficients(dx, waterDepth, m_s0[id], m_v0[iReach], weights);
00355     int stepCount = int(m_dt / weights.dt);
00356     //start the loop for every delta t
00357     float qNew = 0.f;
00358 
00359     float q = 0.f;
00360     for(int i = 0; i < stepCount; i++)
00361     {
00362 
00363         //GetCofficients(dx, waterDepth, m_s0[i], m_v0[iReach], weights);
00364         // unit discharge
00365         qNew = weights.c1*qUpNew + weights.c2*m_qUpCh[iReach][iCell] + weights.c3*qNew;
00366 
00367         if (qNew != qNew)
00368         {
00369             cout << "Error in function Muskingum::ChannelFlow. \n";
00370             cout << "The weights are: " << weights.c1 << ", " << weights.c2 << ", " << weights.c3 << endl;
00371             cout << "qUpNew: " << qUpNew << "\tqUpPre: " << m_qUpCh[iReach][iCell] << "\tqNew: " << qNew << endl;
00372             throw ModelException("CH_MSK","ChannelFlow","Error occurred.");
00373         }
00374 
00375         float tmp = m_chStorage[iReach][iCell] + (qUpNew - qNew) * weights.dt;
00376         if (tmp > 0.f)
00377         {
00378             m_chStorage[iReach][iCell] = tmp;
00379             m_qCh[iReach][iCell] = qNew;
00380         }
00381         else
00382         {
00383             m_qCh[iReach][iCell] = m_chStorage[iReach][iCell] / weights.dt;
00384             m_chStorage[iReach][iCell] = 0.f;
00385         }
00386     
00387         m_qUpCh[iReach][iCell] = qUpNew;
00388         q += m_qCh[iReach][iCell];
00389     }
00390 
00391     q /= stepCount;
00392     m_qCh[iReach][iCell] = q;
00393 }
