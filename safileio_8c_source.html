00200 const wchar_t* Utf8ToWideChar( const char *pszFilename )
00201 {
00202     int nMulti, nWide;
00203     wchar_t *pwszFileName;
00204     
00205     nMulti = strlen(pszFilename) + 1;
00206     nWide = MultiByteToWideChar( CP_UTF8, 0, pszFilename, nMulti, 0, 0);
00207     if( nWide == 0 )
00208     {
00209         return NULL;
00210     }
00211     pwszFileName = (wchar_t*) malloc(nWide * sizeof(wchar_t));
00212     if ( pwszFileName == NULL )
00213     {
00214         return NULL;
00215     }
00216     if( MultiByteToWideChar( CP_UTF8, 0, pszFilename, nMulti, pwszFileName, nWide ) == 0 )
00217     {
00218         free( pwszFileName );
00219         return NULL;
00220     }
00221     return pwszFileName;
00222 }
