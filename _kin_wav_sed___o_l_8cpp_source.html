00407 void KinWavSed_OL::GetTransportCapacity(int id)
00408 {
00409     float q, S0, K;
00410     q = m_V[id] * m_WH[id]*60;   // m2/s -> m2/min                                 // m_Qkin[id]*60;   // convert to m3/min
00411     float s = max(0.001f, m_Slope[id]);
00412     S0 = sin(atan(s));
00413     K = m_USLE_K[id];
00414     float threadhold = 0.046f;
00415     if(q <= 0.0f)
00416         m_Ctrans[id] = 0;
00417     else
00418     {
00419         if(q < threadhold)
00420             m_Ctrans[id] = m_eco1 * K * S0* sqrt(q);   // kg/(m*min) kg per unit width per minute
00421         else
00422             m_Ctrans[id] = m_eco2 * K * S0* pow(q, 2.0f);
00423          // kg/(m*min)  -> convert to kg/m3
00424         m_Ctrans[id] = m_Ctrans[id] / q ;
00425     }
00426 }

00428 void KinWavSed_OL::GetSedimentInFlow(int id)
00429 {
00430     float TC, Df, Dsp, Deposition, concentration, Vol;
00431     GetTransportCapacity(id);
00432     TC = m_Ctrans[id];             //kg/m3
00433     CalcuFlowDetachment(id);
00434     Df = m_DETOverland[id];       //kg/cell
00435     Dsp = m_DETSplash[id];       //kg/cell
00436 
00437     m_Sed_kg[id] += (Df + Dsp);   //sediment in flow
00438     //while the sediment in flow is more than the capacity of the overland flow,
00439     // then the sediment deposition would happen; if the velocity of flow is reduced, the TC is decreased correspondingly,
00440     // then, the sediment deposition may take place.
00441     //MaxConcentration(m_Vol[id], m_Sed_kg[id], id);
00442     Vol = m_Vol[id];
00443     if(Vol > 0)
00444         concentration = m_Sed_kg[id] / Vol;    // convert to kg/m3
00445     else
00446         concentration = 0;
00447     Deposition = max(concentration - TC, 0.0f);   //kg/m3, >0
00448     if (Deposition > 0)
00449     {
00450         m_Sed_kg[id] = TC * Vol;
00451     }
00452     m_SedDep[id] = Deposition * Vol; //kg/cell
00453 }

00521 void KinWavSed_OL::CalcuFlowDetachment(int i)  //i is the id of the cell in the grid map  
00522 {
00523     // correction for slope dx/DX, water spreads out over larger area
00524     float s = max(0.001f, m_Slope[i]);
00525     float S0 = sin(atan(s));
00526     float waterdepth = m_WH[i] / 1000;   // mm convert to m
00527     //float waterdepth = m_Runoff[i] / 1000;   // mm convert to m
00528 
00529     //using Foster equation to calculate overland flow detachment
00530     float Df, waterden, g, shearStr;      //,q
00531     waterden = 1000;
00532     g = 9.8f;
00533     shearStr = waterden * g * waterdepth * S0;
00534     Df = m_Ccoe * m_USLE_C[i] * m_USLE_K[i] * pow(shearStr, 1.5f);
00535     /*q = m_Q[i];
00536     Df = m_Ccoe * m_USLE_C[i] * m_USLE_K[i] * q * S0;*/
00537     // convert kg/(m2*min) to kg/cell
00538     float cellareas = (m_cellWidth/cos(atan(s))) * m_cellWidth;
00539     m_DETOverland[i] = Df * (m_TimeStep/60) * cellareas;
00540 }
