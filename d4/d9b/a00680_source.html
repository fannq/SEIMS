234 void NandPim::CalculateMinerandImmobi() {
235  //soil layer (k)
236  for(int i = 0; i < m_nCells; i++) {
237  for(int k = 0; k < m_nSolLyrs; k++) {
238  //soil layer used to compute soil water and soil temperature factors
239  int kk = 0;
240  if(k == 0) {
241  kk = 1;
242  } else {
243  kk = k;
244  }
245  float sut = 0;
246  // mineralization can occur only if temp above 0 deg
247  if (m_sol_tmp[i][kk] > 0) {
248  // compute soil water factor (sut)
249  sut = 0;
250  if (m_sol_st[i][kk] < 0) {
251  m_sol_st[i][kk] = 0.0000001f;
252  }
253  sut = 0.1 + 0.9 * sqrt(m_sol_st[i][kk] / m_sol_fc[i][kk]);
254  sut = max(0.05, sut);
255 
256  //compute soil temperature factor
257  //variable to hold intermediate calculation result (xx)
258  float xx = 0;
259  //soil temperature factor (cdg)
260  float cdg = 0;
261  xx = m_sol_tmp[i][kk];
262  cdg = 0.9 * xx / (xx + exp(9.93 - 0.312 * xx)) + 0.1;
263  cdg = max(0.1, cdg);
264 
265  // compute combined factor
266  xx = 0;
267  //combined temperature/soil water factor (csf)
268  float csf = 0;
269  xx = cdg * sut;
270  if(xx < 0) {
271  xx = 0;
272  }
273  if(xx > 1000000) {
274  xx = 1000000;
275  }
276  csf = sqrt(xx);
277 
278  // compute flow from active to stable pools
279  //amount of nitrogen moving from active organic to stable organic pool in layer (rwn)
280  float rwn = 0;
281  rwn = 0.1e-4 * (m_sol_aorgn[i][k] * (1 / m_nactfr - 1.) - m_sol_orgn[i][k]);
282  if (rwn > 0.) {
283  rwn = min(rwn, m_sol_aorgn[i][k]);
284  } else {
285  rwn = -(min(abs(rwn), m_sol_orgn[i][k]));
286  }
287  m_sol_orgn[i][k] = max(1.e-6, m_sol_orgn[i][k] + rwn);
288  m_sol_aorgn[i][k] = max(1.e-6, m_sol_aorgn[i][k] - rwn);
289 
290  // compute humus mineralization on active organic n
291  //amount of nitrogen moving from active organic nitrogen pool to nitrate pool in layer (hmn)
292  float hmn = 0;
293  hmn = m_cmn * csf * m_sol_aorgn[i][k];
294  hmn = min(hmn, m_sol_aorgn[i][k]);
295  // compute humus mineralization on active organic p
296  xx = 0;
297  //amount of phosphorus moving from the organic pool to the labile pool in layer (hmp)
298  float hmp = 0;
299  xx = m_sol_orgn[i][k] + m_sol_aorgn[i][k];
300  if (xx > 1e-6) {
301  hmp = 1.4 * hmn * m_sol_orgp[i][k] / xx;
302  } else {
303  hmp = 0;
304  }
305  hmp = min(hmp, m_sol_orgp[i][k]);
306 
307  // move mineralized nutrients between pools;
308  m_sol_aorgn[i][k] = max(1e-6, m_sol_aorgn[i][k] - hmn);
309  m_sol_no3[i][k] = m_sol_no3[i][k] + hmn;
310  m_sol_orgp[i][k] = m_sol_orgp[i][k] - hmp;
311  m_sol_solp[i][k] = m_sol_solp[i][k] + hmp;
312 
313  // compute residue decomp and mineralization of
314  // fresh organic n and p (upper two layers only)
315  //amount of nitrogen moving from fresh organic to nitrate(80%) and active organic(20%) pools in layer (rmn1)
316  float rmn1 = 0;
317  //amount of phosphorus moving from fresh organic to labile(80%) and organic(20%) pools in layer (rmp)
318  float rmp = 0;
319  if (k <= 2) {
320  //the C:N ratio (cnr)
321  float cnr = 0;
322  //the C:P ratio (cnr)
323  float cpr = 0;
324  //the nutrient cycling water factor for layer (ca)
325  float ca = 0;
326  float cnrf = 0;
327  float cprf = 0;
328  if (m_sol_fon[i][k] + m_sol_no3[i][k] > 1e-4) {
329  //Calculate cnr, equation 3:1.2.5 in SWAT Theory 2009, p189
330  cnr = 0.58 * m_sol_rsd[i][k] / (m_sol_fon[i][k] + m_sol_no3[i][k]);
331  if (cnr > 500) {
332  cnr = 500;
333  }
334  cnrf = exp(-0.693 * (cnr - 25) / 25);
335  } else {
336  cnrf = 1;
337  }
338  if (m_sol_fop[i][k] + m_sol_solp[i][k] > 1e-4) {
339  cpr = 0.58 * m_sol_rsd[i][k] / (m_sol_fop[i][k] + m_sol_solp[i][k]);
340  if (cpr > 5000) {
341  cpr = 5000;
342  }
343  cprf = 0.;
344  cprf = exp(-0.693 * (cpr - 200) / 200);
345  } else {
346  cprf = 1;
347  }
348  //decay rate constant (decr)
349  float decr = 0;
350 
351  float rdc = 0;
352  //Calculate ca, equation 3:1.2.8 in SWAT Theory 2009, p190
353  ca = min(min(cnrf, cprf), 1);
354  if (m_idplt[i] > 0) {
355  decr = m_rsdco_pl[(int)m_idplt[i]] * ca * csf;
356  } else {
357  decr = 0.05;
358  }
359  decr = min(decr, 1.0f);
360  m_sol_rsd[i][k] = max(1.e-6, m_sol_rsd[i][k]);
361  rdc = decr * m_sol_rsd[i][k];
362  m_sol_rsd[i][k] = m_sol_rsd[i][k] - rdc;
363  if (m_sol_rsd[i][k] < 0) {
364  m_sol_rsd[i][k] = 0;
365  }
366  rmn1 = decr * m_sol_fon[i][k];
367  m_sol_fop[i][k] = max(1e-6, m_sol_fop[i][k]);
368  rmp = decr * m_sol_fop[i][k];
369 
370  m_sol_fop[i][k] = m_sol_fop[i][k] - rmp;
371  m_sol_fon[i][k] = max(1e-6, m_sol_fon[i][k]) - rmn1;;
372  //Calculate no3, aorgn, solp, orgp, equation 3:1.2.9 in SWAT Theory 2009, p190
373  m_sol_no3[i][k] = m_sol_no3[i][k] + 0.8 * rmn1;
374  m_sol_aorgn[i][k] = m_sol_aorgn[i][k] + 0.2 * rmn1;
375  m_sol_solp[i][k] = m_sol_solp[i][k] + 0.8 * rmp;
376  m_sol_orgp[i][k] = m_sol_orgp[i][k] + 0.2 * rmp;
377  }
378  // compute denitrification
379  //amount of nitrogen lost from nitrate pool in layer due to denitrification
380  float wdn = 0;
381  //denitrification threshold: fraction of field capacity (sdnco)
382  float* sdnco(NULL);
383  //Calculate wdn, equation 3:1.4.1 and 3:1.4.2 in SWAT Theory 2009, p194
384  if (sut >= sdnco[i]) {
385  wdn = m_sol_no3[i][k] * (1. - exp(-m_cdn[i] * cdg * m_sol_cbn[i][k]));
386  } else {
387  wdn = 0;
388  }
389  m_sol_no3[i][k] = m_sol_no3[i][k] - wdn;
390 
391  // Summary calculations
392  m_wshd_hmn = m_wshd_hmn + hmn * (1 / m_nCells);
393  m_wshd_rwn = m_wshd_rwn + rwn * (1 / m_nCells);
394  m_wshd_hmp = m_wshd_hmp + hmp * (1 / m_nCells);
395  m_wshd_rmn = m_wshd_rmn + rmn1 * (1 / m_nCells);
396  m_wshd_rmp = m_wshd_rmp + rmp * (1 / m_nCells);
397  m_wshd_dnit = m_wshd_dnit + wdn * (1 / m_nCells);
398  m_hmntl = m_hmntl + hmn;
399  m_rwntl = m_rwntl + rwn;
400  m_hmptl = m_hmptl + hmp;
401  m_rmn2tl = m_rmn2tl + rmn1;
402  m_rmptl = m_rmptl + rmp;
403  m_wdntl = m_wdntl + wdn;
404  }
405  }
406  }

408 void NandPim::CalculateMinerandVolati() {
409  //soil layer (k)
410  int kk = 0;
411  for(int i = 0; i < m_nCells; i++) {
412  for(int k = 0; k < m_nSolLyrs; k++) {
413  //nitrification/volatilization temperature factor (nvtf)
414  float nvtf = 0;
415  //Calculate nvtf, equation 3:1.3.1 in SWAT Theory 2009, p192
416  nvtf = 0.41 * (m_sol_tmp[i][k] - 5) / 10;
417  if (m_sol_nh3[i][k] > 0 && nvtf >= 0.001) {
418  float sw25 = 0;
419  float swwp = 0;
420  //nitrification soil water factor (swf)
421  float swf = 0;
422  //Calculate nvtf, equation 3:1.3.2 and 3:1.3.3 in SWAT Theory 2009, p192
423  sw25 = m_sol_wpmm[i][k] + 0.25 * m_sol_fc[i][k];
424  swwp = m_sol_wpmm[i][k] + m_sol_st[i][k];
425  if (swwp < sw25) {
426  swf = (swwp - m_sol_wpmm[i][k]) / (sw25 - m_sol_wpmm[i][k]);
427  } else {
428  swf = 1;
429  }
430 
431  if (k == 0) {
432  kk = 0;
433  } else {
434  kk = m_sol_z[k-1][i];
435  }
436  //depth from the soil surface to the middle of the layer (dmidl/mm)
437  float dmidl = 0;
438  //volatilization depth factor (dpf)
439  float dpf = 0;
440  //nitrification regulator (akn)
441  float akn = 0;
442  //volatilization regulator (akn)
443  float akv = 0;
444  //amount of ammonium converted via nitrification and volatilization in layer (rnv)
445  float rnv = 0;
446  //amount of nitrogen moving from the NH3 to the NO3 pool (nitrification) in the layer (rnit)
447  float rnit = 0;
448  //amount of nitrogen lost from the NH3 pool due to volatilization (rvol)
449  float rvol = 0;
450  //volatilization CEC factor (cecf)
451  float cecf = 0.15;
452  //Calculate nvtf, equation 3:1.3.2 and 3:1.3.3 in SWAT Theory 2009, p192
453  dmidl = (m_sol_z[i][k] + kk) / 2;
454  dpf = 1 - dmidl / (dmidl + exp(4.706 - 0.0305 * dmidl));
455  //Calculate rnv, equation 3:1.3.6, 3:1.3.7 and 3:1.3.8 in SWAT Theory 2009, p193
456  akn = nvtf * swf;
457  akv = nvtf * dpf * cecf;
458  rnv = m_sol_nh3[i][k] * (1 - exp(-akn - akv));
459  //Calculate rnit, equation 3:1.3.9 in SWAT Theory 2009, p193
460  rnit = 1 - exp(-akn);
461  //Calculate rvol, equation 3:1.3.10 in SWAT Theory 2009, p193
462  rvol = 1 - exp(-akv);
463 
464  //calculate nitrification (NH3 => NO3)
465  if ((rvol + rnit) > 1e-6) {
466  //Calculate the amount of nitrogen removed from the ammonium pool by nitrification,
467  //equation 3:1.3.11 in SWAT Theory 2009, p193
468  rvol = rnv * rvol / (rvol + rnit);
469  //Calculate the amount of nitrogen removed from the ammonium pool by volatilization,
470  //equation 3:1.3.12 in SWAT Theory 2009, p194
471  rnit = rnv - rvol;
472  if (rnit < 0) {
473  rnit = 0;
474  }
475  m_sol_nh3[i][k] = max(1e-6, m_sol_nh3[i][k] - rnit);
476  }
477  if (m_sol_nh3[i][k] < 0) {
478  rnit = rnit + m_sol_nh3[i][k];
479  m_sol_nh3[i][k] = 0;
480  }
481  m_sol_no3[i][k] = m_sol_no3[i][k] + rnit;
482  //calculate ammonia volatilization
483  m_sol_nh3[i][k] = max(1e-6, m_sol_nh3[i][k] - rvol);
484  if (m_sol_nh3[i][k] < 0) {
485  rvol = rvol + m_sol_nh3[i][k];
486  m_sol_nh3[i][k] = 0;
487  }
488  //summary calculations
489  m_wshd_voln = m_wshd_voln + rvol * (1 / m_nCells);
490  m_wshd_nitn = m_wshd_nitn + rnit * (1 / m_nCells);
491  }
492  }

495 void NandPim::CalculatePflux() {
496  //slow equilibration rate constant (bk)
497  float bk = 0.0006;
498  for(int i = 0; i < m_nCells; i++) {
499  float rto = m_psp[i] / (1 - m_psp[i]);
500  for(int k = 0; k < m_nSolLyrs; k++) {
501  float rmn1 = 0;
502  rmn1 = (m_sol_solp[i][1] - m_sol_actp[i][1] * rto);
503  if (rmn1 > 0.) rmn1 = rmn1 * 0.1;
504  if (rmn1 < 0.) {
505  rmn1 = rmn1 * 0.6;
506  }
507  rmn1 = min(rmn1, m_sol_solp[i][1]);
508  //amount of phosphorus moving from the active mineral to the stable mineral pool in the soil layer (roc)
509  float roc = 0;
510  //Calculate roc, equation 3:2.3.4 and 3:2.3.5 in SWAT Theory 2009, p215
511  roc = bk * (4. * m_sol_actp[i][1] - m_sol_stap[i][1]);
512  if (roc < 0) {
513  roc = roc * 0.1;
514  }
515  roc = min(roc, m_sol_actp[i][1]);
