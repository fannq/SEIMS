88 int PETPriestleyTaylor::Execute()
89 {
90  if(!this->CheckInputData()) return false;
91  if(this->m_pet == NULL)
92  this->m_pet = new float[this->m_nCells];
93  int d = JulianDay(this->m_date);
94 #pragma omp parallel for
95  for (int i = 0; i < m_nCells; ++i)
96  {
99  float raShortWave = m_sr[i] * (1.0f - 0.23f);
100  //if the mean T < T_snow, consider the snow depth is larger than 0.5mm.
101  if(m_tMean[i] < this->m_tSnow)
102  raShortWave = m_sr[i] * (1.0f - 0.8f);
103 
105  m_srMax = MaxSolarRadiation(d, this->m_cellLat[i]);
106 
109  float satVaporPressure = SaturationVaporPressure(m_tMean[i]);
110  float actualVaporPressure = m_rhd[i] * satVaporPressure;
111  float rbo = -(0.34f - 0.139f * sqrt(actualVaporPressure));
112  //cloud cover factor
113  float rto = 0.0f;
114  if (m_srMax >= 1.0e-4)
115  rto = 0.9f * (m_sr[i] / m_srMax) + 0.1f;
116  //net long-wave radiation
117  float tk = m_tMean[i] + 273.15f;
118  float raLongWave = rbo * rto * 4.9e-9f * pow(tk, 4);
119 
120  float raNet = raShortWave + raLongWave;
121 
123  //calculate the slope of the saturation vapor pressure curve
124  float dlt = 4098.f* satVaporPressure / pow((m_tMean[i] + 237.3f), 2);
125 
126  //calculate latent heat of vaporization(MJ/kg, from swat)
127  float latentHeat = LatentHeatVapor(m_tMean[i]);
128 
129  //calculate mean barometric pressure(kPa)
130  float pb = MeanBarometricPressure(m_elev[i]);
131  //psychrometric constant(kPa/deg C)
132  float gma = 1.013e-3f * pb / (0.622f * latentHeat);
133  float pet_alpha = 1.28f;
134 
135  float petValue = pet_alpha * (dlt / (dlt + gma)) * raNet / latentHeat;
136  m_pet[i] = m_petFactor * max(0.0f, petValue);
137  }
138  return 0;
139 }
