43 void SSR_DA::FlowInSoil(int id)
44 {
45  float s0 = max(m_slope[id], 0.01f);
46 
47  float depth[2];
48  depth[0] = m_upSoilDepth;
49  depth[1] = m_rootDepth[id] - m_upSoilDepth;
50  if(depth[1] < 0)
51  {
52  ostringstream oss;
53  oss << "The root depth at cell(" << id << ") is " << m_rootDepth[id] << ", and is less than the upper soil depth (" << m_upSoilDepth << endl;
54  throw ModelException("SSR_DA", "Execute", oss.str());
55  }
56 
57  float flowWidth = m_CellWidth;
58  // there is no land in this cell
59  if (m_streamLink[id] > 0)
60  flowWidth -= m_chWidth[id];
61 
62  int nUpstream = (int)m_flowInIndex[id][0];
63  //cout << "number of layers:" << m_nSoilLayers << endl;
64  for (int j = 0; j < m_nSoilLayers; j++)
65  {
66  float smOld = m_sm[id][j];
67  //sum the upstream overland flow
68  float qUp = 0.0f;
69  for (int upIndex = 1; upIndex <= nUpstream; ++upIndex)
70  {
71  int flowInID = (int)m_flowInIndex[id][upIndex];
72  //cout << id << "\t" << flowInID << "\t" << m_nCells << endl;
73 
74  qUp += m_qi[flowInID][j];// * m_flowInPercentage[id][upIndex];
75  //cout << id << "\t" << flowInID << "\t" << m_nCells << "\t" << m_qi[flowInID][j] << endl;
76  //error happens in this place
77  }
78 
79  if (flowWidth <= 0)
80  {
81  m_qi[id][j] = 0.f;
82  m_qiVol[id][j] = 0.f;
83  continue;
84  }
85 
86  // add upstream water to the current cell
87  float soilVolumn = depth[j]/1000*m_CellWidth*flowWidth; //m3
88  m_sm[id][j] += qUp / soilVolumn;
89 
90  //TEST
91  if(m_sm[id][j] != m_sm[id][j] || m_sm[id][j] < 0.f)
92  {
93  ostringstream oss;
94  oss << "cell id: " << id << "\t layer: " << j << "\nmoisture is less than zero: " << m_sm[id][j] << "\t"
95  << smOld << "\nqUp: " << qUp << "\t depth:" << depth[j] << "\tupper soil depth:" << m_upSoilDepth << endl;
96  throw ModelException("SSR_DA", "Execute:FlowInSoil", oss.str());
97  }
98 
99  m_qi[id][j] = 0.f; //default return value
100 
101  // if soil moisture is below the field capacity, no interflow will be generated
102  if (m_sm[id][j] > m_fc[id][j])
103  {
104  // for the upper two layers, soil may be frozen
105  // also check if there are upstream inflow
106  if(j == 0 && m_soilT[id] <= m_frozenT && qUp <= 0.f)
107  continue;
108 
109  float k, maxSoilWater, soilWater, fcSoilWater;
110  //the moisture content can exceed the porosity in the way the algorithm is implemented
111  if (m_sm[id][j] > m_porosity[id][j])
112  k = m_ks[id][j];
113  else
114  {
115  float dcIndex = 2.f/m_poreIndex[id][j] + 3.f; // pore disconnectedness index
116  k = m_ks[id][j] * pow(m_sm[id][j]/m_porosity[id][j], dcIndex);
117  }
118  m_qi[id][j] = m_ki * s0 * k * m_dt/3600.f * depth[j]/1000.f / flowWidth; // the unit is mm
119 
120  maxSoilWater = depth[j] * m_porosity[id][j];
121  soilWater = depth[j] * m_sm[id][j];
122  fcSoilWater = depth[j] * m_fc[id][j];
123 
124  if(soilWater - m_qi[id][j] < fcSoilWater)
125  m_qi[id][j] = soilWater - fcSoilWater;
126 
127  soilWater -= m_qi[id][j];
128  if(soilWater > maxSoilWater)
129  m_qi[id][j] += soilWater - maxSoilWater;
130 
131  m_qiVol[id][j] = m_qi[id][j]/1000*m_CellWidth*flowWidth; //m3
132  //Adjust the moisture content in the current layer, and the layer immediately below it
133  m_sm[id][j] -= m_qi[id][j]/depth[j];
134 
135  if(m_sm[id][j] != m_sm[id][j] || m_sm[id][j] < 0.f)
136  {
137  ostringstream oss;
138  oss << id << "\t" << j << "\nmoisture is less than zero: " << m_sm[id][j] << "\t" << depth[j];
139  throw ModelException("SSR_DA", "Execute", oss.str());
140  }
141  }
142  }

146 int SSR_DA::Execute()
147 {
148  CheckInputData();
149  initalOutputs();
150 
151  if(m_qi == NULL)
152  {
153  m_qi = new float*[m_nCells];
154  m_qiVol = new float*[m_nCells];
155 
156  for (int i = 0; i < m_nCells; i++)
157  {
158  m_qi[i] = new float[m_nSoilLayers];
159  m_qiVol[i] = new float[m_nSoilLayers];
160  for (int j = 0; j < m_nSoilLayers; j++)
161  {
162  m_qi[i][j] = 0.f;
163  m_qiVol[i][j] = 0.f;
164  }
165  }
166  }
167 
168  for (int iLayer = 0; iLayer < m_nRoutingLayers; ++iLayer)
169  {
170  // There are not any flow relationship within each routing layer.
171  // So parallelization can be done here.
172  int nCells = (int)m_routingLayers[iLayer][0];
173 
174  #pragma omp parallel for
175  for (int iCell = 1; iCell <= nCells; ++iCell)
176  {
177  int id = (int)m_routingLayers[iLayer][iCell];
178  FlowInSoil(id);
179  }
180  }
181 
182  //cout << "end flowinsoil" << endl;
183  for(int i = 0; i < m_nSubbasin; i++)
184  m_qiSubbasin[i] = 0.f;
185 
186  float qiAllLayers = 0.f;
187  for(int i = 0; i < m_nCells; i++)
188  {
189  if(m_streamLink[i] > 0)
190  {
191  qiAllLayers = 0.f;
192  for(int j = 0; j < m_nSoilLayers; j++)
193  qiAllLayers += m_qiVol[j][i];
194  //cout << m_nSubbasin << "\tsubbasin:" << m_subbasin[i] << "\t" << qiAllLayers << endl;
195  if(m_nSubbasin > 1)
196  m_qiSubbasin[int(m_subbasin[i])] += qiAllLayers/86400.f;
197  else
198  m_qiSubbasin[1] += qiAllLayers/86400.f;
199  }
200  }
201 
202  for(int i = 1; i < m_nSubbasin; i++)
203  m_qiSubbasin[0] += m_qiSubbasin[i];
204 
205  return 0;
