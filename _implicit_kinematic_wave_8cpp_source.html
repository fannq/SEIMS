00167 float ImplicitKinematicWave::GetNewQ(float qIn, float qLast, float surplus, float alpha, float dt, float dx)
00168 {
00169     /* Using Newton-Raphson Method */
00170     float  ab_pQ, dtX, C;  //auxillary vars
00171     int   count;
00172     float Qkx; //iterated discharge, becomes Qnew
00173     float fQkx; //function
00174     float dfQkx;  //derivative
00175     const float _epsilon = 1e-12f;
00176     const float beta = 0.6f;
00177 
00178     /* if no input then output = 0 */
00179     if ((qIn + qLast) <= -surplus*dx)//0)
00180     {
00181         //itercount = -1;
00182         return(0);
00183     }
00184 
00185     /* common terms */
00186     ab_pQ = alpha*beta*pow(((qLast+qIn)/2),beta-1);
00187     // derivative of diagonal average (space-time)
00188 
00189     dtX = dt/dx;
00190     C = dtX*qIn + alpha*pow(qLast,beta) + dt*surplus;
00191     //dt/dx*Q = m3/s*s/m=m2; a*Q^b = A = m2; surplus*dt = s*m2/s = m2
00192     //C is unit volume of water
00193     // first gues Qkx
00194     Qkx   = (dtX * qIn + qLast * ab_pQ + dt * surplus) / (dtX + ab_pQ);
00195 
00196     // VJ 050704, 060830 infil so big all flux is gone
00197     //VJ 110114 without this de iteration cannot be solved for very small values
00198     if (Qkx < MIN_FLUX)
00199     {
00200         //itercount = -2;
00201         return(0);
00202     }
00203 
00204     Qkx   = max(Qkx, MIN_FLUX);
00205 
00206     count = 0;
00207     do {
00208         fQkx  = dtX * Qkx + alpha * pow(Qkx, beta) - C;   /* Current k */
00209         dfQkx = dtX + alpha * beta * pow(Qkx, beta - 1);  /* Current k */
00210         Qkx   -= fQkx / dfQkx;                                /* Next k */
00211         Qkx   = max(Qkx, MIN_FLUX);
00212         count++;
00213         //qDebug() << count << fQkx << Qkx;
00214     } while(fabs(fQkx) > _epsilon && count < MAX_ITERS);
00215 
00216     if(Qkx != Qkx)
00217         throw ModelException("IKW_OL", "GetNewQ", "Error in iteration!");
00218 
00219     //itercount = count;
00220     return Qkx;
00221 }

00227     const float beta = 0.6f;
00228     const float _23 = 2.0f/3.0f;
00229     float beta1 = 1/beta;
00230 
00231     float h = m_sr[id]/1000.f;
00232 
00233     float Perim = 2.f*h + m_flowWidth[id];
00234     float r = 0;
00235     if (Perim > 0)
00236         r = h*m_flowWidth[id]/Perim;
00237 
00238     float sSin = sqrt(sin(m_sRadian[id]));
00239     m_alpha[id] = pow(m_n[id]/sSin * pow(Perim, _23),beta);
00240 
00241     if (m_alpha[id] > 0)
00242         m_q[id] = pow((m_flowWidth[id]*h)/m_alpha[id], beta1);
00243     else
00244         m_q[id] = 0;
00245 
00246     m_vel[id] = pow(r, _23)*sSin/m_n[id];
00247 
00248     float flowWidth = m_flowWidth[id];
00249     float flowLen = m_flowLen[id];
00250 
00251     //sum the upstream overland flow
00252     float qUp = 0.0f;
00253     for (int k = 1; k <= (int)m_flowInIndex[id][0]; ++k)
00254     {
00255         int flowInID = (int)m_flowInIndex[id][k];
00256         if (m_streamLink[flowInID] <= 0) // if the upstream cell is not a channel cell
00257             qUp += m_q[flowInID];
00258     }
00259 
00260     // if the channel width is greater than the cell width
00261     if (m_streamLink[id] >= 0 && flowWidth <= 0)
00262     {
00263         m_q[id] = qUp;
00264         m_sr[id] = 0.f;
00265         return;
00266     }
00267 
00268     // check whether overland flow routing is needed
00269     if (qUp < MIN_FLUX && m_q[id] < MIN_FLUX)
00270     {
00271         m_sr[id] = 0.f;
00272         m_q[id] = 0.f;
00273         return;
00274     }
00275 
00276     // calcluate infiltration surplus (m2/s)
00277     float surplus = 0.f;
00278     if (m_infilCapacitySurplus != NULL)
00279         surplus = -m_infilCapacitySurplus[id]/1000.f * flowWidth / m_dtStorm;
00280 
00281     float qIn = m_q[id];
00282     m_q[id] = GetNewQ(qUp, qIn, surplus, m_alpha[id], m_dtStorm, flowLen);
00283     
00284     float hNew = (m_alpha[id]*pow(m_q[id], 0.6f)) / flowWidth; // unit m
00285     //float hTest = h + (qUp - m_q[id])*m_dtStorm/(flowWidth*flowLen);
00286     m_sr[id] = hNew * 1000;
00287 
00288     float reInfil = (qUp - m_q[id])*m_dtStorm/(flowWidth*flowLen) + h - hNew;
00289     reInfil *= 1000.f;
00290 
00291     //if (abs(reInfil) < 0.001f)
00292     //  reInfil = 0.f;
00293     //else if(abs(reInfil - m_infilCapacitySurplus[id]) < 0.001f)
00294     //  reInfil = m_infilCapacitySurplus[id];

00297     //  throw  ModelException("IKW_OL", "OverlandFlow", "Reinfiltration exceeded range!");
00298 
00299     if (reInfil < 0.f)
00300         reInfil = 0.f;
00301 
00302     m_infil[id] += reInfil;
00303     if (m_accumuDepth != NULL)
00304         m_accumuDepth[id] += reInfil;
00305

00307         m_infilCapacitySurplus[id] -= reInfil;
