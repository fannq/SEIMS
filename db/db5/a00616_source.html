141 void IKW_REACH::initalOutputs()
142 {
143  if(m_nreach <= 0)
144  throw ModelException("IKW_REACH","initalOutputs","The cell number of the input can not be less than zero.");
145 
146  if (m_reachLayers.empty())
147  {
148  CheckInputData();
149  for (int i = 1; i <= m_nreach; i++)
150  {
151  int order = (int)m_chOrder[i];
152  m_reachLayers[order].push_back(i);
153  }
154  }
155 
156  //initial channel storage
157  if (m_chStorage == NULL)
158  {
159  m_chStorage = new float[m_nreach+1];
160  m_qIn = new float[m_nreach+1];
161  m_qOut = new float[m_nreach+1];
162  m_bankStorage = new float[m_nreach+1];
163  m_seepage = new float[m_nreach+1];
164  m_qsCh = new float[m_nreach+1];
165  m_qiCh = new float[m_nreach+1];
166  m_qgCh = new float[m_nreach+1];
167  m_chWTdepth = new float[m_nreach+1];
168 
169  #pragma omp parallel for
170  for (int i=1; i<=m_nreach; i++)
171  {
172  float qiSub = 0.f;
173  float qgSub = 0.f;
174  if(m_qiSub != NULL)
175  qiSub = m_qiSub[i];
176  if(m_qgSub != NULL)
177  qgSub = m_qgSub[i];
178  m_seepage[i] = 0.f;
179  m_bankStorage[i] = m_Bnk0 * m_chLen[i];
180  m_chStorage[i] = m_Chs0 * m_chLen[i];
181  m_qIn[i] = 0.f;
182  m_qOut[i] = m_qsSub[i] + qiSub + qgSub;
183  m_qsCh[i] = m_qsSub[i];
184  m_qiCh[i] = qiSub;
185  m_qgCh[i] = qgSub;
186  m_chWTdepth[i] = 0.f;
187 
188  }
189  }

192 int IKW_REACH::Execute()
193 {
194  initalOutputs();
195 
196  map<int, vector<int> >::iterator it;
197  for (it = m_reachLayers.begin(); it != m_reachLayers.end(); it++)
198  {
199  // There are not any flow relationship within each routing layer.
200  // So parallelization can be done here.
201  int nReaches = it->second.size();
202  // the size of m_reachLayers (map) is equal to the maximum stream order
203  #pragma omp parallel for
204  for (int i = 0; i < nReaches; ++i)
205  {
206  int reachIndex = it->second[i]; // index in the array
207  ChannelFlow(reachIndex);
208  }
209  }
210 
211  return 0;

501 // modified from OpenLISEM
510 float IKW_REACH::GetNewQ(float qIn, float qLast, float surplus, float alpha, float dt, float dx)
511 {
512  /* Using Newton-Raphson Method */
513  float ab_pQ, dtX, C; //auxillary vars
514  int count;
515  float Qkx; //iterated discharge, becomes Qnew
516  float fQkx; //function
517  float dfQkx; //derivative
518  const float _epsilon = 1e-12f;
519  const float beta = 0.6f;
520 
521  /* if no input then output = 0 */
522  if ((qIn + qLast) <= -surplus*dx)//0)
523  {
524  //itercount = -1;
525  return(0);
526  }
527 
528  /* common terms */
529  ab_pQ = alpha*beta*pow(((qLast+qIn)/2),beta-1);
530  // derivative of diagonal average (space-time)
531 
532  dtX = dt/dx;
533  C = dtX*qIn + alpha*pow(qLast,beta) + dt*surplus;
534  //dt/dx*Q = m3/s*s/m=m2; a*Q^b = A = m2; surplus*dt = s*m2/s = m2
535  //C is unit volume of water
536  // first gues Qkx
537  Qkx = (dtX * qIn + qLast * ab_pQ + dt * surplus) / (dtX + ab_pQ);
538 
539  // VJ 050704, 060830 infil so big all flux is gone
540  //VJ 110114 without this de iteration cannot be solved for very small values
541  if (Qkx < MIN_FLUX)
542  {
543  //itercount = -2;
544  return(0);
545  }
546 
547  Qkx = max(Qkx, MIN_FLUX);
548 
549  count = 0;
550  do {
551  fQkx = dtX * Qkx + alpha * pow(Qkx, beta) - C; /* Current k */
552  dfQkx = dtX + alpha * beta * pow(Qkx, beta - 1); /* Current k */
553  Qkx -= fQkx / dfQkx; /* Next k */
554  Qkx = max(Qkx, MIN_FLUX);
555  count++;
556  //qDebug() << count << fQkx << Qkx;
557  } while(fabs(fQkx) > _epsilon && count < MAX_ITERS_CH);
558 
559  if(Qkx != Qkx)
560  throw ModelException("IKW_OL", "GetNewQ", "Error in iteration!");
561 
562  //itercount = count;
563  return Qkx;
564 }

566 void IKW_REACH::ChannelFlow(int i)
567 {
568  float st0 = m_chStorage[i];
569 
570  float qiSub = 0.f;
571  if(m_qiSub != NULL)
572  qiSub = m_qiSub[i];
573  float qgSub = 0.f;
574  if(m_qgSub != NULL)
575  qgSub = m_qgSub[i];
576 
578  // first add all the inflow water
579  // 1. water from this subbasin
580  float qIn = m_qsSub[i] + qiSub + qgSub;
581 
582  // 2. water from upstream reaches
583  float qsUp = 0.f;
584  float qiUp = 0.f;
585  float qgUp = 0.f;
586  for (size_t j = 0; j < m_reachUpStream[i].size(); ++j)
587  {
588  int upReachId = m_reachUpStream[i][j];
589  qsUp += m_qsCh[upReachId];
590  qiUp += m_qiCh[upReachId];
591  qgUp += m_qgCh[upReachId];
592  }
593  qIn += qsUp + qiUp + qgUp;
594  qIn += m_qUpReach; // m_qUpReach is zero for not-parallel program and qsUp, qiUp and qgUp are zero for parallel computing
595 
596  // 3. water from bank storage
597  float bankOut = m_bankStorage[i] * (1 - exp(-m_aBank));
598 
599  m_bankStorage[i] -= bankOut;
600  qIn += bankOut/m_dt;
601 
602  // add inflow water to storage
603  m_chStorage[i] += qIn*m_dt;
604 
606  // then subtract all the outflow water
607  // 1. transmission losses to deep aquifer, which is lost from the system
608  // the unit of kchb is mm/hr
609  float seepage = m_Kchb/1000.f /3600.f * m_chWidth[i] * m_chLen[i] * m_dt;
610  if(qgSub < 0.001f)
611  {
612  if (m_chStorage[i] > seepage)
613  {
614  m_seepage[i] = seepage;
615  m_chStorage[i] -= seepage;
616  }
617  else
618  {
619  m_seepage[i] = m_chStorage[i];
620  m_chStorage[i] = 0.f;
621  m_qOut[i] = 0.f;
622  m_qsCh[i] = 0.f;
623  m_qiCh[i] = 0.f;
624  m_qgCh[i] = 0.f;
625  return;
626  }
627  }
628  else
629  {
630  m_seepage[i] = 0.f;
631  }
632 
633  // 2. calculate transmission losses to bank storage
634  float dch = m_chStorage[i]/(m_chWidth[i]*m_chLen[i]);
635  float bankInLoss = 2 * m_Kbank/1000.f/3600.f * dch * m_chLen[i] * m_dt ; // m3/s
636  bankInLoss = 0.f;
637  if (m_chStorage[i] > bankInLoss)
638  {
639  m_chStorage[i] -= bankInLoss;
640  }
641  else
642  {
643  bankInLoss = m_chStorage[i];
