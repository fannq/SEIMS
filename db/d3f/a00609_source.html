95 int SRD_MB::Execute()
96 {
97  this->CheckInputData();
98 
99  this->initalOutputs();
100 
101  //the first time
102  if(m_isInitial)
103  {
104  int count = 0;
105  for(int i=0;i<this->m_cellSize;i++)
106  {
107  if((this->m_tMin[i] + this->m_tMax[i]) / 2 < this->m_tsnow) {this->m_SA[i] = this->m_swe0; count++;} //winter
108  else this->m_SA[i] = 0.0f; // other seasons
109  }
110 
111  m_swe = this->m_swe0 * count / this->m_cellSize;
112  m_isInitial = false;
113  }
114 
115  if(this->m_lastSWE == -99.0f) this->m_lastSWE = this->m_swe;
116  if(m_swe < 0.01) //all cells have not snow, so snow redistribution is 0.
117  {
118  if(this->m_lastSWE >= 0.01)
119  {
120  for (int rw = 0; rw < this->m_cellSize; rw++)
121  m_SR[rw] = 0.0f;
122  }
123 
124  this->m_lastSWE = this->m_swe;
125  return true;
126  }
127 
128 
129  //if some cells have snow, start to calculate snow redistribution.
130  if(m_w == NULL) m_w = new float[this->m_cellSize];
131  if(m_wt == NULL) //get wt. wt is constant for the watershed, so save it in m_wt to save time.
132  {
133  m_wt = new float[this->m_cellSize];
134  for (int rw = 0; rw < this->m_cellSize; rw++)
135  m_wt[rw] = 1.0f + 1 / this->m_k_slope * this->m_slope_wind[rw] + 1 / this->m_k_curvature * this->m_curva_wind[rw];
136  }
137 
138  float totalW = 0.0f;
139  for ( int rw = 0; rw < this->m_cellSize; rw++)
140  {
141  float wl = 1.0f;
142  float shc = this->m_shc[rw] * 1000;
143  if(this->m_swe < shc)
144  {
145  wl = shc / this->m_shc_crop +this->m_swe * ( 1 - shc / this->m_shc_crop) / shc ;
146  }
147 
148  m_w[rw] = m_wt[rw] * wl; //save w in the m_w array, this save time for following loop.
149  totalW += m_w[rw];
150  }
151 
152  float weight = this->m_cellSize / totalW; //weight
153  float u = 0.0f; //average wind speed
154  for (int rw = 0; rw < this->m_wsSize; rw++) u+=this->m_ws[rw];
155  u /= this->m_wsSize;
156 
157  for ( int rw = 0; rw < this->m_cellSize; rw++)
158  {
159  float snow = this->m_SA[rw];
160  float tmean = (this->m_tMin[rw] + this->m_tMax[rw]) / 2; //if the temperature is higher than t0, the redistribution is 0
161  if(tmean < this->m_tsnow) snow+= (1+this->m_kblow) * this->m_Pnet[rw];
162  if(snow < 0.01) // if snow is lower than a very small positive value (0.01), consider it is equal to 0.
163  {
164  this->m_SR[rw] = 0.0f;
165  continue;
166  }
167 
168  if(tmean >= this->m_t0)
169  {
170  this->m_SR[rw] = 0.0f;
171  continue;
172  }
173 
174  float wr = m_w[rw] * weight;
175  float sap = m_swe * wr;
176  float scp = sap - this->m_SA[rw];
177 
178  float ut = float(this->m_ut0 + 0.0033 * pow((this->m_tMin[rw] + this->m_tMax[rw])/2 - this->m_t_wind,2.0f));
179  float ww = 0.0f;
180  if(u-ut >= this->m_u0) ww = 1.0f;
181  if(u-ut >= 0 && u-ut < this->m_u0) ww = (u-ut) / this->m_u0;
182 
183  float sr = scp * ww;
184 
185  if(abs(sr) <= snow) this->m_SR[rw] = sr;
186  else this->m_SR[rw] = snow;
187  }
188 
189  this->m_lastSWE = this->m_swe;
190  return 0;
191 }
