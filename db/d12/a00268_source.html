54 int peukerdouglas(char *felfile, char *ssfile,float *p)
55 {
56  MPI_Init(NULL,NULL);
57  {
58  int rank,size;
59  MPI_Comm_rank(MCW,&rank);
60  MPI_Comm_size(MCW,&size);
61  if(rank==0)printf("PeukerDouglas version %s\n",TDVERSION);
62 
63  double begint = MPI_Wtime();
64 
65  long x,y;
66  int k,ik,jk,jomax,iomax,bound;
67  float ndve,emax;
68  float* ndveptr;
69  /* Read elevation headers */
70  tiffIO felev(felfile, FLOAT_TYPE); //input elevation
71  long totalX = felev.getTotalX(); //Globabl x and y
72  long totalY = felev.getTotalY();
73  double dx = felev.getdx(); //cell x and y
74  double dy = felev.getdy();
75  //if(rank==0)
76  // {
77  // float timeestimate=(1e-7*totalX*totalY/pow((double) size,1))/60+1; // Time estimate in minutes
78  // fprintf(stderr,"This run may take on the order of %.0f minutes to complete.\n",timeestimate);
79  // fprintf(stderr,"This estimate is very approximate. \nRun time is highly uncertain as it depends on the complexity of the input data \nand speed and memory of the computer. This estimate is based on our testing on \na dual quad core Dell Xeon E5405 2.0GHz PC with 16GB RAM.\n");
80  // fflush(stderr);
81  // }
82 
83 
84  ndveptr = (float*)felev.getNodata();
85  ndve = *ndveptr;
86  //Create partition and read data
87  tdpartition *elev;
88  elev = CreateNewPartition(felev.getDatatype(), totalX, totalY, dx, dy, felev.getNodata());
89  int elevnx = elev->getnx();
90  int elevny = elev->getny();
91  int globalxstart, globalystart;
92  elev->localToGlobal(0, 0, globalxstart, globalystart);
93 
94  felev.read((long)globalxstart, (long)globalystart, (long)elevny, (long)elevnx, elev->getGridPointer());
95 
96  //Record time reading files
97  double readt = MPI_Wtime();
98 
99 //Create empty partition to store new information
100  tdpartition *selev;
101  selev = CreateNewPartition(FLOAT_TYPE, totalX, totalY, dx, dy, felev.getNodata());//selev is to smooth the elevation
102 
103 //Create empty partition to store new information
104  tdpartition *ss;
105  short ssnodata =-2;
106  ss = CreateNewPartition(SHORT_TYPE, totalX, totalY, dx, dy, ssnodata);//-2 nodata value
107 
108 // SHARE elev to populate the borders
109  elev->share();
110 
111  float floatTemp1;
112  //--Smooth the Grid--
113  for(y=0; y<elevny; y++){
114  for(x=0; x <elevnx; x++)
115  {
116  // NEED function in linearpart to evaluate whether is on edge.
117  // Suggestion **->isOnEdge(x,y) to return true if on edge of global domain or adjacent to a no data value, otherwise to return false.
118  // FIX code below to if on a global edge or adjacent to a no-data value do first "if block"
119 
120  if(rank == 0 && y == 0 || rank == (size-1) && y == (elevny-1) )
121  {
122  selev->setData(x,y,elev->getData(x, y,floatTemp1 ));
123  ss->setData(x,y,(short)0);
124  }
125  else if(x == 0 || x == (elevnx-1) || elev->isNodata(x,y))
126  {
127  selev->setData(x,y,elev->getData(x, y,floatTemp1 ));
128  ss->setData(x,y,(short)0);
129  }
130  else
131  { // THIS is for all the remainder of cells
132 
133  ss->setData((long)x, (long)y, (short)1); // Initializing to 1 for all non edge grid cells
134 
135  float elevwsum = p[0] * elev->getData(x, y, floatTemp1 );
136  float wsum=p[0];
137  if(p[1] > 0.)
138  for(k=1; k<=7; k=k+2)
139  {
140  if(!elev->isNodata(x+d1[k],y+d2[k]))
141  {
142  elevwsum += elev->getData(x+d1[k],y+d2[k],floatTemp1) * (p[1]);
143  wsum += p[1];
144  }
145  }
146  if(p[2] > 0.)
147  for(k=2; k<=8; k=k+2)
148  {
149  if(!elev->isNodata(x+d1[k],y+d2[k]))
150  {
151  elevwsum += elev->getData(x+d1[k],y+d2[k],floatTemp1) *p[2];
152  wsum += p[2];
153  }
154  }
155  elevwsum=elevwsum/wsum;
156  selev->setData(x,y,elevwsum);
157  }
158  }
159  }
160 
161  //-- Put smoothed elevations back in elevation grid--
162  for(y=0; y < elevny; y++)
163  {
164  for(x=0; x < elevnx; x++)
165  {
166  elev->setData(x,y,selev->getData(x,y,floatTemp1));
167  }
168  }
169  elev->share();
170 
171  //--Calculate Streams--
172  for(y=-1; y < elevny; y++){
173  for(x=0; x < elevnx-1; x++)
174  {
175  emax =elev->getData(x,y,floatTemp1);
176  iomax=0;
177  jomax=0;
178  bound=0; /* .false. */
179 
180  /* --FIRST PASS FLAG MAX ELEVATION IN GROUP OF FOUR */
181  for(ik=0; ik<2; ik++)
182  for(jk=1-ik; jk < 2; jk++)
183  {
184  if(elev->isNodata(x+jk,y+ik))
185  bound= 1;
186  else if(elev->getData(x+jk,y+ik,floatTemp1) > emax)
187  {
188  emax=floatTemp1;
189  iomax=ik;
190  jomax=jk;
191  }
192  }
193  /* c---Unflag max pixel */
194  ss->setData(x+jomax,y+iomax,(short)0);
195  /* c---Unflag pixels where the group of 4 touches a boundary */
196  if(bound == 1)
197  {
198  for(ik=0; ik < 2; ik++)
199  for(jk=0; jk< 2; jk++)
200  {
201  ss->setData(x+jk,y+ik,(short)0);
202  }
203  }else{
204  /* i.e. unflag flats. */
205  for(ik=0; ik < 2; ik++)
206  for(jk=0; jk< 2; jk++)
207  {
208  if(elev->getData(x+jk,y+ik,floatTemp1) == emax)
209  {
210  ss->setData(x+jk,y+ik,(short)0);
211  }
212  }
213  }
214 
215  }
216  }
217  //Stop timer
218  double computet = MPI_Wtime();
219  tiffIO outelev(ssfile,SHORT_TYPE,&ssnodata, felev);
220 
221  outelev.write((long)globalxstart, (long)globalystart, (long)elevny, (long)elevnx, ss->getGridPointer());
222  double writet = MPI_Wtime();
223  double dataRead, compute, write, total,temp;
224  dataRead = readt-begint;
225  compute = computet-readt;
226  write = writet-computet;
227  total = writet - begint;
228 
229  MPI_Allreduce (&dataRead, &temp, 1, MPI_DOUBLE, MPI_SUM, MCW);
230  dataRead = temp/size;
231  MPI_Allreduce (&compute, &temp, 1, MPI_DOUBLE, MPI_SUM, MCW);
232  compute = temp/size;
233  MPI_Allreduce (&write, &temp, 1, MPI_DOUBLE, MPI_SUM, MCW);
234  write = temp/size;
235  MPI_Allreduce (&total, &temp, 1, MPI_DOUBLE, MPI_SUM, MCW);
236  total = temp/size;
237 
238  if( rank == 0)
239  printf("Processors: %d\nRead time: %f\nCompute time: %f\nWrite time: %f\nTotal time: %f\n",
240  size , dataRead, compute, write,total);
241 
242  }
243  MPI_Finalize();
244  return 0;
245 }
