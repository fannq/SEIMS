371 void ModuleFactory::ReadDLL(string& id, string& dllID)
372 {
373  // the dll file is already read, return
374  if (m_instanceFuncs.find(id) != m_instanceFuncs.end())
375  return;
376 
377  // check if the dll file exists
378  string moduleFileName = trim(m_modulePath) + string(dllID) + string(Tag_ModuleExt);
379  utils util;
380  if(!util.FileExists(moduleFileName)) throw ModelException("ModuleFactory", "ReadDLL", moduleFileName + " does not exist or has no read permission!");
381 
382  //load library
383 #ifndef linux
384  HINSTANCE handle = LoadLibrary(TEXT(moduleFileName.c_str()));
385  if (handle == NULL) throw ModelException("ModuleFactory", "ReadDLL", "Could not load " + moduleFileName);
386  m_instanceFuncs[id] = InstanceFunction(GetProcAddress(HMODULE(handle), "GetInstance"));
387  m_metadataFuncs[id] = MetadataFunction(GetProcAddress(HMODULE(handle), "MetadataInformation"));
388 #else
389  void* handle = dlopen(moduleFileName.c_str(), RTLD_LAZY);
390  if (handle == NULL)
391  {
392  cout << dlerror() << endl;
393  throw ModelException("ModuleFactory", "ReadDLL", "Could not load " + moduleFileName);
394  }
395  m_instanceFuncs[id] = InstanceFunction(dlsym(handle, "GetInstance"));
396  m_metadataFuncs[id] = MetadataFunction(dlsym(handle, "MetadataInformation"));
397 #endif
398  m_dllHandles.push_back(handle);
399  if(m_instanceFuncs[id] == NULL)
400  throw ModelException("ModuleFactory", "ReadDLL", moduleFileName + " does not implement API function: GetInstance");
401  if(m_metadataFuncs[id] == NULL)
402  throw ModelException("ModuleFactory", "ReadDLL", moduleFileName + " does not implement API function: MetadataInformation");

803 bool ModuleFactory::LoadSettingsFromFile(const char* filename, vector< vector<string> >& settings)
804 {
805  bool bStatus = false;
806  ifstream myfile;
807  string line;
808  utils utl;
809 
810  try
811  {
812  // open the file
813  myfile.open(filename, ios::in);
814  if (myfile.is_open())
815  {

817  {
818  if (myfile.good())
819  {
820  getline(myfile, line);
821  //utl.TrimSpaces(line);
822  line = trim(line);
823  if ((line.size() > 0) && (line[0] != '#')) // ignore comments and empty lines
824  {
825  // parse the line into separate item
826  vector<string> tokens = utl.SplitString(line, '|');
827  // is there anything in the token list?
828  if (tokens.size() > 0)
829  {
830  for (size_t i=0; i<tokens.size(); i++)
831  {
832  //utl.TrimSpaces(tokens[i]);
833  tokens[i] = trim(tokens[i]);
834  }
835  // is there anything in the first item?
836  if (tokens[0].size() > 0)
837  {
838  // there is something to add so resize the header list to append it
839  int sz = settings.size(); // get the current number of rows
840  settings.resize(sz+1); // resize with one more row
841 
842  settings[sz] = tokens;
843 
844  bStatus = true; // consider this a success
845  } // if there is nothing in the first item of the token list there is nothing to add to the header list
846  }
847  }
848  }
849  }
850  bStatus = true;
851  myfile.close();
852  }
853  }
854  catch (...)
855  {
856  myfile.close();
857  //throw;
858  }
859 
860  return bStatus;
