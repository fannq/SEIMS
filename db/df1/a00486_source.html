182 int SplashEro_Park::Execute()
183 {
184  //set the input data which was not available right now, this will be delete when the data or module is available
185  //setNotAvailableInput();
186 
187  CheckInputData();
188 
189  initial();
190  //StatusMsg("executing SplashEro_Park");
191 #pragma omp parallel for
192  for (int i = 0; i < m_size; i++)
193  {
194  //intensity in mm/h
195  float RainInten = m_Rain[i] * 3600 / m_TimeStep;
196 
197  //rainfall between plants in mm //Rainc->Drc = Rain->Drc * _dx/DX->Drc;
198  // correction for slope dx/DX, water spreads out over larger area
199  float s = max(0.001f, m_Slope[i]);
200  float S0 = sin(atan(s));
201  float Dm = 0.00124f * pow(RainInten, 0.182f);
202  float waterdepth = (m_sr[i] + m_depression[i]) / 1000; // mm convert to m
203  float Fw, Dr;
204  if (waterdepth > Dm)
205  {
206  Fw = exp(1 - waterdepth / Dm);
207  }
208  else
209  {
210  Fw = 1;
211  }
212  // kg/(m2*min)
213  Dr = m_Omega * Fw * m_USLE_C[i] * m_USLE_K[i] * pow(RainInten,2.f) * (2.96f * pow(S0,0.79f) + 0.56f);
214  // convert kg/(m2*min) to kg/cell
215  float cellareas = (m_cellWith/cos(atan(s))) * m_cellWith;
216  m_DETSplash[i] = Dr * (m_TimeStep/60) * cellareas;
217  // // Deal with all exceptions:
218  //if (m_SnowCover[i]>0)
219  //{
220  // m_DETSplash[i] = (1-m_SnowCover[i]) * m_DETSplash[i]; //no splash on snow deck
221  //}
222 
223  }
224  //StatusMsg("end of executing SplashEro_Park");
