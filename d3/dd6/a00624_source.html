33 int PER_PI::Execute()
34 {
35  CheckInputData();
36 
37  if(m_perc == NULL)
38  {
39  m_perc = new float*[m_nCells];
40 
41  #pragma omp parallel for
42  for (int i = 0; i < m_nCells; i++)
43  {
44  m_perc[i] = new float[m_nSoilLayers];
45  for (int j = 0; j < m_nSoilLayers; j++)
46  m_perc[i][j] = 0.f;
47  }
48  }
49  if(m_upSoilDepth == NULL)
50  m_upSoilDepth = new float[m_nSoilLayers];
51  #pragma omp parallel for
52  for (int i = 0; i < m_nCells; i++)
53  {
54  float k, maxSoilWater, fcSoilWater, swater;
55  // Update soil layers from solid two layers to multi-layers by m_nSoilLayers. By LJ
56  int curSoilLayers = -1, j;
57  m_upSoilDepth[0] = m_soilDepth[i][0];
58  for(j = 1; j < m_nSoilLayers; j++)
59  {
60  if(!FloatEqual(m_soilDepth[i][j],NODATA_VALUE))
61  m_upSoilDepth[j] = m_soilDepth[i][j] - m_soilDepth[i][j-1];
62  else
63  break;
64  }
65  curSoilLayers = j;
66 
67  m_somo[i][0] += m_infil[i] / m_upSoilDepth[0];
68  for (int j = 0; j < curSoilLayers; j++)
69  {
70  //No movement if soil moisture is below field capacity
71  m_perc[i][j] = 0.f;
72  // for the upper two layers, soil may be frozen
73  if(j == 0 && m_soilT[i] <= m_frozenT)
74  continue;
75  if (m_somo[i][j] > m_fc[i][j])
76  {
77  maxSoilWater = m_upSoilDepth[j] * m_porosity[i][j];
78  swater = m_upSoilDepth[j] * m_somo[i][j];
79 
80  fcSoilWater = m_upSoilDepth[j] * m_fc[i][j];
81 
82  //the moisture content can exceed the porosity in the way the algorithm is implemented
83  if (m_somo[i][j] > m_porosity[i][j])
84  k = m_ks[i][j];
85  else
86  {
87  float dcIndex = 2.f/m_poreIndex[i][j] + 3.f; // pore disconnectedness index
88  k = m_ks[i][j] * pow(m_somo[i][j]/m_porosity[i][j], dcIndex);
89  }
90 
91  m_perc[i][j] = k * m_dt/3600.f;
92 
93  if(swater - m_perc[i][j] < fcSoilWater)
94  m_perc[i][j] = swater - fcSoilWater;
95  else if(swater - m_perc[i][j] > maxSoilWater)
96  m_perc[i][j] = swater - maxSoilWater;
97 
98  //Adjust the moisture content in the current layer, and the layer immediately below it
99  m_somo[i][j] -= m_perc[i][j]/m_upSoilDepth[j];
100 
101  if(j < curSoilLayers-1)
102  m_somo[i][j+1] += m_perc[i][j]/m_upSoilDepth[j+1];
103 
104  if(m_somo[i][j] != m_somo[i][j] || m_somo[i][j] < 0.f)
105  {
106  cout << MID_PER_PI<<" CELL:" << i <<", Layer: "<<j<< "\tPerco:" << swater << "\t" <<
107  fcSoilWater << "\t" << m_perc[i][j] << "\t" << m_upSoilDepth[j] << "\tValue:" << m_somo[i][j] << endl;
108  throw ModelException(MID_PER_PI, "Execute", "moisture is less than zero.");
109  }
110  }
111  }
112  }
113  return 0;
114 }
