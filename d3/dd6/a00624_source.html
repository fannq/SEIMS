35 int PER_PI::Execute()
36 {
37  CheckInputData();
38 
39  if(m_perc == NULL)
40  {
41  m_perc = new float*[m_nCells];
42 
43  #pragma omp parallel for
44  for (int i = 0; i < m_nCells; i++)
45  {
46  m_perc[i] = new float[m_nSoilLayers];
47  for (int j = 0; j < m_nSoilLayers; j++)
48  m_perc[i][j] = 0.f;
49  }
50  }
51 
52  #pragma omp parallel for
53  for (int i = 0; i < m_nCells; i++)
54  {
55  float k, maxSoilWater, fcSoilWater, swater;
56  float depth[2];
57  depth[0] = m_upSoilDepth;
58  depth[1] = m_rootDepth[i] - m_upSoilDepth;
59  if(depth[1] < 0)
60  {
61  ostringstream oss;
62  oss << "The root depth at cell(" << i << ") is " << m_rootDepth[i] << ", and is less than the upper soil depth (" << m_upSoilDepth << endl;
63  throw ModelException("PER_PI", "Execute", oss.str());
64  }
65 
66  m_sm[i][0] += m_infil[i] / depth[0];
67  for (int j = 0; j < m_nSoilLayers; j++)
68  {
69  //No movement if soil moisture is below field capacity
70  m_perc[i][j] = 0.f;
71 
72  // for the upper two layers, soil may be frozen
73  if(j == 0 && m_soilT[i] <= m_frozenT)
74  continue;
75 
76  if (m_sm[i][j] > m_fc[i][j])
77  {
78  maxSoilWater = depth[j] * m_porosity[i][j];
79  swater = depth[j] * m_sm[i][j];
80 
81  fcSoilWater = depth[j] * m_fc[i][j];
82 
83  //the moisture content can exceed the porosity in the way the algorithm is implemented
84  if (m_sm[i][j] > m_porosity[i][j])
85  k = m_ks[i][j];
86  else
87  {
88  float dcIndex = 2.f/m_poreIndex[i][j] + 3.f; // pore disconnectedness index
89  k = m_ks[i][j] * pow(m_sm[i][j]/m_porosity[i][j], dcIndex);
90  }
91 
92  m_perc[i][j] = k * m_dt/3600.f;
93 
94  if(swater - m_perc[i][j] < fcSoilWater)
95  m_perc[i][j] = swater - fcSoilWater;
96  else if(swater - m_perc[i][j] > maxSoilWater)
97  m_perc[i][j] = swater - maxSoilWater;
98 
99  //Adjust the moisture content in the current layer, and the layer immediately below it
100  m_sm[i][j] -= m_perc[i][j]/depth[j];
101 
102  if(j < m_nSoilLayers-1)
103  m_sm[i][j+1] += m_perc[i][j]/depth[j+1];
104 
105  if(m_sm[i][j] != m_sm[i][j] || m_sm[i][j] < 0.f)
106  {
107  cout << "PER_PI CELL:" << i << "\tPerco:" << swater << "\t" <<
108  fcSoilWater << "\t" << m_perc[i][j] << "\t" << depth[j] << "\tValue:" << m_sm[i][j] << endl;
109  throw ModelException("PER_PI", "Execute", "moisture is less than zero.");
110  }
111  }
112  }
113  }
114  return 0;
115 }
