00077 float PETHargreaves::MaxSolarRadiation(int day,float lat)
00078 {
00079       lat = lat*3.1415926/180;
00080       //Calculate Daylength
00081       //calculate solar declination: equation 2.1.2 in SWAT manual
00082       float sd = asin(0.4f * sin((day - 82.0f) / 58.09f));  //365/2pi = 58.09
00083 
00084       //calculate the relative distance of the earth from the sun
00085       //the eccentricity of the orbit
00086       //equation 2.1.1 in SWAT manual
00087       float dd = 1.0f + 0.033f * cos(day / 58.09f);
00088 
00089       //daylength = 2 * Acos(-Tan(sd) * Tan(lat)) / omega
00090       //where the angular velocity of the earth's rotation, omega, is equal
00091       //to 15 deg/hr or 0.2618 rad/hr and 2/0.2618 = 7.6374
00092       //equation 2.1.6 in SWAT manual
00093 
00094       float h = 0.0f;
00095       float ch = -sin(lat) * tan(sd) / cos(lat);
00096       if (ch > 1.f) //ch will be >= 1. if latitude exceeds +/- 66.5 deg in winter
00097         h = 0.0f;
00098       else if (ch >= -1.0f)
00099         h = acos(ch);
00100       else
00101         h = 3.1416f; //latitude exceeds +/- 66.5 deg in summer
00102       
00103       float dayl = 7.6394f * h;
00104           
00105       //Calculate Potential (maximum) Radiation !!
00106       //equation 2.2.7 in SWAT manual
00107       float ys = sin(lat) * sin(sd);
00108       float yc = cos(lat) * cos(sd);
00109       
00110       return 30.f * dd * (h * ys + yc * sin(h));

00122 #pragma omp parallel for
00123     for (int i = 0; i < m_size; ++i)
00124     {   
00125         float tMean = (m_tMax[i] + m_tMin[i])/2;

00134         float petValue = 0.0023f * h0 * pow(m_tMax[i]-m_tMin[i], 0.5f)

00136         m_pet[i] = m_petFactor * max(0.0f, petValue);
