00061 int HS_WB::Execute()
00062 {   
00063     
00064     initalOutputs();
00065 
00067     #pragma omp parallel for
00068     for (int i = 1; i <= m_nReaches; i++)
00069     {
00070         m_qsTemp[i] = 0.f;
00071         m_qiTemp[i] = 0.f;
00072     }
00073 
00074     int reachId;
00075     for (int i = 0; i < m_nCells; i++)
00076     {
00077         if(m_streamLink[i] > 0)
00078         {
00079 #ifdef MULTIPLY_REACHES
00080             reachId = int(m_streamLink[i]);
00081 #else
00082             reachId = 1;
00083 #endif
00084             m_qsTemp[reachId] += m_qs[i];
00085             if (m_qi != NULL)
00086                 m_qiTemp[reachId] += m_qi[i];
00087         }
00088     }
00089 
00090     m_qsInput[0] = 0.f;
00091     m_qiInput[0] = 0.f;
00092     for (int i = 1; i <= m_nReaches; i++)
00093     {
00094         m_qsInput[i] = (m_qsInput[i]*(m_tsCounter-1) + m_qsTemp[i]) / m_tsCounter;
00095         m_qiInput[i] = (m_qiInput[i]*(m_tsCounter-1) + m_qiTemp[i]) / m_tsCounter;
00096 
00097         m_qsInput[0] += m_qsInput[i];
00098         m_qiInput[0] += m_qiInput[i];
00099     }
00100 
00101     m_tsCounter++;
00102 
00103     return 0;
00104 }
