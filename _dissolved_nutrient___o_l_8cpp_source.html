00208 void DissolvedNutrient_OL::concentration(int id)
00209 {
00210     float Fcl = m_clayFrac[id];
00211     float K_NH4 = 1.34f + 0.038f*Fcl;    // partition coefficient
00212     float K_P = 100 + 2.5f * Fcl;     // partition coefficient
00213     // dissoloved nutrient concentration
00214     //concentration at the end of the time step (Knisel et al. 1994)
00215     float Q = m_CELLQ[id];   //m3/s
00216     float h = m_CELLH[id] * m_cellA[id]/1000;  //m3
00217     float f = m_infil[id] * m_cellA[id]/1000;  //m3
00218     float S = m_sd[id] * m_cellA[id]/1000;     //m3
00219     float p = m_Porosity[id];
00220     if(S!=0)
00221     {
00222         if (h==0)
00223         {
00224             h = m_whtoCh[id];
00225         }
00226         m_X2 = (h + f) / S;
00227         m_X3 = h / S;
00228     }
00229     else
00230     {
00231         m_X2 = 0;
00232         m_X3 = 0;
00233     }
00234     float temp1 = Q * m_dt + f + S;    //m3
00235     float temp2 = 2.65f*(1-p);
00236     float X_NH4 = temp1 /(p + temp2 * K_NH4);
00237     float Cfin_NH4 = m_PreConcNH4[id] * exp(-X_NH4);   //kg/kg-soil
00238 
00239     float X_P = temp1 /(p + temp2 * K_P);
00240     float Cfin_P = m_PreConcP[id] * exp(-X_P);    //kg/kg-soil
00241 
00242     float coe_extra = 0;   //extraction coefficient
00243     if(K_NH4 <=1.f) 
00244         coe_extra = 0.5f;
00245     else if(K_NH4>1.f && K_NH4 <=10.f)
00246         coe_extra = 0.598f * exp(-0.179f * K_NH4);
00247     else
00248         coe_extra = 0.1f;
00249 
00250     float conc_NH4 = Cfin_NH4 * coe_extra * 1000 / (1 + K_NH4 * coe_extra);   //kg/m3
00251 
00252     float conc_P = 0.1f * Cfin_P /(1 + 0.1f * K_P) * 1000; // kg/m3
00253 
00254     float X_NO3;
00255     if(p!=0)
00256         X_NO3 = temp1 / p;
00257     else
00258         X_NO3 = 0.001;
00259     float Cfin_NO3 = m_PreConcNO3[id] * exp(-X_NO3);   //kg/kg-soil
00260     float conc_NO3 = 0.5f * Cfin_NO3 * 1000;
00261     //conver to kg/s
00262     float wat = Q + f/m_dt;   // m3/s
00263     m_CP[id] = conc_P * wat;
00264     m_CNH4[id] = conc_NH4 * wat;
00265     m_CNO3[id] = conc_NO3 * wat;
00266 
00267     // set nutrients concentration of this time step to previous time step
00268     /*m_PreConcP[id] = Cfin_P;
00269     m_PreConcNH4[id] = Cfin_NH4;
00270     m_PreConcNO3[id] = Cfin_NO3;*/
00271 }

00304 void DissolvedNutrient_OL::OverlandFlow(int id)
00305 {
00306     concentration(id);
00307     //sum the upstream overland flow
00308     float flowwidth = m_flowWidth[id];
00309     float PUp = 0.0f;
00310     float NH4Up = 0.0f;
00311     float NO3Up = 0.0f;
00312     for (int k = 1; k <= (int)m_flowInIndex[id][0]; ++k)
00313     {
00314         int flowInID = (int)m_flowInIndex[id][k];
00315         PUp  += m_CP[flowInID];
00316         NH4Up  += m_CNH4[flowInID];
00317         NO3Up  += m_CNO3[flowInID];
00318     }
00319     // if the channel width is greater than the cell width
00320     if (m_streamLink[id] >= 0 && flowwidth <=0)
00321     {
00322         m_DissovPToCh[id] = PUp * m_dt;  // convert to kg
00323         m_AmmoniumToCh[id] = NH4Up * m_dt;
00324         m_NitrateToCh[id] = NO3Up * m_dt;
00325         return;
00326     }
00327     
00328     float Pinflow = PUp + m_CP[id];
00329     float NH4inflow = NH4Up + m_CNH4[id];
00330     float NO3inflow = NO3Up + m_CNO3[id];
00331     //The amount of dissolved nutrient in transit 
00332     float Ptransit = (m_SNPT_P[id] + Pinflow)/(1 + m_X2);
00333     float NH4transit = (m_SNPT_NH4[id] + NH4inflow)/(1 + m_X2);
00334     float NO3transit = (m_SNPT_NO3[id] + NO3inflow)/(1 + m_X2);
00335     //The outflow of dissolved nutrient  kg/s
00336     m_DissovP[id] = Ptransit * m_X3;
00337     m_Ammonium[id] = NH4transit * m_X3;
00338     m_Nitrate[id] = NO3transit * m_X3;
00339     //The new value of SNPT
00340     float ratio = max(0.0f, 1-m_X2);
00341     m_SNPT_P[id] = Pinflow + m_DissovP[id]*ratio;
00342     m_SNPT_NH4[id] = NH4inflow + m_Ammonium[id]*ratio;
00343     m_SNPT_NO3[id] = NO3inflow + m_Nitrate[id]*ratio;
00344 
00345     // compute nutrients to channel 
00346     float fractiontochannel = 0.0f;
00347     if (m_chWidth[id] > 0)
00348     {
00349         float tem = m_ChV[id] * m_dt;
00350         fractiontochannel = 2*tem / (m_dx - m_chWidth[id]);
00351         fractiontochannel = min(fractiontochannel, 1.0f);
00352         // convert to kg
00353         m_DissovPToCh[id] = m_DissovP[id] * m_dt * fractiontochannel;
00354         m_AmmoniumToCh[id] = m_Ammonium[id] * m_dt * fractiontochannel;
00355         m_NitrateToCh[id] = m_Nitrate[id] * m_dt * fractiontochannel;
00356     }
00357     /*if (m_chWidth[id] > 0)
00358     {
00359     float flowA = m_flowLen[id] * m_chWidth[id];
00360     float ratioA = flowA / m_cellA[id];
00361     m_DissovPToCh[id] = m_DissovP[id] * m_dt * ratioA;
00362     m_AmmoniumToCh[id] = m_Ammonium[id] * m_dt * ratioA;
00363     m_NitrateToCh[id] = m_Nitrate[id] * m_dt * ratioA;
00364     }*/
00365 }
