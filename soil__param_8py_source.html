00020     #wilting point
00021     wpt = -0.024*s + 0.487*c + 0.006*om + 0.005*s*om - 0.013*c*om + 0.068*s*c + 0.031
00022     wp = 1.14*wpt - 0.02
00023     
00024     #field capacity
00025     fct = -0.251*s + 0.195*c + 0.011*om + 0.006*s*om - 0.027*c*om + 0.452*s*c + 0.299
00026     fc = fct + 1.283*fct*fct - 0.374*fct - 0.015
00027     
00028     s33t = 0.278*s + 0.034*c + 0.022*om - 0.018*s*om - 0.027*c*om - 0.584*s*c + 0.078
00029     s33 = 1.636*s33t - 0.107
00030     
00031     sat = fc + s33 - 0.097*s + 0.043
00032     
00033     pn = 2.65*(1.0 - sat)
00034     
00035     # df means density effects
00036     p_df = pn
00037     sat_df = 1 - p_df/2.65
00038     fc_df = fc - 0.2*(sat - sat_df)

00040     # pore disconnectedness index
