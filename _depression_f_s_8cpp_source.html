00081 int DepressionFS::Execute()
00082 {
00083     initalOutputs();
00084 
00085     if(m_checkInput)
00086     {
00087         CheckInputData();   
00088         m_checkInput = false;
00089     }
00090 
00091     #pragma omp parallel for
00092     for (int i = 0; i < m_size; ++i)
00093     {
00094         // sr is temporarily used to stored the water depth including the depression storage
00095         float hWater = m_sr[i];
00096         if (hWater <= m_depCap[i])
00097         {
00098             m_sd[i] = hWater;
00099             m_sr[i] = 0.f;
00100         }
00101         else
00102         {
00103             m_sd[i] = m_depCap[i];
00104             m_sr[i] = hWater - m_depCap[i];
00105         }
00106         
00107         m_storageCapSurplus[i] = m_depCap[i] - m_sd[i];
00108 
00109     }
00110 
00111     return 0;
00112 }
