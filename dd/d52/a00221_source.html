61 template <class elemType>
62 void GPRO::DeComposition<elemType>::
63 _smpl1DDcmp(int &subBegin, int &subEnd,
64  int glbBegin, int glbEnd,
65  int nSubSpcs, int iSubSpc) const {
66  int glbRange = glbEnd - glbBegin + 1;
67  int remainder = glbRange % nSubSpcs;
68  int blockSize;
69  if(remainder == 0) {
70  blockSize = glbRange / nSubSpcs;
71  subBegin = glbBegin + iSubSpc * blockSize;
72  subEnd = subBegin + blockSize - 1;
73  }
74  else {
75  blockSize = glbRange / nSubSpcs + 1;
76  if(iSubSpc < remainder) {
77  subBegin = glbBegin + iSubSpc * blockSize;
78  subEnd = subBegin + blockSize - 1;
79  }
80  else {
81  if(iSubSpc == remainder) {
82  subBegin = glbBegin+ iSubSpc * blockSize;
83  subEnd = subBegin + blockSize - 2;
84  }
85  else {
86  subBegin = glbBegin
87  + remainder * blockSize
88  + (iSubSpc - remainder)*(blockSize-1);
89  subEnd = subBegin + blockSize - 2;
90  }
91  }
92  }
93 }

115 template <class elemType>
116 bool GPRO::DeComposition<elemType>::
117 rowDcmp(MetaData &metaData,
118  int nSubSpcs) const
119 {
120  if(nSubSpcs < 1 ||
121  nSubSpcs > _glbWorkBR.nRows()) {
122  cerr << __FILE__ << " " << __FUNCTION__ \
123  << " Error: invalid number of SubSpaces (" << nSubSpcs << ")" \
124  << " considering the global working bounding rectangle (" << _glbWorkBR << ")" \
125  << endl;
126  return false;
127  }
128 
129  DomDcmpType dcmpType = ROWWISE_DCMP;
130  metaData._domDcmpType = dcmpType;
131 
132  int glbBegin = _glbWorkBR.nwCorner().iRow();
133  int glbEnd = _glbWorkBR.seCorner().iRow();
134 
135 
136  int iSubSpc = metaData.myrank;
137  int subBegin, subEnd;
138  _smpl1DDcmp(subBegin, subEnd,
139  glbBegin, glbEnd,
140  nSubSpcs, iSubSpc);
141  CellCoord nwCorner(subBegin + _pNbrhood->minIRow(),
142  0);
143  CellCoord seCorner(subEnd + _pNbrhood->maxIRow(),
144  _glbDims.nCols() - 1);
145  CoordBR subMBR(nwCorner, seCorner);
146 
147  metaData._MBR = subMBR;
148 
149  SpaceDims dims(subMBR.nRows(), subMBR.nCols());
150  metaData._localdims = dims;
151  CoordBR workBR;
152  if(!_pNbrhood->calcWorkBR(workBR, dims))
153  {
154  return false;
155  }
156  metaData._localworkBR = workBR;
157 
158  return true;
159 }
