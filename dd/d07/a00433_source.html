166 void clsSiteData::readSiteBasicInfo(string tableName)
167 {
168  string path = m_databasePath + File_HydroClimateDB;
169  if(!DBManager::IsTableExist(path,"stations"))
170  throw ModelException( "clsSiteData","readSiteBasicInfo",
171  "The database " + path +
172  " dose not exist or the table stations does not exist in this database.");
173 
174  if(!DBManager::IsTableExist(path,tableName))
175  throw ModelException( "clsSiteData","readSiteBasicInfo",
176  "The database " + path +
177  " dose not exist or the table "+tableName+" does not exist in this database.");
178 
179  //read data
180  DBManager dbman;
181  try
182  {
183  // open the hydrclimate database
184  dbman.Open(path);
185  // if there is no error
186  if (!dbman.IsError())
187  {
188  // create the SQL query for the stations table
189  //ID, NAME, XPR, YPR, LAT, LONG, ELEVATION, TYPE, UNITS, AREA, STARTDATE, ENDDATE, INTERVAL, TABLENAME
190  string strSQL = "SELECT ID, NAME, XPR, YPR, LAT, LONG, ELEVATION, AREA, TYPE FROM stations WHERE TABLENAME='" +
191  tableName+"'";
192  // run the query
193  slTable* tbl = dbman.Load(strSQL);
194  // if the query is successful
195  if (tbl != NULL)
196  {
197  // if there is at least one record
198  if (tbl->nRows > 0)
199  {
200  utils util;
201 
202  // remember row 0 contains the field names not values
203  m_ID = atoi(tbl->FieldValue(1, 0).c_str());
204  m_Name = tbl->FieldValue(1, 1).c_str();
205  m_XPR = (float)atof(tbl->FieldValue(1, 2).c_str());
206  m_YPR = (float)atof(tbl->FieldValue(1, 3).c_str());
207  m_Latitude = (float)atof(tbl->FieldValue(1, 4).c_str());
208  m_Longitude = (float)atof(tbl->FieldValue(1, 5).c_str());
209  m_Elevation = (float)atof(tbl->FieldValue(1, 6).c_str());
210  m_Area = (float)atof(tbl->FieldValue(1, 7).c_str());
211 
212  string type = tbl->FieldValue(1, 8);
213  if(clsHydroClimateData::IsHydroClimateDataType(&type))
214  {
215  m_timeSerieseData[type] = new clsHydroClimateData(m_databasePath,tableName);
216  }
217  else
218  {
219  throw ModelException( "clsSiteData","readSiteBasicInfo",
220  "The data type for 'station:" + m_Name + ",tableName:"+tableName+
221  "' is not correct.");
222  }
223  }
224  else
225  throw ModelException( "clsSiteData","readSiteBasicInfo",
226  "There is no row whose table name is " +tableName);
227 
228  delete tbl;
229  }
230  tbl = NULL;
231  }
232  dbman.Close();
233  }
