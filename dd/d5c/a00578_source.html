223  float qUp = 0.f;
224  float hUp = 0.f;
225 
226  if (iReach == 0 && iCell == 0)

228  qUp = m_qUpReach;
229  }
230 
231  // inflow from upstream channel
232  if (iCell == 0)// inflow of this cell is the last cell of the upstream reach
233  {
234  for (size_t i = 0; i < m_reachUpStream[iReach].size(); ++i)

237  if (upReachId >= 0)

239  int upCellsNum = m_reachs[upReachId].size();

241  qUp += m_qCh[upReachId][upCellsNum-1];
242

244  if (hWater > hUp)
245  hUp = hWater;
246  }
247  }
248  }
249  else
250  {
251  qUp = m_qCh[iReach][iCell-1];
252  hUp = m_hCh[iReach][iCell-1];
253  }
254 
255  float b = 0.6f;
256  float h = m_hCh[iReach][iCell];
257  float dx = m_flowLen[iReach][iCell];
258 
259  float qLat = m_prec[id]/1000.f * m_chWidth[id];
260  if (m_qs != NULL)
261  qLat += m_qs[id]/dx;
262  if (m_qi != NULL)
263  qLat += m_qi[id]/dx;
264 
265  if (qLat < MIN_FLUX && qUp < MIN_FLUX)
266  {
267  m_hCh[iReach][iCell] = 0.f;
268  m_qCh[iReach][iCell] = 0.f;
269  return;
270  }
271 
272 
273  float perim = 2.f*h + m_chWidth[iReach];
274  float sf = (hUp - m_elevation[id] - h)/dx;
275  if (sf < MINI_SLOPE)
276  sf = MINI_SLOPE;
277  float c = 1.f/3600.f * m_manningScalingFactor * m_reachN[iReach] * pow(perim, _23) / sqrt(sf);
278  c = pow(c, 0.6f);
279 
280 
281  float d = 1.f;
282  int counter = 0;
283  float qLast = m_qCh[iReach][iCell];
284  float qNew = qLast;
285  if (qNew < MIN_FLUX)
286  qNew = qLat;
287 
288  while (abs(d) > MIN_FLUX && counter < 10)
289  {
290  d = (qNew*m_dt/dx + c*Power(qNew, 0.6f) - qUp*m_dt/dx - c*Power(qLast, 0.6f) - qLat*m_dt) / (m_dt/dx + c*0.6f/Power(qNew, 0.4f));
291 
292  //if(d != d)
293  // int test = 1;
294  qNew -= d;
295  counter++;
296  }
297 
298  if (qNew < 0.f)
299  qNew = 0.f;
300 
301  float qAvail = m_hCh[iReach][iCell]*m_chWidth[iReach]*dx/m_dt + qLat*dx + qUp;
302 
303  if (qNew > qAvail)
304  {
305  m_qCh[iReach][iCell] = qAvail;
306  m_hCh[iReach][iCell] = 0.f;
307  }
308  else
309  {
310  m_qCh[iReach][iCell] = qNew;//m_hCh[iReach][iCell]

320 
321  initalOutputs();
322  //Output1DArray(m_nCells, m_prec, "f:\\p2.txt");

325  {
326  // There are not any flow relationship within each routing layer.
