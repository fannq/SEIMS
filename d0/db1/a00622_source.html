122 int IUH_IF::Execute()
123 {
124  this->CheckInputData();
125 
126  this->initalOutputs();
127 
128  #pragma omp parallel for
129  for (int n = 0; n<m_nsub+1; n++)
130  {
131  m_Q_SBIF[n] = 0.0f; // delete value of last time step
132  }
133 
134 
135  int nt = 0;
136  float qs_cell = 0.0f;
137  float area = m_CellWidth * m_CellWidth;
138 
139  //#pragma omp parallel for
140  for (int i = 0; i < m_nCells; i++)
141  {
142  //forward one time step
143  for(int j=0; j<m_cellFlowCols; j++)
144  {
145  if(j != m_cellFlowCols-1)
146  m_cellFlow[i][j] = m_cellFlow[i][j+1];
147  else
148  m_cellFlow[i][j] = 0.0f;
149  }
150 
151  //add today's flow
152  int subi = (int)m_subbasin[i];
153 
154  if (m_nsub == 1)
155  {
156  subi = 1;
157  }
158  else if(subi >= m_nsub+1)
159  {
160  ostringstream oss;
161  oss << subi;
162  throw ModelException("IUH_IF","Execute","The subbasin "+oss.str()+" is invalid.");
163  }
164 
165  float v_rs = m_ssru[i];
166  if (v_rs > 0.0f)
167  {
168  int min = int(this->m_iuhCell[i][0]);
169  int max = int(this->m_iuhCell[i][1]);
170  int col = 2;
171  for(int k=min;k<=max;k++)
172  {
173  this->m_cellFlow[i][k] += v_rs/1000.0f * m_iuhCell[i][col] * area/m_TimeStep;
174  col++;
175  }
176  }
177  //#pragma omp critical
178  {
179  m_Q_SBIF[subi] += this->m_cellFlow[i][0]; //get new value
180  }
181  }
182 
183  float tmp = 0.f;
184  //#pragma omp parallel for reduction(+:tmp)
185  for (int n = 1; n<m_nsub+1; n++)
186  {
187  tmp += m_Q_SBIF[n]; //get overland flow routing for entire watershed.
188  }
189  m_Q_SBIF[0] = tmp;
