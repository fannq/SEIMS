90 float PETHargreaves::MaxSolarRadiation(int day,float lat)
91 {
92  lat = lat*3.1415926/180;
93  //Calculate Daylength
94  //calculate solar declination: equation 1:1.1.2 in SWAT Theory 2009, p31
95  float sd = asin(0.4f * sin((day - 82.0f) / 58.09f));
96 
97  //calculate the relative distance of the earth from the sun
98  //also called the eccentricity correction factor of the orbit, Duffie and Beckman(1980)
99  //equation 1:1.1.1 in SWAT Theory 2009, p30
100  float dd = 1.0f + 0.033f * cos(day / 58.09f);
101 
102  //daylength = 2 * Acos(-Tan(sd) * Tan(lat)) / omega
103  //where the angular velocity of the earth's rotation, omega, is equal
104  //to 15 deg/hr or 0.2618 rad/hr and 2/0.2618 = 7.6374
105  //equation 2.1.6 in SWAT manual
106 
107  float h = 0.0f;
109  float ch = -sin(lat) * tan(sd) / cos(lat);
110  if (ch > 1.f) //ch will be >= 1. if latitude exceeds +/- 66.5 deg in winter
111  h = 0.0f;
112  else if (ch >= -1.0f)
113  h = acos(ch);
114  else
115  h = 3.1416f; //latitude exceeds +/- 66.5 deg in summer
116 
117  float dayl = 7.6394f * h;
118 
119  //Calculate Potential (maximum) Radiation !!
121  float ys = sin(lat) * sin(sd);
122  float yc = cos(lat) * cos(sd);
124  return 30.f * dd * (h * ys + yc * sin(h));
125 }

127 int PETHargreaves::Execute()
128 {
129  if(!this->CheckInputData()) return false;
130  if(this->m_pet == NULL)
131  {
132  this->m_pet = new float[this->m_size];
133  }
134  int d = JulianDay(this->m_date);
135 #pragma omp parallel for
136  for (int i = 0; i < m_size; ++i)
137  {
138  float tMean = (m_tMax[i] + m_tMin[i])/2;
139  float srMax = this->MaxSolarRadiation(d, m_latitude[i]);
140 
142  float latentHeat = 2.501f - 0.002361f * tMean;
145  float h0 = srMax * 37.59f / 30.0f;
148  float petValue = m_HCoef_pet * h0 * pow(m_tMax[i]-m_tMin[i], 0.5f)
149  * (tMean + 17.8f) / latentHeat;
150  m_pet[i] = m_petFactor * max(0.0f, petValue);
151  }
152  return 0;
153 }
