75 ### Revised by LiangJun Zhu, 2016-5-21
76 ### @param[in] nlyrs Soil layers number
77 ### @param[in] depth Depth from soil surface to bottom of layer (mm), e.g., 100, 400, 800.
78 ### @param[in] om organic matter
79 ### @param[in] clay
80 ### @param[in] rock
81 ### @param[in] bd bulk density
82 def SoilChemProperties(nlyrs, depth, om, clay, rock, bd):
83  ## Calculate thick of each layer
84  sol_thick = [depth[0]]
85  for k in range(1, nlyrs):
86  sol_thick.append(depth[k] - depth[k - 1])
87  cbn = numpy.multiply(om, 0.58) ## van bemmelen coefficient
88  ## Define some const values
89  nactfr = .02
90  ## Initialize some layer-based temporary and output variables
91  sol_mass = numpy.zeros((nlyrs))
92  sol_cmass = numpy.zeros((nlyrs))
93 
94  sol_fop = numpy.ones((nlyrs)) # phosphorus stored in the fresh organic (residue) pool
95  sol_fon = numpy.ones((nlyrs)) # nitrogen stored in the fresh organic (residue) pool
96  sol_cov = numpy.ones((nlyrs)) # residue on soil surface
97 
98  sol_orgn = numpy.ones((nlyrs))
99  sol_no3 = numpy.zeros((nlyrs))
100  sol_aorgn = numpy.zeros((nlyrs))
101  sol_orgp = numpy.ones((nlyrs))
102  sol_solp = numpy.zeros((nlyrs))
103  sol_actp = numpy.zeros((nlyrs))
104  sol_stap = numpy.zeros((nlyrs))
105  sol_hum = numpy.zeros((nlyrs))
106  sol_rsd = numpy.zeros((nlyrs)) # Organic matter in the soil layer classified as residue, currently assigned 0. TODO
107 
108  ## Initialize some single temporary and output variables
109  cmup_kgh = 0. ## kg/ha current soil carbon for first soil layer
110  cmtot_kgh = 0. ## kg/ha current soil carbon integrated - aggregating all soil layers
111  sumno3 = 0.
112  sumorgn = 0.
113  summinp = 0.
114  sumorgp = 0.
115  ## Calculate by layer
116  ## calculate sol_cbn for lower layers if only have upper layer's data
117  if nlyrs >= 3 and cbn[2] <= 0:
118  for i in range(2,nlyrs):
119  tmpDepth = depth[i] - depth[1]
120  cbn[i] = cbn[i-1] * numpy.exp(-.001 * tmpDepth)
121 
122  for lyr in range(nlyrs):
123  sol_mass[lyr] = (sol_thick[lyr] / 1000.) * 10000. * 1000. * bd[lyr] * (1 - rock[lyr]/100.) ## soil mass of current layer (kg)
124  sol_cmass[lyr] = sol_mass[lyr] * (cbn[lyr] / 100.) ## Carbon mass of current layer (kg)
125  if lyr == 0:
126  cmup_kgh = sol_cmass[lyr]
127  cmtot_kgh = cmtot_kgh + sol_cmass[lyr]
128 
129 
130  ## calculate initial nutrient contents of layers, profile and
131  ## average in soil for the entire watershed
132  ## convert mg/kg (ppm) to kg/ha
133 
134  for j in range(nlyrs):
135  wt1 = bd[j] * sol_thick[j] / 100. ## mg/kg => kg/ha
136  conv_wt = 1.e6 * wt1 ## kg/kg => kg/ha
137 
138  sol_fop[j] = sol_rsd[j] * .0010
139  sol_fon[j] = sol_rsd[j] * .0055
140  sol_cov = sol_rsd[1]
141 
142  #if (sol_no3[j] <= 0.) :
143  zdst = math.exp(-depth[j] / 1000.)
144  sol_no3[j] = 10. * zdst * .7
145  sol_no3[j] = sol_no3[j] * wt1 ## mg/kg => kg/ha
146  sumno3 = sumno3 + sol_no3[j]
147 
148  sol_orgn[j] = sol_orgn[j] * wt1
149  if (sol_orgn[j] > 0.0001):
150  sol_orgn[j] = sol_orgn[j] * wt1 ## mg/kg => kg/ha
151  else:
152  ## assume C:N ratio of 10:1
153  sol_orgn[j] = 10000. * (cbn[j] / 14.) * wt1 ## CN ratio changed back to 14
154  sol_aorgn[j] = sol_orgn[j] * nactfr
155  sol_orgn[j] = sol_orgn[j] * (1. - nactfr)
156 
157  sumorgn = sumorgn + sol_aorgn[j] + sol_orgn[j] + sol_fon[j]
158 
159  sol_orgp[j] = sol_orgp[j] * wt1
160  sol_solp[j] = sol_solp[j] * wt1
161 
162  if (sol_orgp[j] > 0.0001):
163  sol_orgp[j] = sol_orgp[j] * wt1 ## mg/kg => kg/ha
164  else:
165  ## assume N:P ratio of 8:1
166  sol_orgp[j] = .125 * sol_orgn[j]
167  if (sol_solp[j] > 0.0001) :
168  sol_solp[j] = sol_solp[j] * wt1 ## mg/kg => kg/ha
169  else:
170  ## assume initial concentration of 5 mg/kg
171  sol_solp[j] = 5. * wt1
172 
173  solp = 0.
174  actp = 0.
175  ## Set active pool based on dynamic PSP MJW
176  if (conv_wt != 0):
177  solp = (sol_solp[j] / conv_wt) * 1000000.
178  if (clay[j] > 0.):
179  psp = -0.045 * math.log(clay[j])+ (0.001 * solp)
180  psp = psp - (0.035 * cbn[j]) + 0.43
181  else:
182  psp = 0.4
183  ## Limit PSP range
184  if (psp <.05) :
185  psp = 0.05
186  elif (psp > 0.9):
187  psp = 0.9
188 
189  sol_actp[j] = sol_solp[j] * (1. - psp) / psp
190 
191  ## Set Stable pool based on dynamic coefficant
192  if (conv_wt != 0) : ## From White et al 2009
193  ## convert to concentration for ssp calculation
194  actp = sol_actp[j] / conv_wt * 1000000.
195  solp = sol_solp[j] / conv_wt * 1000000.
196  ## estimate Total Mineral P in this soil based on data from sharpley 2004
197  ssp = 25.044 * ((actp + solp) ** -0.3833)
198  ##limit SSP Range
199  if (ssp > 7.):
200  ssp = 7.
201  if (ssp < 1.):
202  ssp = 1.
203  sol_stap[j] = ssp * (sol_actp[j] + sol_solp[j]) #define stableP
204  else:
205  ## The original code
206  sol_stap[j] = 4. * sol_actp[j]
207 
208  sol_hum[j] = cbn[j] * wt1 * 17200.
209  summinp = summinp + sol_solp[j] + sol_actp[j] + sol_stap[j]
210  sumorgp = sumorgp + sol_orgp[j] + sol_fop[j]
211 
212  return (list(sol_fop), list(sol_fon), list(sol_no3), list(sol_orgn), list(sol_aorgn), list(sol_orgp),
213  list(sol_solp), list(sol_actp), list(sol_stap), list(sol_hum), sol_cov, sumno3,
214  sumorgn, summinp, sumorgp)
