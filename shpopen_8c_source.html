00554         psSHP->fpSHP = psSHP->sHooks.FOpen(pszFullname, pszAccess );
00555     }
00556     
00557     if( psSHP->fpSHP == NULL )
00558     {
00559         char *pszMessage = (char *) malloc(strlen(pszBasename)*2+256);
00560         sprintf( pszMessage, "Unable to open %s.shp or %s.SHP.", 
00561                   pszBasename, pszBasename );

00564 
00565         free( psSHP );
00566         free( pszBasename );

00569         return NULL;

00571 
00572     sprintf( pszFullname, "%s.shx", pszBasename );
00573     psSHP->fpSHX =  psSHP->sHooks.FOpen(pszFullname, pszAccess );
00574     if( psSHP->fpSHX == NULL )
00575     {
00576         sprintf( pszFullname, "%s.SHX", pszBasename );
00577         psSHP->fpSHX = psSHP->sHooks.FOpen(pszFullname, pszAccess );
00578     }
00579     
00580     if( psSHP->fpSHX == NULL )
00581     {
00582         char *pszMessage = (char *) malloc(strlen(pszBasename)*2+256);
00583         sprintf( pszMessage, "Unable to open %s.shx or %s.SHX.", 
00584                   pszBasename, pszBasename );
00585         psHooks->Error( pszMessage );
00586         free( pszMessage );
00587 
00588         psSHP->sHooks.FClose( psSHP->fpSHP );
00589         free( psSHP );
00590         free( pszBasename );
00591         free( pszFullname );
00592         return( NULL );
00593     }
00594 
00595     free( pszFullname );
00596     free( pszBasename );
00597 
00598 /* -------------------------------------------------------------------- */
00599 /*  Read the file size from the SHP file.               */
00600 /* -------------------------------------------------------------------- */
00601     pabyBuf = (uchar *) malloc(100);
00602     psSHP->sHooks.FRead( pabyBuf, 100, 1, psSHP->fpSHP );
00603 
00604     psSHP->nFileSize = ((unsigned int)pabyBuf[24] * 256 * 256 * 256
00605                         + (unsigned int)pabyBuf[25] * 256 * 256
00606                         + (unsigned int)pabyBuf[26] * 256
00607                         + (unsigned int)pabyBuf[27]) * 2;
00608 
00609 /* -------------------------------------------------------------------- */
00610 /*  Read SHX file Header info                                           */
00611 /* -------------------------------------------------------------------- */
00612     if( psSHP->sHooks.FRead( pabyBuf, 100, 1, psSHP->fpSHX ) != 1 
00613         || pabyBuf[0] != 0 
00614         || pabyBuf[1] != 0 
00615         || pabyBuf[2] != 0x27 
00616         || (pabyBuf[3] != 0x0a && pabyBuf[3] != 0x0d) )
00617     {
00618         psSHP->sHooks.Error( ".shx file is unreadable, or corrupt." );
00619         psSHP->sHooks.FClose( psSHP->fpSHP );
00620         psSHP->sHooks.FClose( psSHP->fpSHX );
00621         free( psSHP );
00622 
00623         return( NULL );

00627         + pabyBuf[25] * 256 * 256 + pabyBuf[24] * 256 * 256 * 256;

00629 
00630     psSHP->nShapeType = pabyBuf[32];

00646         return( NULL );

00651 /* -------------------------------------------------------------------- */

00658     psSHP->adBoundsMin[1] = dValue;
00659 
00660     if( bBigEndian ) SwapWord( 8, pabyBuf+52 );
00661     memcpy( &dValue, pabyBuf+52, 8 );
00662     psSHP->adBoundsMax[0] = dValue;
00663 
00664     if( bBigEndian ) SwapWord( 8, pabyBuf+60 );
00665     memcpy( &dValue, pabyBuf+60, 8 );
00666     psSHP->adBoundsMax[1] = dValue;
00667 
00668     if( bBigEndian ) SwapWord( 8, pabyBuf+68 );     /* z */
00669     memcpy( &dValue, pabyBuf+68, 8 );
00670     psSHP->adBoundsMin[2] = dValue;
00671     
00672     if( bBigEndian ) SwapWord( 8, pabyBuf+76 );
00673     memcpy( &dValue, pabyBuf+76, 8 );
00674     psSHP->adBoundsMax[2] = dValue;
00675     
00676     if( bBigEndian ) SwapWord( 8, pabyBuf+84 );     /* z */
00677     memcpy( &dValue, pabyBuf+84, 8 );
00678     psSHP->adBoundsMin[3] = dValue;
00679 
00680     if( bBigEndian ) SwapWord( 8, pabyBuf+92 );
00681     memcpy( &dValue, pabyBuf+92, 8 );
00682     psSHP->adBoundsMax[3] = dValue;
00683 
00684     free( pabyBuf );
00685

00687 /*  Read the .shx file to get the offsets to each record in     */
00688 /*  the .shp file.                          */
00689 /* -------------------------------------------------------------------- */
00690     psSHP->nMaxRecords = psSHP->nRecords;
00691 
00692     psSHP->panRecOffset = (unsigned int *)
00693         malloc(sizeof(unsigned int) * MAX(1,psSHP->nMaxRecords) );
00694     psSHP->panRecSize = (unsigned int *)
00695         malloc(sizeof(unsigned int) * MAX(1,psSHP->nMaxRecords) );
00696     pabyBuf = (uchar *) malloc(8 * MAX(1,psSHP->nRecords) );
00697 
00698     if (psSHP->panRecOffset == NULL ||
00699         psSHP->panRecSize == NULL ||
00700         pabyBuf == NULL)
00701     {
00702         char szError[200];
00703 
00704         sprintf(szError, 
00705                 "Not enough memory to allocate requested memory (nRecords=%d).\n"
00706                 "Probably broken SHP file", 
00707                 psSHP->nRecords );
00708         psSHP->sHooks.Error( szError );
00709         psSHP->sHooks.FClose( psSHP->fpSHP );
00710         psSHP->sHooks.FClose( psSHP->fpSHX );
00711         if (psSHP->panRecOffset) free( psSHP->panRecOffset );
00712         if (psSHP->panRecSize) free( psSHP->panRecSize );
00713         if (pabyBuf) free( pabyBuf );
00714         free( psSHP );
00715         return( NULL );
00716     }
00717 
00718     if( (int) psSHP->sHooks.FRead( pabyBuf, 8, psSHP->nRecords, psSHP->fpSHX ) 
00719         != psSHP->nRecords )
00720     {
00721         char szError[200];
00722 
00723         sprintf( szError, 
00724                  "Failed to read all values for %d records in .shx file.",
00725                  psSHP->nRecords );
00726         psSHP->sHooks.Error( szError );
00727 
00728         /* SHX is short or unreadable for some reason. */
00729         psSHP->sHooks.FClose( psSHP->fpSHP );
00730         psSHP->sHooks.FClose( psSHP->fpSHX );
00731         free( psSHP->panRecOffset );
00732         free( psSHP->panRecSize );
00733         free( pabyBuf );
00734         free( psSHP );
00735 
00736         return( NULL );
00737     }
00738     
00739     /* In read-only mode, we can close the SHX now */
00740     if (strcmp(pszAccess, "rb") == 0)
00741     {
00742         psSHP->sHooks.FClose( psSHP->fpSHX );
00743         psSHP->fpSHX = NULL;
00744     }
00745 
00746     for( i = 0; i < psSHP->nRecords; i++ )
00747     {
00748         int32       nOffset, nLength;
00749 
00750         memcpy( &nOffset, pabyBuf + i * 8, 4 );
00751         if( !bBigEndian ) SwapWord( 4, &nOffset );
00752 
00753         memcpy( &nLength, pabyBuf + i * 8 + 4, 4 );
00754         if( !bBigEndian ) SwapWord( 4, &nLength );
00755 
00756         psSHP->panRecOffset[i] = nOffset*2;
00757         psSHP->panRecSize[i] = nLength*2;
00758     }
00759     free( pabyBuf );
00760 
00761     return( psSHP );
00762 }

00855 /************************************************************************/
00856 
00857 SHPHandle SHPAPI_CALL
00858 SHPCreateLL( const char * pszLayer, int nShapeType, SAHooks *psHooks )
00859 
00860 {
00861     char    *pszBasename = NULL, *pszFullname = NULL;
00862     int     i;
00863     SAFile  fpSHP = NULL, fpSHX = NULL;
00864     uchar       abyHeader[100];
00865     int32   i32;
00866     double  dValue;
00867     
00868 /* -------------------------------------------------------------------- */
00869 /*      Establish the byte order on this system.                        */
00870 /* -------------------------------------------------------------------- */
00871     i = 1;
00872     if( *((uchar *) &i) == 1 )
00873         bBigEndian = FALSE;
00874     else
00875         bBigEndian = TRUE;
00876 
00877 /* -------------------------------------------------------------------- */
00878 /*  Compute the base (layer) name.  If there is any extension   */
00879 /*  on the passed in filename we will strip it off.         */
00880 /* -------------------------------------------------------------------- */
00881     pszBasename = (char *) malloc(strlen(pszLayer)+5);
00882     strcpy( pszBasename, pszLayer );
00883     for( i = strlen(pszBasename)-1; 
00884          i > 0 && pszBasename[i] != '.' && pszBasename[i] != '/'
00885              && pszBasename[i] != '\\';
00886          i-- ) {}
00887 
00888     if( pszBasename[i] == '.' )
00889         pszBasename[i] = '\0';
00890 
00891 /* -------------------------------------------------------------------- */
00892 /*      Open the two files so we can write their headers.               */
00893 /* -------------------------------------------------------------------- */
00894     pszFullname = (char *) malloc(strlen(pszBasename) + 5);
00895     sprintf( pszFullname, "%s.shp", pszBasename );
00896     fpSHP = psHooks->FOpen(pszFullname, "wb" );
00897     if( fpSHP == NULL )
00898     {
00899         psHooks->Error( "Failed to create file .shp file." );
00900         goto error;
00901     }
00902 
00903     sprintf( pszFullname, "%s.shx", pszBasename );
00904     fpSHX = psHooks->FOpen(pszFullname, "wb" );
00905     if( fpSHX == NULL )
00906     {
00907         psHooks->Error( "Failed to create file .shx file." );
00908         goto error;
00909     }
00910 
00911     free( pszFullname ); pszFullname = NULL;
00912     free( pszBasename ); pszBasename = NULL;
00913 
00914 /* -------------------------------------------------------------------- */
00915 /*      Prepare header block for .shp file.                             */
00916 /* -------------------------------------------------------------------- */
00917     for( i = 0; i < 100; i++ )
00918         abyHeader[i] = 0;
00919 
00920     abyHeader[2] = 0x27;                /* magic cookie */
00921     abyHeader[3] = 0x0a;
00922 
00923     i32 = 50;                       /* file size */
00924     ByteCopy( &i32, abyHeader+24, 4 );
00925     if( !bBigEndian ) SwapWord( 4, abyHeader+24 );
00926     
00927     i32 = 1000;                     /* version */
00928     ByteCopy( &i32, abyHeader+28, 4 );
00929     if( bBigEndian ) SwapWord( 4, abyHeader+28 );
00930     
00931     i32 = nShapeType;                   /* shape type */
00932     ByteCopy( &i32, abyHeader+32, 4 );
00933     if( bBigEndian ) SwapWord( 4, abyHeader+32 );
00934 
00935     dValue = 0.0;                   /* set bounds */
00936     ByteCopy( &dValue, abyHeader+36, 8 );
00937     ByteCopy( &dValue, abyHeader+44, 8 );
00938     ByteCopy( &dValue, abyHeader+52, 8 );
00939     ByteCopy( &dValue, abyHeader+60, 8 );
00940 
00941 /* -------------------------------------------------------------------- */
00942 /*      Write .shp file header.                                         */
00943 /* -------------------------------------------------------------------- */
00944     if( psHooks->FWrite( abyHeader, 100, 1, fpSHP ) != 1 )
00945     {
00946         psHooks->Error( "Failed to write .shp header." );
00947         goto error;
00948     }
00949 
00950 /* -------------------------------------------------------------------- */
00951 /*      Prepare, and write .shx file header.                            */
00952 /* -------------------------------------------------------------------- */
00953     i32 = 50;                       /* file size */
00954     ByteCopy( &i32, abyHeader+24, 4 );
00955     if( !bBigEndian ) SwapWord( 4, abyHeader+24 );
00956     
00957     if( psHooks->FWrite( abyHeader, 100, 1, fpSHX ) != 1 )
00958     {
00959         psHooks->Error( "Failed to write .shx header." );
00960         goto error;
00961     }
00962 
00963 /* -------------------------------------------------------------------- */
00964 /*      Close the files, and then open them as regular existing files.  */
00965 /* -------------------------------------------------------------------- */
00966     psHooks->FClose( fpSHP );
00967     psHooks->FClose( fpSHX );
00968 
00969     return( SHPOpenLL( pszLayer, "r+b", psHooks ) );
00970 
00971 error:
00972     if (pszFullname) free(pszFullname);
00973     if (pszBasename) free(pszBasename);
00974     if (fpSHP) psHooks->FClose( fpSHP );
00975     if (fpSHX) psHooks->FClose( fpSHX );

01010 void SHPAPI_CALL

01048 SHPObject SHPAPI_CALL1(*)

01057

01059     psObject->nSHPType = nSHPType;
01060     psObject->nShapeId = nShapeId;
01061     psObject->bMeasureIsUsed = FALSE;
01062 
01063 /* -------------------------------------------------------------------- */
01064 /*  Establish whether this shape type has M, and Z values.      */
01065 /* -------------------------------------------------------------------- */

01067         || nSHPType == SHPT_POINTM
01068         || nSHPType == SHPT_POLYGONM

01071         bHasM = TRUE;

01075              || nSHPType == SHPT_POINTZ
01076              || nSHPType == SHPT_POLYGONZ
01077              || nSHPType == SHPT_MULTIPOINTZ
01078              || nSHPType == SHPT_MULTIPATCH )

01080         bHasM = TRUE;
01081         bHasZ = TRUE;

01084     {
01085         bHasM = FALSE;
01086         bHasZ = FALSE;

01091 /*      defaults to ring.                                               */
01092 /* -------------------------------------------------------------------- */
01093     if( nSHPType == SHPT_ARC || nSHPType == SHPT_POLYGON
01094         || nSHPType == SHPT_ARCM || nSHPType == SHPT_POLYGONM
01095         || nSHPType == SHPT_ARCZ || nSHPType == SHPT_POLYGONZ
01096         || nSHPType == SHPT_MULTIPATCH )

01121     }
01122

01124 /*      Capture vertices.  Note that X, Y, Z and M are optional.        */
01125 /* -------------------------------------------------------------------- */
01126     if( nVertices > 0 )

01128         psObject->padfX = (double *) calloc(sizeof(double),nVertices);
01129         psObject->padfY = (double *) calloc(sizeof(double),nVertices);
01130         psObject->padfZ = (double *) calloc(sizeof(double),nVertices);

01137             if( padfY != NULL )

01139             if( padfZ != NULL && bHasZ )

01149 /*      Compute the extents.                                            */
01150 /* -------------------------------------------------------------------- */
01151     psObject->nVertices = nVertices;
01152     SHPComputeExtents( psObject );
01153

02219 /*                          SHPRewindObject()                           */
02220 /*                                                                      */
02221 /*      Reset the winding of polygon objects to adhere to the           */
02222 /*      specification.                                                  */
02223 /************************************************************************/
02224 
02225 int SHPAPI_CALL
02226 SHPRewindObject( SHPHandle hSHP, SHPObject * psObject )
02227 
02228 {
02229     int  iOpRing, bAltered = 0;
02230 
02231 /* -------------------------------------------------------------------- */
02232 /*      Do nothing if this is not a polygon object.                     */
02233 /* -------------------------------------------------------------------- */
02234     if( psObject->nSHPType != SHPT_POLYGON
02235         && psObject->nSHPType != SHPT_POLYGONZ
02236         && psObject->nSHPType != SHPT_POLYGONM )
02237         return 0;
02238 
02239     if( psObject->nVertices == 0 || psObject->nParts == 0 )
02240         return 0;
02241 
02242 /* -------------------------------------------------------------------- */
02243 /*      Process each of the rings.                                      */
02244 /* -------------------------------------------------------------------- */
02245     for( iOpRing = 0; iOpRing < psObject->nParts; iOpRing++ )
02246     {
02247         int      bInner, iVert, nVertCount, nVertStart, iCheckRing;
02248         double   dfSum, dfTestX, dfTestY;
02249 
02250 /* -------------------------------------------------------------------- */
02251 /*      Determine if this ring is an inner ring or an outer ring        */
02252 /*      relative to all the other rings.  For now we assume the         */
02253 /*      first ring is outer and all others are inner, but eventually    */
02254 /*      we need to fix this to handle multiple island polygons and      */
02255 /*      unordered sets of rings.                                        */
02256 /*                                                                      */
02257 /* -------------------------------------------------------------------- */
02258 
02259         /* Use point in the middle of segment to avoid testing
02260          * common points of rings.
02261          */
02262         dfTestX = ( psObject->padfX[psObject->panPartStart[iOpRing]]
02263                     + psObject->padfX[psObject->panPartStart[iOpRing] + 1] ) / 2;
02264         dfTestY = ( psObject->padfY[psObject->panPartStart[iOpRing]]
02265                     + psObject->padfY[psObject->panPartStart[iOpRing] + 1] ) / 2;
02266 
02267         bInner = FALSE;
02268         for( iCheckRing = 0; iCheckRing < psObject->nParts; iCheckRing++ )
02269         {
02270             int iEdge;
02271 
02272             if( iCheckRing == iOpRing )
02273                 continue;
02274             
02275             nVertStart = psObject->panPartStart[iCheckRing];
02276 
02277             if( iCheckRing == psObject->nParts-1 )
02278                 nVertCount = psObject->nVertices 
02279                     - psObject->panPartStart[iCheckRing];
02280             else
02281                 nVertCount = psObject->panPartStart[iCheckRing+1] 
02282                     - psObject->panPartStart[iCheckRing];
02283 
02284             for( iEdge = 0; iEdge < nVertCount; iEdge++ )
02285             {
02286                 int iNext;
02287 
02288                 if( iEdge < nVertCount-1 )
02289                     iNext = iEdge+1;
02290                 else
02291                     iNext = 0;
02292 
02293                 /* Rule #1:
02294                  * Test whether the edge 'straddles' the horizontal ray from the test point (dfTestY,dfTestY)
02295                  * The rule #1 also excludes edges collinear with the ray.
02296                  */
02297                 if ( ( psObject->padfY[iEdge+nVertStart] < dfTestY
02298                        && dfTestY <= psObject->padfY[iNext+nVertStart] )
02299                      || ( psObject->padfY[iNext+nVertStart] < dfTestY
02300                           && dfTestY <= psObject->padfY[iEdge+nVertStart] ) )
02301                 {
02302                     /* Rule #2:
02303                      * Test if edge-ray intersection is on the right from the test point (dfTestY,dfTestY)
02304                      */
02305                     double const intersect = 
02306                         ( psObject->padfX[iEdge+nVertStart]
02307                           + ( dfTestY - psObject->padfY[iEdge+nVertStart] ) 
02308                           / ( psObject->padfY[iNext+nVertStart] - psObject->padfY[iEdge+nVertStart] )
02309                           * ( psObject->padfX[iNext+nVertStart] - psObject->padfX[iEdge+nVertStart] ) );
02310
