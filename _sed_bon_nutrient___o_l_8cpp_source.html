00202 void SedBonNutrient_OL::concentration(int id)
00203 {
00204     float Fcl = m_clayFrac[id];
00205     float K_NH4 = 1.34f + 0.038f*Fcl;    // partition coefficient
00206     float K_P = 100 + 2.5f * Fcl;     // partition coefficient
00207     // sendiment bound nutrient concentration
00208     //concentration at the end of the time step 
00209     float Q = m_CELLQ[id];   //m3/s
00210     float h = m_CELLH[id] * m_cellA[id]/1000;  //m3
00211     float f = m_infil[id] * m_cellA[id]/1000;  //m3
00212     float S = m_sd[id] * m_cellA[id]/1000;     //m3
00213     float p = m_Porosity[id];
00214     if(S!=0)
00215     {
00216         if (h==0)
00217         {
00218             h = m_whtoCh[id];
00219         }
00220         m_X2 = (h + f) / S;
00221         m_X3 = h / S;
00222     }
00223     else
00224     {
00225         m_X2 = 0;
00226         m_X3 = 0;
00227     }
00228     float temp1 = Q * m_dt + f + S;    //m3
00229     float temp2 = 2.65f*(1-p);
00230     float X_NH4 = temp1 /(p + temp2 * K_NH4);
00231     float Cfin_NH4 = m_PreConcNH4[id] * exp(-X_NH4);   //kg/kg-soil
00232 
00233     float X_P = temp1 /(p + temp2 * K_P);
00234     float Cfin_P = m_PreConcP[id] * exp(-X_P);    //kg/kg-soil
00235 
00236     float V = m_soilMoisture[id] * 1000;    //kg/m3
00237     float M = m_Density[id];    //kg/m3
00238     float P0_NH4 = K_NH4 * M * Cfin_NH4 / V;    //kg/kg-soil     
00239     float P0_P = K_NH4 * M * Cfin_NH4 / V;    //kg/kg-soil
00240     // newly generated sediment-bound nutrient (kg/s)
00241     float Sed = m_SedKG[id] / m_dt;   //newly generated sediment (kg/s)
00242     m_CP_Sed[id] = P0_P * Sed;    
00243     m_CNH4_Sed[id] = P0_NH4 * Sed;   
00244 }

00248 {
00249     const float beta = 0.6f;
00250     const float _23 = 2.0f/3;
00251     float Perim,R,S,n;
00252 
00253     for (int i=0; i<m_size; i++)
00254     {
00255         if (m_whtoCh[i] > 0.0001f)
00256         {
00257             Perim = 2 * m_whtoCh[i]/1000 + m_chWidth[i];    // mm to m -> /1000 
00258             if (Perim > 0)
00259                 R = m_whtoCh[i]/1000 * m_chWidth[i] / Perim;
00260             else
00261                 R = 0.0f;
00262             S = sin(atan(m_Slope[i]));   //sine of the slope
00263             S = max(0.001f, S);
00264             n = m_ManningN[i];  //manning n
00265             m_ChV[i] = pow(R, _23) * sqrt(S)/n;
00266         }
00267         else
00268             m_ChV[i] = 0;
00269     }

00272 void SedBonNutrient_OL::OverlandFlow(int id)
00273 {
00274     concentration(id);
00275     //sum the upstream overland flow
00276     float flowwidth = m_flowWidth[id];
00277     float PUp = 0.0f;
00278     float NH4Up = 0.0f;
00279     for (int k = 1; k <= (int)m_flowInIndex[id][0]; ++k)
00280     {
00281         int flowInID = (int)m_flowInIndex[id][k];
00282         PUp  += m_CP_Sed[flowInID];
00283         NH4Up  += m_CNH4_Sed[flowInID];
00284     }
00285     // if the channel width is greater than the cell width
00286     if (m_streamLink[id] >= 0 && flowwidth <=0)
00287     {
00288         m_SedBonPToCh[id] = PUp * m_dt;  // convert to kg
00289         m_SedBonAmmoniumToCh[id] = NH4Up * m_dt;
00290         return;
00291     }
00292     
00293     //An initial test is done on the discharge at the end of the time interval.  When the discharge is equal to zero
00294     if (m_CP_Sed[id]==0)
00295     {
00296     float Pinflow = PUp;
00297     float NH4inflow = NH4Up ;
00298     //The amount of sediment bound nutrient in transit 
00299     float Ptransit = (m_PPT_P[id] + Pinflow)/2;
00300     float NH4transit = (m_PPT_NH4[id] + NH4inflow)/2;
00301     //The outflow of sediment bound nutrient  kg/s
00302     m_SedBonP[id] = 0;
00303     m_SedBonAmmonium[id] = 0;
00304     //The new value of PPT
00305     m_PPT_P[id] = Pinflow;
00306     m_PPT_NH4[id] = NH4inflow;
00307     }
00308     //When the discharge is positive
00309     else
00310     {
00311     float Pinflow = PUp + m_CP_Sed[id];
00312     float NH4inflow = NH4Up + m_CNH4_Sed[id];
00313     //The amount of sediment bound nutrient in transit 
00314     float Ptransit = (m_PPT_P[id] + Pinflow)/(1 + m_X1);
00315     float NH4transit = (m_PPT_NH4[id] + NH4inflow)/(1 + m_X1);
00316     //The outflow of sediment bound nutrient  kg/s
00317     float Q = m_CELLQ[id];   //m3/s
00318     float S = m_sd[id] * m_cellA[id]/1000;     //m3
00319     m_SedBonP[id] = (Ptransit * Q) / S;   
00320     m_SedBonAmmonium[id] = (NH4transit * Q) / S;
00321     //The new value of PPT
00322     float ratio = max(0.0f, 1-m_X1);
00323     m_PPT_P[id] = Pinflow + m_SedBonP[id]*ratio;
00324     m_PPT_NH4[id] = NH4inflow + m_SedBonAmmonium[id]*ratio;
00325     }
00326     // compute nutrients to channel 
00327     float fractiontochannel = 0.0f;
00328     if (m_chWidth[id] > 0)
00329     {
00330         float tem = m_ChV[id] * m_dt;
00331         fractiontochannel = 2*tem / (m_dx - m_chWidth[id]);
00332         fractiontochannel = min(fractiontochannel, 1.0f);
00333         // convert to kg
00334         m_SedBonPToCh[id] = m_SedBonP[id] * m_dt * fractiontochannel;
00335         m_SedBonAmmoniumToCh[id] = m_SedBonAmmonium[id] * m_dt * fractiontochannel;
00336     }
00337 }
