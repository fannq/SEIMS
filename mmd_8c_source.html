00025 /*************************************************************************
00026 *  genmmd  -- multiple minimum external degree
00027 *  purpose -- this routine implements the minimum degree
00028 *     algorithm. it makes use of the implicit representation
00029 *     of elimination graphs by quotient graphs, and the notion
00030 *     of indistinguishable nodes. It also implements the modifications
00031 *     by multiple elimination and minimum external degree.
00032 *     Caution -- the adjacency vector adjncy will be destroyed.
00033 *  Input parameters --
00034 *     neqns -- number of equations.
00035 *     (xadj, adjncy) -- the adjacency structure.
00036 *     delta  -- tolerance value for multiple elimination.
00037 *     maxint -- maximum machine representable (short) integer
00038 *               (any smaller estimate will do) for marking nodes.
00039 *  Output parameters --
00040 *     perm -- the minimum degree ordering.
00041 *     invp -- the inverse of perm.
00042 *     *ncsub -- an upper bound on the number of nonzero subscripts
00043 *               for the compressed storage scheme.
00044 *  Working parameters --
00045 *     head -- vector for head of degree lists.
00046 *     invp  -- used temporarily for degree forward link.
00047 *     perm  -- used temporarily for degree backward link.
00048 *     qsize -- vector for size of supernodes.
00049 *     list -- vector for temporary linked lists.
00050 *     marker -- a temporary marker vector.
00051 *  Subroutines used -- mmdelm, mmdint, mmdnum, mmdupd.
00052 **************************************************************************/
00053 void genmmd(idx_t neqns, idx_t *xadj, idx_t *adjncy, idx_t *invp, idx_t *perm,
00054      idx_t delta, idx_t *head, idx_t *qsize, idx_t *list, idx_t *marker,
00055      idx_t maxint, idx_t *ncsub)
00056 {
00057     idx_t  ehead, i, mdeg, mdlmt, mdeg_node, nextmd, num, tag;
00058 
00059     if (neqns <= 0)  
00060       return;
00061 
00062     /* Adjust from C to Fortran */
00063     xadj--; adjncy--; invp--; perm--; head--; qsize--; list--; marker--;
00064 
00065     /* initialization for the minimum degree algorithm. */
00066     *ncsub = 0;
00067     mmdint(neqns, xadj, adjncy, head, invp, perm, qsize, list, marker);
00068 
00069     /*  'num' counts the number of ordered nodes plus 1. */
00070     num = 1;
00071 
00072     /* eliminate all isolated nodes. */
00073     nextmd = head[1];
00074     while (nextmd > 0) {
00075       mdeg_node = nextmd;
00076       nextmd = invp[mdeg_node];
00077       marker[mdeg_node] = maxint;
00078       invp[mdeg_node] = -num;
00079       num = num + 1;
00080     }
00081 
00082     /* search for node of the minimum degree. 'mdeg' is the current */
00083     /* minimum degree; 'tag' is used to facilitate marking nodes.   */
00084     if (num > neqns) 
00085       goto n1000;
00086     tag = 1;
00087     head[1] = 0;
00088     mdeg = 2;
00089 
00090     /* infinite loop here ! */
00091     while (1) {
00092       while (head[mdeg] <= 0) 
00093         mdeg++;
00094 
00095       /* use value of 'delta' to set up 'mdlmt', which governs */
00096       /* when a degree update is to be performed.              */
00097       mdlmt = mdeg + delta;
00098       ehead = 0;
00099 
00100 n500:
00101       mdeg_node = head[mdeg];
00102       while (mdeg_node <= 0) {
00103         mdeg++;
00104 
00105         if (mdeg > mdlmt) 
00106           goto n900;
00107         mdeg_node = head[mdeg];
00108       };
00109 
00110       /*  remove 'mdeg_node' from the degree structure. */
00111       nextmd = invp[mdeg_node];
00112       head[mdeg] = nextmd;
00113       if (nextmd > 0)  
00114         perm[nextmd] = -mdeg;
00115       invp[mdeg_node] = -num;
00116       *ncsub += mdeg + qsize[mdeg_node] - 2;
00117       if ((num+qsize[mdeg_node]) > neqns)  
00118         goto n1000;
00119 
00120       /*  eliminate 'mdeg_node' and perform quotient graph */
00121       /*  transformation. reset 'tag' value if necessary.    */
00122       tag++;
00123       if (tag >= maxint) {
00124         tag = 1;
00125         for (i = 1; i <= neqns; i++)
00126           if (marker[i] < maxint)  
00127             marker[i] = 0;
00128       };
00129 
00130       mmdelm(mdeg_node, xadj, adjncy, head, invp, perm, qsize, list, marker, maxint, tag);
00131 
00132       num += qsize[mdeg_node];
00133       list[mdeg_node] = ehead;
00134       ehead = mdeg_node;
00135       if (delta >= 0) 
00136         goto n500;
00137 
00138  n900:
00139       /* update degrees of the nodes involved in the  */
00140       /* minimum degree nodes elimination.            */
00141       if (num > neqns)  
00142         goto n1000;
00143       mmdupd( ehead, neqns, xadj, adjncy, delta, &mdeg, head, invp, perm, qsize, list, marker, maxint, &tag);
00144     }; /* end of -- while ( 1 ) -- */
00145 
00146 n1000:
00147     mmdnum( neqns, perm, invp, qsize );
00148 
00149     /* Adjust from Fortran back to C*/
00150     xadj++; adjncy++; invp++; perm++; head++; qsize++; list++; marker++;
00151 }

00155 *           mmdelm ...... multiple minimum degree elimination
00156 * Purpose -- This routine eliminates the node mdeg_node of minimum degree
00157 *     from the adjacency structure, which is stored in the quotient
00158 *     graph format. It also transforms the quotient graph representation
00159 *     of the elimination graph.
00160 * Input parameters --
00161 *     mdeg_node -- node of minimum degree.
00162 *     maxint -- estimate of maximum representable (short) integer.
00163 *     tag    -- tag value.
00164 * Updated parameters --
00165 *     (xadj, adjncy) -- updated adjacency structure.
00166 *     (head, forward, backward) -- degree doubly linked structure.
00167 *     qsize -- size of supernode.
00168 *     marker -- marker vector.
00169 *     list -- temporary linked list of eliminated nabors.

00172      idx_t *backward, idx_t *qsize, idx_t *list, idx_t *marker, idx_t maxint, idx_t tag)
00173 {
00174     idx_t   element, i,   istop, istart, j,
00175           jstop, jstart, link,
00176           nabor, node, npv, nqnbrs, nxnode,
00177           pvnode, rlmt, rloc, rnode, xqnbr;
00178 
00179     /* find the reachable set of 'mdeg_node' and */
00180     /* place it in the data structure.           */
00181     marker[mdeg_node] = tag;
00182     istart = xadj[mdeg_node];
00183     istop = xadj[mdeg_node+1] - 1;
00184 
00185     /* 'element' points to the beginning of the list of  */
00186     /* eliminated nabors of 'mdeg_node', and 'rloc' gives the */
00187     /* storage location for the next reachable node.   */
00188     element = 0;
00189     rloc = istart;
00190     rlmt = istop;
00191     for ( i = istart; i <= istop; i++ ) {
00192         nabor = adjncy[i];
00193         if ( nabor == 0 ) break;
00194         if ( marker[nabor] < tag ) {
00195            marker[nabor] = tag;
00196            if ( forward[nabor] < 0 )  {
00197               list[nabor] = element;
00198               element = nabor;
00199            } else {
00200               adjncy[rloc] = nabor;
00201               rloc++;
00202            };
00203         }; /* end of -- if -- */
00204     }; /* end of -- for -- */
00205 
00206   /* merge with reachable nodes from generalized elements. */
00207   while ( element > 0 ) {
00208       adjncy[rlmt] = -element;
00209       link = element;
00210 
00211 n400:
00212       jstart = xadj[link];
00213       jstop = xadj[link+1] - 1;
00214       for ( j = jstart; j <= jstop; j++ ) {
00215           node = adjncy[j];
00216           link = -node;
00217           if ( node < 0 )  goto n400;
00218           if ( node == 0 ) break;
00219           if ((marker[node]<tag)&&(forward[node]>=0)) {
00220              marker[node] = tag;
00221              /*use storage from eliminated nodes if necessary.*/
00222              while ( rloc >= rlmt ) {
00223                    link = -adjncy[rlmt];
00224                    rloc = xadj[link];
00225                    rlmt = xadj[link+1] - 1;
00226              };
00227              adjncy[rloc] = node;
00228              rloc++;
00229           };
00230       }; /* end of -- for ( j = jstart; -- */
00231       element = list[element];
00232     };  /* end of -- while ( element > 0 ) -- */
00233     if ( rloc <= rlmt ) adjncy[rloc] = 0;
00234     /* for each node in the reachable set, do the following. */
00235     link = mdeg_node;
00236 
00237 n1100:
00238     istart = xadj[link];
00239     istop = xadj[link+1] - 1;
00240     for ( i = istart; i <= istop; i++ ) {
00241         rnode = adjncy[i];
00242         link = -rnode;
00243         if ( rnode < 0 ) goto n1100;
00244         if ( rnode == 0 ) return;
00245 
00246         /* 'rnode' is in the degree list structure. */
00247         pvnode = backward[rnode];
00248         if (( pvnode != 0 ) && ( pvnode != (-maxint) )) {
00249            /* then remove 'rnode' from the structure. */
00250            nxnode = forward[rnode];
00251            if ( nxnode > 0 ) backward[nxnode] = pvnode;
00252            if ( pvnode > 0 ) forward[pvnode] = nxnode;
00253            npv = -pvnode;
00254            if ( pvnode < 0 ) head[npv] = nxnode;
00255         };
00256 
00257         /* purge inactive quotient nabors of 'rnode'. */
00258         jstart = xadj[rnode];
00259         jstop = xadj[rnode+1] - 1;
00260         xqnbr = jstart;
00261         for ( j = jstart; j <= jstop; j++ ) {
00262             nabor = adjncy[j];
00263             if ( nabor == 0 ) break;
00264             if ( marker[nabor] < tag ) {
00265                 adjncy[xqnbr] = nabor;
00266                 xqnbr++;
00267             };
00268         };
00269 
00270         /* no active nabor after the purging. */
00271         nqnbrs = xqnbr - jstart;
00272         if ( nqnbrs <= 0 ) {
00273            /* merge 'rnode' with 'mdeg_node'. */
00274            qsize[mdeg_node] += qsize[rnode];
00275            qsize[rnode] = 0;
00276            marker[rnode] = maxint;
00277            forward[rnode] = -mdeg_node;
00278            backward[rnode] = -maxint;
00279         } else {
00280            /* flag 'rnode' for degree update, and  */
00281            /* add 'mdeg_node' as a nabor of 'rnode'.      */
00282            forward[rnode] = nqnbrs + 1;
00283            backward[rnode] = 0;
00284            adjncy[xqnbr] = mdeg_node;
00285            xqnbr++;
00286            if ( xqnbr <= jstop )  adjncy[xqnbr] = 0;
00287         };
00288       }; /* end of -- for ( i = istart; -- */
00289       return;
00290  }

00292 /***************************************************************************
00293 *    mmdint ---- mult minimum degree initialization
00294 *    purpose -- this routine performs initialization for the
00295 *       multiple elimination version of the minimum degree algorithm.
00296 *    input parameters --
00297 *       neqns  -- number of equations.
00298 *       (xadj, adjncy) -- adjacency structure.
00299 *    output parameters --
00300 *       (head, dfrow, backward) -- degree doubly linked structure.
00301 *       qsize -- size of supernode ( initialized to one).
00302 *       list -- linked list.
00303 *       marker -- marker vector.
00304 ****************************************************************************/
00305 idx_t  mmdint(idx_t neqns, idx_t *xadj, idx_t *adjncy, idx_t *head, idx_t *forward,
00306      idx_t *backward, idx_t *qsize, idx_t *list, idx_t *marker)
00307 {
00308     idx_t  fnode, ndeg, node;
00309 
00310     for ( node = 1; node <= neqns; node++ ) {
00311         head[node] = 0;
00312         qsize[node] = 1;
00313         marker[node] = 0;
00314         list[node] = 0;
00315     };
00316 
00317     /* initialize the degree doubly linked lists. */
00318     for ( node = 1; node <= neqns; node++ ) {
00319         ndeg = xadj[node+1] - xadj[node]/* + 1*/;   /* george */
00320         if (ndeg == 0)
00321           ndeg = 1;
00322         fnode = head[ndeg];
00323         forward[node] = fnode;
00324         head[ndeg] = node;
00325         if ( fnode > 0 ) backward[fnode] = node;
00326         backward[node] = -ndeg;
00327     };
00328     return 0;
00329 }

00331 /****************************************************************************
00332 * mmdnum --- multi minimum degree numbering
00333 * purpose -- this routine performs the final step in producing
00334 *    the permutation and inverse permutation vectors in the
00335 *    multiple elimination version of the minimum degree
00336 *    ordering algorithm.
00337 * input parameters --
00338 *     neqns -- number of equations.
00339 *     qsize -- size of supernodes at elimination.
00340 * updated parameters --
00341 *     invp -- inverse permutation vector. on input,
00342 *             if qsize[node] = 0, then node has been merged
00343 *             into the node -invp[node]; otherwise,
00344 *            -invp[node] is its inverse labelling.
00345 * output parameters --
00346 *     perm -- the permutation vector.
00347 ****************************************************************************/
00348 void mmdnum(idx_t neqns, idx_t *perm, idx_t *invp, idx_t *qsize)
00349 {
00350   idx_t father, nextf, node, nqsize, num, root;
00351 
00352   for ( node = 1; node <= neqns; node++ ) {
00353       nqsize = qsize[node];
00354       if ( nqsize <= 0 ) perm[node] = invp[node];
00355       if ( nqsize > 0 )  perm[node] = -invp[node];
00356   };
00357 
00358   /* for each node which has been merged, do the following. */
00359   for ( node = 1; node <= neqns; node++ ) {
00360       if ( perm[node] <= 0 )  {
00361 
00362      /* trace the merged tree until one which has not */
00363          /* been merged, call it root.                    */
00364          father = node;
00365          while ( perm[father] <= 0 )
00366             father = - perm[father];
00367 
00368          /* number node after root. */
00369          root = father;
00370          num = perm[root] + 1;
00371          invp[node] = -num;
00372          perm[root] = num;
00373 
00374          /* shorten the merged tree. */
00375          father = node;
00376          nextf = - perm[father];
00377          while ( nextf > 0 ) {
00378             perm[father] = -root;
00379             father = nextf;
00380             nextf = -perm[father];
00381          };
00382       };  /* end of -- if ( perm[node] <= 0 ) -- */
00383   }; /* end of -- for ( node = 1; -- */
00384 
00385   /* ready to compute perm. */
00386   for ( node = 1; node <= neqns; node++ ) {
00387         num = -invp[node];
00388         invp[node] = num;
00389         perm[num] = node;
00390   };
00391   return;
00392 }
