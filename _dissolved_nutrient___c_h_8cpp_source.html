00410 {
00411     float Qsn = 0;
00412     float totsed = sed + Sin*dt;  // add upstream sed to sed present in cell
00413     float totwater = vol + Qin*dt;   // add upstream water to volume water in cell

00421 void DissolvedNutrient_CH::GetNutrientInFlow(int iReach, int iCell, int id)
00422 {
00423     //sum the upstream overland flow
00424     float Qin = 0.f;
00425     float PUp = 0.0f;
00426     float NH4Up = 0.0f;
00427     float NO3Up = 0.0f; 
00428     if (iCell == 0)// inflow of this cell is the last cell of the upstream reach
00429     {
00430         for (int i = 0; i < m_reachUpStream[iReach].size(); ++i)
00431         {
00432             int upReachId = m_reachUpStream[iReach][i];
00433             if (upReachId >= 0)
00434             {
00435                 int upCellsNum = m_reachs[upReachId].size();
00436                 int upCellId = m_reachs[upReachId][upCellsNum-1];
00437                 Qin += m_ChQkin[upReachId][upCellsNum-1];
00438                 PUp += m_DissovP[upReachId][upCellsNum-1];

00440                 NO3Up += m_Nitrate[upReachId][upCellsNum-1];
00441             }
00442         }
00443     }
00444     else
00445     {
00446         Qin = m_ChQkin[iReach][iCell-1]; 
00447         PUp = m_DissovP[iReach][iCell-1];   //the iCell-1 is the prior of the iCell
00448         NH4Up = m_Ammonium[iReach][iCell-1];
00449         NO3Up = m_Nitrate[iReach][iCell-1];
00450     }
00451     
00452     this->WaterVolumeCalc(iReach, iCell, id);
00453     float WtVol = m_ChVol[iReach][iCell];
00454

00456     m_Ammonium_KG[iReach][iCell] = m_AmmoniumToCh[id];

00464     // no more dissovP outflow than total dissovP in cell
00465     m_DissovP[iReach][iCell] = min(m_DissovP[iReach][iCell], tem);

00467     tem = PUp * m_dt + m_DissovP_KG[iReach][iCell] - m_DissovP[iReach][iCell] * m_dt;
00468     // new dissovP volume based on all fluxes or dissovP present

00470 
00471     tem = NH4Up + m_Ammonium_KG[iReach][iCell]/m_dt;   
00472     m_Ammonium[iReach][iCell] = min(m_Ammonium[iReach][iCell], tem);
