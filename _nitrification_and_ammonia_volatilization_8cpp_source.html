00283 int NitrificationAndAmmoniaVolatilization::Execute()
00284 {
00285     //check the data
00286     CheckInputData();   
00287 
00288     initalOutputs();
00289 
00290     #pragma omp parallel for
00291     for(int i=0; i < m_nLayers; i++)
00292     {
00293         for(int j=0; j < m_size; j++)
00294         {
00295             if (m_SoilT[j] < 5)
00296             continue;
00297 
00298             m_nitvolTF[i][j] = 0.41f * (m_SoilT[j] - 5) / 10;
00299          
00300             int hydroIndex = max(0, i-1);
00301             float r = 0.25f * m_FieldCap[hydroIndex][j] - 0.75f * m_Wiltingpoint[hydroIndex][j];
00302             if (m_SOMO[hydroIndex][j] < r)
00303                 m_nitWF[i][j] = (m_SOMO[hydroIndex][j]- m_Wiltingpoint[hydroIndex][j]) / (0.25f * (m_FieldCap[hydroIndex][j] - m_Wiltingpoint[hydroIndex][j]));
00304             else
00305                 m_nitWF[i][j] = 1.0f;
00306          
00307             m_midZ[0][j] = m_Depth[0][j] / 2;
00308             m_midZ[1][j] = (m_Depth[1][j] - m_Depth[0][j]) / 2;
00309             m_midZ[2][j] = (m_Depth[2][j] - m_Depth[1][j]) / 2;
00310             
00311             m_volDF[i][j] = 1 - m_midZ[i][j] / (m_midZ[i][j] + exp(4.706f - 0.0305f * m_midZ[i][j]));
00312             float m_catEF = 0.15f;
00313             m_nitR[i][j] = m_nitvolTF[i][j] * m_nitWF[i][j];
00314             m_volR[i][j] = m_nitvolTF[i][j] * m_volDF[i][j] * m_catEF;
00315             m_nitvolAmmon[i][j] = m_Ammon[i][j] * (1 - exp(- m_nitR[i][j] - m_volR[i][j]));
00316             m_nitLostF[i][j] = 1 - exp(- m_nitR[i][j]);
00317             m_volLostF[i][j] = 1 - exp(- m_volR[i][j]);
00318             m_conNitra[i][j] = m_nitvolAmmon[i][j] * m_nitLostF[i][j] / (m_nitLostF[i][j] + m_volLostF[i][j]);
00319             m_conAmm[i][j] = m_nitvolAmmon[i][j] * m_volLostF[i][j] / (m_nitLostF[i][j] + m_volLostF[i][j]);
00320 
00321             m_Nitrate[i][j] += m_conNitra[i][j];
00322             m_Ammon[i][j] -= m_nitvolAmmon[i][j]; 
00323         }
00324     }
00325 
00326     return 0;
