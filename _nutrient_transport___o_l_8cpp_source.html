00658     initalOutputs();
00659 
00660     int area = m_cellWidth * m_cellWidth / 10000;
00661 
00662     // calculate amount of nutrient for transpotation in each cell 
00663     #pragma omp parallel for
00664     for(int i=0; i < m_nLayers; i++)
00665     {
00666         for(int j=0; j < m_size; j++)
00667         { 
00668             int hydroIndex = max(0, i-1);
00669             float sr = 0.f;
00670             if (i == 0)
00671                 sr = m_sr[j];
00672             else
00673                 sr = 0.f;
00674             
00675             m_mobQ[i][j] = sr + m_ssr[hydroIndex][j] + m_perc[hydroIndex][j] + 1.e-10;
00676 
00677             m_satW[i][j] = m_porosity[hydroIndex][j] * m_Depth[i][j];
00678             float r = (-m_mobQ[i][j]) / ((1-m_aniExcF) * m_satW[i][j]);
00679             float conNit = 0.f;
00680             conNit = m_Nitrate[i][j] * (1- exp(r)) / m_mobQ[i][j];
00681             m_conNit[i][j] = max(conNit,0.f); 
00682             float surNit = 0.f;
00683             surNit = m_nitPC * m_conNit[0][j] * m_sr[j];
00684             m_surNit[j] = min(surNit, m_Nitrate[0][j]);
00685             m_Nitrate[0][j] -= m_surNit[j];
00686 
00687             float nitPC = 0.f;
00688             if (i == 0)
00689                 nitPC = m_nitPC;
00690             else
00691                 nitPC = 1.f;
00692             float latNit = m_nitPC * m_conNit[i][j] * m_ssr[hydroIndex][j];

00699             if(i < m_nLayers-1)
00700                 m_Nitrate[i+1][j] += m_perNit[i][j];
00701 
00702             float K_NH4 = 0.083 * m_clayFrac[0][j] + 1.34f;   
00703             float pk = m_porosity[hydroIndex][j] + 2.65f * (1 - m_porosity[hydroIndex][j]) * K_NH4;
00704             float c_Amm = m_Ammon[i][j] * exp( - m_mobQ[i][j] * area * 0.001 / pk );

00708                 coe_extra = 0.5f;
00709             else if(K_NH4>1.f && K_NH4 <=10.f)
00710                 coe_extra = 0.598f * exp(-0.179f * K_NH4);
00711             else
00712                 coe_extra = 0.1f;
00713             float conAmm =  c_Amm * coe_extra / ((1 + K_NH4 * coe_extra) * m_mobQ[i][j]);
00714             m_conAmm[i][j] = max(conAmm, 0.f);
00715             float surAmm = m_conAmm[0][j] * m_sr[j];
00716             m_surAmm[j] = min(surAmm, m_Ammon[0][j]);
00717             m_Ammon[0][j] -= m_surAmm[j];
00718 
00719             if ( m_sr[j] > 0.f)
00720                 m_conSed[j] = m_sedimentYield[j] / (10.f * area * m_sr[j]);
